31class MailThread(models.AbstractModel): 

32 ''' mail_thread model is meant to be inherited by any model that needs to 

33 act as a discussion topic on which messages can be attached. Public 

34 methods are prefixed with ``message_`` in order to avoid name 

35 collisions with methods of the models that will inherit from this class. 

36 

37 ``mail.thread`` defines fields used to handle and display the 

38 communication history. ``mail.thread`` also manages followers of 

39 inheriting classes. All features and expected behavior are managed 

40 by mail.thread. Widgets has been designed for the 7.0 and following 

41 versions of OpenERP. 

42 

43 Inheriting classes are not required to implement any method, as the 

44 default implementation will work for any model. However it is common 

45 to override at least the ``message_new`` and ``message_update`` 

46 methods (calling ``super``) to add model-specific behavior at 

47 creation and update of a thread when processing incoming emails. 

48 

49 Options: 

50 - _mail_flat_thread: if set to True, all messages without parent_id 

51 are automatically attached to the first message posted on the 

52 ressource. If set to False, the display of Chatter is done using 

53 threads, and no parent_id is automatically set. 

54 

55 MailThread features can be somewhat controlled through context keys : 

56 

57 - ``mail_create_nosubscribe``: at create or message_post, do not subscribe 

58 uid to the record thread 

59 - ``mail_create_nolog``: at create, do not log the automatic '<Document> 

60 created' message 

61 - ``mail_notrack``: at create and write, do not perform the value tracking 

62 creating messages 

63 - ``tracking_disable``: at create and write, perform no MailThread features 

64 (auto subscription, tracking, post, ...) 

65 - ``mail_save_message_last_post``: at message_post, update message_last_post 

66 datetime field 

67 - ``mail_auto_delete``: auto delete mail notifications; True by default 

68 (technical hack for templates) 

69 - ``mail_notify_force_send``: if less than 50 email notifications to send, 

70 send them directly instead of using the queue; True by default 

71 - ``mail_notify_user_signature``: add the current user signature in 

72 email notifications; True by default 

73 '''

978 @api.model 

979 def message_route(self, message, message_dict, model=None, thread_id=None, custom_values=None): 

980 """ Attempt to figure out the correct target model, thread_id, 

981 custom_values and user_id to use for an incoming message. 

982 Multiple values may be returned, if a message had multiple 

983 recipients matching existing mail.aliases, for example. 

984 

985 The following heuristics are used, in this order: 

986 

987 * if the message replies to an existing thread by having a Message-Id 

988 that matches an existing mail_message.message_id, we take the original 

989 message model/thread_id pair and ignore custom_value as no creation will 

990 take place 

991 * if the message replies to an existing thread by having In-Reply-To or 

992 References matching odoo model/thread_id Message-Id and if this thread 

993 has messages without message_id, take this model/thread_id pair and 

994 ignore custom_value as no creation will take place (6.1 compatibility) 

995 * look for a mail.alias entry matching the message recipients and use the 

996 corresponding model, thread_id, custom_values and user_id. This could 

997 lead to a thread update or creation depending on the alias 

998 * fallback on provided ``model``, ``thread_id`` and ``custom_values`` 

999 * raise an exception as no route has been found 

1000 

1001 :param string message: an email.message instance 

1002 :param dict message_dict: dictionary holding parsed message variables 

1003 :param string model: the fallback model to use if the message does not match 

1004 any of the currently configured mail aliases (may be None if a matching 

1005 alias is supposed to be present) 

1006 :type dict custom_values: optional dictionary of default field values 

1007 to pass to ``message_new`` if a new record needs to be created. 

1008 Ignored if the thread record already exists, and also if a matching 

1009 mail.alias was found (aliases define their own defaults) 

1010 :param int thread_id: optional ID of the record/thread from ``model`` to 

1011 which this mail should be attached. Only used if the message does not 

1012 reply to an existing thread and does not match any mail alias. 

1013 :return: list of routes [(model, thread_id, custom_values, user_id, alias)] 

1014 

1015 :raises: ValueError, TypeError 

1016 """ 

1017 if not isinstance(message, Message): 

1018 raise TypeError('message must be an email.message.Message at this point') 

1019 MailMessage = self.env['mail.message'] 

1020 Alias, dest_aliases = self.env['mail.alias'], self.env['mail.alias'] 

1021 bounce_alias = self.env['ir.config_parameter'].get_param("mail.bounce.alias") 

1022 fallback_model = model 

1023 

1024 # get email.message.Message variables for future processing 

1025 local_hostname = socket.gethostname() 

1026 message_id = message.get('Message-Id') 

1027 

1028 # compute references to find if message is a reply to an existing thread 

1029 references = tools.decode_message_header(message, 'References') 

1030 in_reply_to = tools.decode_message_header(message, 'In-Reply-To').strip() 

1031 thread_references = references or in_reply_to 

1032 reply_match, reply_model, reply_thread_id, reply_hostname, reply_private = tools.email_references(thread_references) 

1033 

1034 # author and recipients 

1035 email_from = tools.decode_message_header(message, 'From') 

1036 email_from_localpart = (tools.email_split(email_from) or [''])[0].split('@', 1)[0].lower() 

1037 email_to = tools.decode_message_header(message, 'To') 

1038 email_to_localpart = (tools.email_split(email_to) or [''])[0].split('@', 1)[0].lower() 

1039 

1040 # Delivered-To is a safe bet in most modern MTAs, but we have to fallback on To + Cc values 

1041 # for all the odd MTAs out there, as there is no standard header for the envelope's `rcpt_to` value. 

1042 rcpt_tos = ','.join([ 

1043 tools.decode_message_header(message, 'Delivered-To'), 

1044 tools.decode_message_header(message, 'To'), 

1045 tools.decode_message_header(message, 'Cc'), 

1046 tools.decode_message_header(message, 'Resent-To'), 

1047 tools.decode_message_header(message, 'Resent-Cc')]) 

1048 rcpt_tos_localparts = [e.split('@')[0].lower() for e in tools.email_split(rcpt_tos)] 

1049 

1050 # 0. Verify whether this is a bounced email and use it to collect bounce data and update notifications for customers 

1051 if bounce_alias and bounce_alias in email_to_localpart: 

1052 # Bounce regex: typical form of bounce is bounce_alias+128-crm.lead-34@domain 

1053 # group(1) = the mail ID; group(2) = the model (if any); group(3) = the record ID 

1054 bounce_re = re.compile("%s\+(\d+)-?([\w.]+)?-?(\d+)?" % re.escape(bounce_alias), re.UNICODE)

1104 msg_references = [ref for ref in tools.mail_header_msgid_re.findall(thread_references) if 'reply_to' not in ref]

1224 @api.model 

1225 def message_process(self, model, message, custom_values=None, 

1226 save_original=False, strip_attachments=False, 

1227 thread_id=None): 

1228 """ Process an incoming RFC2822 email message, relying on 

1229 ``mail.message.parse()`` for the parsing operation, 

1230 and ``message_route()`` to figure out the target model. 

1231 

1232 Once the target model is known, its ``message_new`` method 

1233 is called with the new message (if the thread record did not exist) 

1234 or its ``message_update`` method (if it did). 

1235 

1236 There is a special case where the target model is False: a reply 

1237 to a private message. In this case, we skip the message_new / 

1238 message_update step, to just post a new message using mail_thread 

1239 message_post. 

1240 

1241 :param string model: the fallback model to use if the message 

1242 does not match any of the currently configured mail aliases 

1243 (may be None if a matching alias is supposed to be present) 

1244 :param message: source of the RFC2822 message 

1245 :type message: string or xmlrpclib.Binary 

1246 :type dict custom_values: optional dictionary of field values 

1247 to pass to ``message_new`` if a new record needs to be created. 

1248 Ignored if the thread record already exists, and also if a 

1249 matching mail.alias was found (aliases define their own defaults) 

1250 :param bool save_original: whether to keep a copy of the original 

1251 email source attached to the message after it is imported. 

1252 :param bool strip_attachments: whether to strip all attachments 

1253 before processing the message, in order to save some space. 

1254 :param int thread_id: optional ID of the record/thread from ``model`` 

1255 to which this mail should be attached. When provided, this 

1256 overrides the automatic detection based on the message 

1257 headers. 

1258 """ 

1259 # extract message bytes - we are forced to pass the message as binary because 

1260 # we don't know its encoding until we parse its headers and hence can't 

1261 # convert it to utf-8 for transport between the mailgate script and here. 

1262 if isinstance(message, xmlrpclib.Binary): 

1263 message = str(message.data) 

1264 # Warning: message_from_string doesn't always work correctly on unicode, 

1265 # we must use utf-8 strings here :-( 

1266 if isinstance(message, unicode): 

1267 message = message.encode('utf-8') 

1268 msg_txt = email.message_from_string(message) 

1269 

1270 # parse the message, verify we are not in a loop by checking message_id is not duplicated 

1271 msg = self.message_parse(msg_txt, save_original=save_original) 

1272 if strip_attachments: 

1273 msg.pop('attachments', None) 

1274 

1275 if msg.get('message_id'): # should always be True as message_parse generate one if missing 

1276 existing_msg_ids = self.env['mail.message'].search([('message_id', '=', msg.get('message_id'))]) 

1277 if existing_msg_ids: 

1278 _logger.info('Ignored mail from %s to %s with Message-Id %s: found duplicated Message-Id during processing', 

1279 msg.get('from'), msg.get('to'), msg.get('message_id')) 

1280 return False 

1281 

1282 # find possible routes for the message 

1283 routes = self.message_route(msg_txt, msg, model, thread_id, custom_values) 

1284 thread_id = self.message_route_process(msg_txt, msg, routes) 

1285 return thread_id

1392 def _message_extract_payload(self, message, save_original=False): 

1393 """Extract body as HTML and attachments from the mail message""" 

1394 attachments = [] 

1395 body = u'' 

1396 if save_original: 

1397 attachments.append(self._Attachment('original_email.eml', message.as_string(), {})) 

1398 

1399 # Be careful, content-type may contain tricky content like in the 

1400 # following example so test the MIME type with startswith() 

1401 # 

1402 # Content-Type: multipart/related; 

1403 # boundary="_004_3f1e4da175f349248b8d43cdeb9866f1AMSPR06MB343eurprd06pro_"; 

1404 # type="text/html" 

1405 if message.get_content_maintype() == 'text': 

1406 encoding = message.get_content_charset() 

1407 body = message.get_payload(decode=True) 

1408 body = tools.ustr(body, encoding, errors='replace') 

1409 if message.get_content_type() == 'text/plain': 

1410 # text/plain -> <pre/> 

1411 body = tools.append_content_to_html(u'', body, preserve=True) 

1412 else: 

1413 alternative = False 

1414 mixed = False 

1415 html = u'' 

1416 for part in message.walk(): 

1417 if part.get_content_type() == 'multipart/alternative': 

1418 alternative = True 

1419 if part.get_content_type() == 'multipart/mixed': 

1420 mixed = True 

1421 if part.get_content_maintype() == 'multipart': 

1422 continue # skip container 

1423 # part.get_filename returns decoded value if able to decode, coded otherwise. 

1424 # original get_filename is not able to decode iso-8859-1 (for instance). 

1425 # therefore, iso encoded attachements are not able to be decoded properly with get_filename 

1426 # code here partially copy the original get_filename method, but handle more encoding 

1427 filename = part.get_param('filename', None, 'content-disposition') 

1428 if not filename: 

1429 filename = part.get_param('name', None) 

1430 if filename: 

1431 if isinstance(filename, tuple): 

1432 # RFC2231 

1433 filename = email.utils.collapse_rfc2231_value(filename).strip() 

1434 else: 

1435 filename = tools.decode_smtp_header(filename) 

1436 encoding = part.get_content_charset() # None if attachment 

1437 

1438 # 0) Inline Attachments -> attachments, with a third part in the tuple to match cid / attachment 

1439 if filename and part.get('content-id'): 

1440 inner_cid = part.get('content-id').strip('><') 

1441 attachments.append(self._Attachment(filename, part.get_payload(decode=True), {'cid': inner_cid})) 

1442 continue 

1443 # 1) Explicit Attachments -> attachments 

1444 if filename or part.get('content-disposition', '').strip().startswith('attachment'): 

1445 attachments.append(self._Attachment(filename or 'attachment', part.get_payload(decode=True), {})) 

1446 continue 

1447 # 2) text/plain -> <pre/> 

1448 if part.get_content_type() == 'text/plain' and (not alternative or not body): 

1449 body = tools.append_content_to_html(body, tools.ustr(part.get_payload(decode=True), 

1450 encoding, errors='replace'), preserve=True) 

1451 # 3) text/html -> raw 

1452 elif part.get_content_type() == 'text/html': 

1453 # mutlipart/alternative have one text and a html part, keep only the second 

1454 # mixed allows several html parts, append html content 

1455 append_content = not alternative or (html and mixed) 

1456 html = tools.ustr(part.get_payload(decode=True), encoding, errors='replace') 

1457 if not append_content: 

1458 body = html 

1459 else: 

1460 body = tools.append_content_to_html(body, html, plaintext=False) 

1461 # 4) Anything else -> attachment 

1462 else: 

1463 attachments.append(self._Attachment(filename or 'attachment', part.get_payload(decode=True), {})) 

1464 

1465 body, attachments = self._message_extract_payload_postprocess(message, body, attachments) 

1466 return body, attachments

1468 @api.model 

1469 def message_parse(self, message, save_original=False): 

1470 """Parses a string or email.message.Message representing an 

1471 RFC-2822 email, and returns a generic dict holding the 

1472 message details. 

1473 

1474 :param message: the message to parse 

1475 :type message: email.message.Message | string | unicode 

1476 :param bool save_original: whether the returned dict 

1477 should include an ``original`` attachment containing 

1478 the source of the message 

1479 :rtype: dict 

1480 :return: A dict with the following structure, where each 

1481 field may not be present if missing in original 

1482 message:: 

1483 

1484 { 'message_id': msg_id, 

1485 'subject': subject, 

1486 'from': from, 

1487 'to': to, 

1488 'cc': cc, 

1489 'body': unified_body, 

1490 'attachments': [('file1', 'bytes'), 

1491 ('file2', 'bytes')} 

1492 } 

1493 """ 

1494 msg_dict = { 

1495 'message_type': 'email', 

1496 } 

1497 if not isinstance(message, Message): 

1498 if isinstance(message, unicode): 

1499 # Warning: message_from_string doesn't always work correctly on unicode, 

1500 # we must use utf-8 strings here :-( 

1501 message = message.encode('utf-8') 

1502 message = email.message_from_string(message) 

1503 

1504 message_id = message['message-id'] 

1505 if not message_id: 

1506 # Very unusual situation, be we should be fault-tolerant here 

1507 message_id = "<%s@localhost>" % time.time() 

1508 _logger.debug('Parsing Message without message-id, generating a random one: %s', message_id) 

1509 msg_dict['message_id'] = message_id 

1510 

1511 if message.get('Subject'): 

1512 msg_dict['subject'] = tools.decode_smtp_header(message.get('Subject')) 

1513 

1514 # Envelope fields not stored in mail.message but made available for message_new() 

1515 msg_dict['from'] = tools.decode_smtp_header(message.get('from')) 

1516 msg_dict['to'] = tools.decode_smtp_header(message.get('to')) 

1517 msg_dict['cc'] = tools.decode_smtp_header(message.get('cc')) 

1518 msg_dict['email_from'] = tools.decode_smtp_header(message.get('from')) 

1519 partner_ids = self._message_find_partners(message, ['To', 'Cc']) 

1520 msg_dict['partner_ids'] = [(4, partner_id) for partner_id in partner_ids] 

1521 

1522 if message.get('Date'): 

1523 try: 

1524 date_hdr = tools.decode_smtp_header(message.get('Date')) 

1525 parsed_date = dateutil.parser.parse(date_hdr, fuzzy=True) 

1526 if parsed_date.utcoffset() is None: 

1527 # naive datetime, so we arbitrarily decide to make it 

1528 # UTC, there's no better choice. Should not happen, 

1529 # as RFC2822 requires timezone offset in Date headers. 

1530 stored_date = parsed_date.replace(tzinfo=pytz.utc) 

1531 else: 

1532 stored_date = parsed_date.astimezone(tz=pytz.utc) 

1533 except Exception: 

1534 _logger.info('Failed to parse Date header %r in incoming mail ' 

1535 'with message-id %r, assuming current date/time.', 

1536 message.get('Date'), message_id) 

1537 stored_date = datetime.datetime.now() 

1538 msg_dict['date'] = stored_date.strftime(tools.DEFAULT_SERVER_DATETIME_FORMAT)

1597 @api.multi 

1598 def _find_partner_from_emails(self, emails, res_model=None, res_id=None, check_followers=True, force_create=False, exclude_aliases=True): 

1599 """ Utility method to find partners from email addresses. The rules are : 

1600 1 - check in document (model | self, id) followers 

1601 2 - try to find a matching partner that is also an user 

1602 3 - try to find a matching partner 

1603 4 - create a new one if force_create = True 

1604 

1605 :param list emails: list of email addresses 

1606 :param string model: model to fetch related record; by default self 

1607 is used. 

1608 :param boolean check_followers: check in document followers 

1609 :param boolean force_create: create a new partner if not found 

1610 :param boolean exclude_aliases: do not try to find a partner that could match an alias. Normally aliases 

1611 should not be used as partner emails but it could be the case due to some 

1612 strange manipulation 

1613 """ 

1614 if res_model is None: 

1615 res_model = self._name 

1616 if res_id is None and self.ids: 

1617 res_id = self.ids[0] 

1618 followers = self.env['res.partner'] 

1619 if res_model and res_id: 

1620 record = self.env[res_model].browse(res_id) 

1621 if hasattr(record, 'message_partner_ids'): 

1622 followers = record.message_partner_ids 

1623 

1624 Partner = self.env['res.partner'].sudo() 

1625 Users = self.env['res.users'].sudo() 

1626 partner_ids = [] 

1627 

1628 for contact in emails: 

1629 partner_id = False 

1630 email_address = tools.email_split(contact) 

1631 if not email_address: 

1632 partner_ids.append(partner_id) 

1633 continue 

1634 if exclude_aliases and self.env['mail.alias'].search([('alias_name', 'ilike', email_address)], limit=1): 

1635 partner_ids.append(partner_id) 

1636 continue 

1637 

1638 email_address = email_address[0] 

1639 # first try: check in document's followers 

1640 partner_id = next((partner.id for partner in followers if partner.email == email_address), False) 

1641 

1642 # second try: check in partners that are also users 

1643 # Escape special SQL characters in email_address to avoid invalid matches 

1644 email_address = (email_address.replace('\\', '\\\\').replace('%', '\\%').replace('_', '\\_')) 

1645 email_brackets = "<%s>" % email_address 

1646 if not partner_id: 

1647 # exact, case-insensitive match 

1648 partners = Users.search([('email', '=ilike', email_address)], limit=1).mapped('partner_id') 

1649 if not partners: 

1650 # if no match with addr-spec, attempt substring match within name-addr pair 

1651 partners = Users.search([('email', 'ilike', email_brackets)], limit=1).mapped('partner_id') 

1652 partner_id = partners.id 

1653 # third try: check in partners 

1654 if not partner_id: 

1655 # exact, case-insensitive match 

1656 partners = Partner.search([('email', '=ilike', email_address)], limit=1) 

1657 if not partners: 

1658 # if no match with addr-spec, attempt substring match within name-addr pair 

1659 partners = Partner.search([('email', 'ilike', email_brackets)], limit=1) 

1660 partner_id = partners.id 

1661 if not partner_id and force_create: 

1662 partner_id = self.env['res.partner'].name_create(contact)[0] 

1663 partner_ids.append(partner_id) 

1664 return partner_ids
