169 @api.model_cr 

170 def _lookup_report(self, name): 

171 """ 

172 Look up a report definition. 

173 """ 

174 join = os.path.join 

175 

176 # First lookup in the deprecated place, because if the report definition 

177 # has not been updated, it is more likely the correct definition is there. 

178 # Only reports with custom parser sepcified in Python are still there. 

179 if 'report.' + name in odoo.report.interface.report_int._reports: 

180 return odoo.report.interface.report_int._reports['report.' + name] 

181 

182 self._cr.execute("SELECT * FROM ir_act_report_xml WHERE report_name=%s", (name,)) 

183 row = self._cr.dictfetchone() 

184 if not row: 

185 raise Exception("Required report does not exist: %s" % name) 

186 

187 if row['report_type'] in ('qweb-pdf', 'qweb-html'): 

188 return row['report_name'] 

189 elif row['report_rml'] or row['report_rml_content_data']: 

190 kwargs = {} 

191 if row['parser']: 

192 kwargs['parser'] = getattr(odoo.addons, row['parser']) 

193 return report_sxw('report.'+row['report_name'], row['model'], 

194 join('addons', row['report_rml'] or '/'), 

195 header=row['header'], register=False, **kwargs) 

196 elif row['report_xsl'] and row['report_xml']: 

197 return report_rml('report.'+row['report_name'], row['model'], 

198 join('addons', row['report_xml']), 

199 row['report_xsl'] and join('addons', row['report_xsl']), 

200 register=False) 

201 else: 

202 raise Exception("Unhandled report type: %s" % row)

262 @api.depends('view_ids.view_mode', 'view_mode', 'view_id.type') 

263 def _compute_views(self): 

264 """ Compute an ordered list of the specific view modes that should be 

265 enabled when displaying the result of this action, along with the 

266 ID of the specific view to use for each mode, if any were required. 

267 

268 This function hides the logic of determining the precedence between 

269 the view_modes string, the view_ids o2m, and the view_id m2o that 

270 can be set on the action. 

271 """ 

272 for act in self: 

273 act.views = [(view.view_id.id, view.view_mode) for view in act.view_ids] 

274 got_modes = [view.view_mode for view in act.view_ids] 

275 all_modes = act.view_mode.split(',') 

276 missing_modes = [mode for mode in all_modes if mode not in got_modes] 

277 if missing_modes: 

278 if act.view_id.type in missing_modes: 

279 # reorder missing modes to put view_id first if present 

280 missing_modes.remove(act.view_id.type) 

281 act.views.append((act.view_id.id, act.view_id.type)) 

282 act.views.extend([(False, mode) for mode in missing_modes])

572 def _check_expression(self, expression, model): 

573 """ Check python expression (condition, write_expression). Each step of 

574 the path must be a valid many2one field, or an integer field for the last 

575 step. 

576 

577 :param str expression: a python expression, beginning by 'obj' or 'object' 

578 :param model: a record of the model 'ir.model' 

579 :returns tuple: (is_valid, target_model_name, error_msg) 

580 """ 

581 if not model: 

582 return (False, None, 'Your expression cannot be validated because the Base Model is not set.') 

583 # fetch current model 

584 model_name = model.model 

585 # transform expression into a path that should look like 'object.many2onefield.many2onefield' 

586 path = expression.split('.') 

587 initial = path.pop(0) 

588 if initial not in ['obj', 'object', 'record']: 

589 return (False, None, 'Your expression should begin with obj, object, record.\nAn expression builder is available in the help tab.') 

590 # analyze path 

591 while path: 

592 step = path.pop(0) 

593 field = self.env[model_name]._fields.get(step) 

594 if not field: 

595 return (False, None, 'Part of the expression (%s) is not recognized as a column in the model %s.' % (step, model_name)) 

596 ftype = field.type 

597 if ftype not in ['many2one', 'int']: 

598 return (False, None, 'Part of the expression (%s) is not a valid column type (is %s, should be a many2one or an int)' % (step, ftype)) 

599 if ftype == 'int' and path: 

600 return (False, None, 'Part of the expression (%s) is an integer field that is only allowed at the end of an expression' % (step)) 

601 if ftype == 'many2one': 

602 model_name = field.comodel_name 

603 return (True, model_name, None)

956 @api.multi 

957 def run(self): 

958 """ Runs the server action. For each server action, the condition is 

959 checked. Note that a void (``False``) condition is considered as always 

960 valid. If it is verified, the run_action_<STATE> method is called. This 

961 allows easy overriding of the server actions. 

962 

963 :param dict context: context should contain following keys 

964 

965 - active_id: id of the current object (single mode) 

966 - active_model: current model that should equal the action's model 

967 

968 The following keys are optional: 

969 

970 - active_ids: ids of the current records (mass mode). If active_ids 

971 and active_id are present, active_ids is given precedence. 

972 

973 :return: an action_id to be executed, or False is finished correctly without 

974 return action 

975 """ 

976 res = False 

977 for action in self: 

978 eval_context = self._get_eval_context(action) 

979 condition = action.condition 

980 if condition is False: 

981 # Void (aka False) conditions are considered as True 

982 condition = True 

983 if hasattr(self, 'run_action_%s_multi' % action.state): 

984 expr = safe_eval(str(condition), eval_context) 

985 if not expr: 

986 continue 

987 # call the multi method 

988 run_self = self.with_context(eval_context['context']) 

989 func = getattr(run_self, 'run_action_%s_multi' % action.state) 

990 res = func(action, eval_context=eval_context) 

991 

992 elif hasattr(self, 'run_action_%s' % action.state): 

993 active_id = self._context.get('active_id') 

994 if not active_id and self._context.get('onchange_self'): 

995 active_id = self._context['onchange_self']._origin.id 

996 active_ids = self._context.get('active_ids', [active_id] if active_id else []) 

997 for active_id in active_ids: 

998 # run context dedicated to a particular active_id 

999 run_self = self.with_context(active_ids=[active_id], active_id=active_id) 

1000 eval_context["context"] = run_self._context 

1001 expr = safe_eval(str(condition), eval_context) 

1002 if not expr: 

1003 continue 

1004 # call the single method related to the action: run_action_<STATE> 

1005 func = getattr(run_self, 'run_action_%s' % action.state) 

1006 res = func(action, eval_context=eval_context) 

1007 return res
