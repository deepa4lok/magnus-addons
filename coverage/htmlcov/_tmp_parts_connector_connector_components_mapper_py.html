52def changed_by(*args): 

53 """ Decorator for the mapping methods (:py:func:`mapping`) 

54 

55 When fields are modified in Odoo, we want to export only the 

56 modified fields. Using this decorator, we can specify which fields 

57 updates should trigger which mapping method. 

58 

59 If ``changed_by`` is empty, the mapping is always active. 

60 

61 As far as possible, this decorator should be used for the exports, 

62 thus, when we do an update on only a small number of fields on a 

63 record, the size of the output record will be limited to only the 

64 fields really having to be exported. 

65 

66 Usage:: 

67 

68 @changed_by('input_field') 

69 @mapping 

70 def any(self, record): 

71 return {'output_field': record['input_field']} 

72 

73 :param ``*args``: field names which trigger the mapping when modified 

74 

75 """ 

76 def register_mapping(func): 

77 func.changed_by = args 

78 return func 

79 return register_mapping

100def none(field): 

101 """ A modifier intended to be used on the ``direct`` mappings. 

102 

103 Replace the False-ish values by None. 

104 It can be used in a pipeline of modifiers when . 

105 

106 Example:: 

107 

108 direct = [(none('source'), 'target'), 

109 (none(m2o_to_external('rel_id'), 'rel_id')] 

110 

111 :param field: name of the source field in the record 

112 :param binding: True if the relation is a binding record 

113 """ 

114 def modifier(self, record, to_attr): 

115 if callable(field): 

116 result = field(self, record, to_attr) 

117 else: 

118 result = record[field] 

119 if not result: 

120 return None 

121 return result 

122 return modifier

145def m2o_to_external(field, binding=None): 

146 """ A modifier intended to be used on the ``direct`` mappings. 

147 

148 For a many2one, get the external ID and returns it. 

149 

150 When the field's relation is not a binding (i.e. it does not point to 

151 something like ``magento.*``), the binding model needs to be provided 

152 in the ``binding`` keyword argument. 

153 

154 Example:: 

155 

156 direct = [(m2o_to_external('country_id', 

157 binding='magento.res.country'), 'country'), 

158 (m2o_to_external('magento_country_id'), 'country')] 

159 

160 :param field: name of the source field in the record 

161 :param binding: name of the binding model is the relation is not a binding 

162 """ 

163 def modifier(self, record, to_attr): 

164 if not record[field]: 

165 return False 

166 column = self.model._fields[field] 

167 if column.type != 'many2one': 

168 raise ValueError('The column %s should be a Many2one, got %s' % 

169 (field, type(column))) 

170 rel_id = record[field].id 

171 if binding is None: 

172 binding_model = column.comodel_name 

173 else: 

174 binding_model = binding 

175 binder = self.binder_for(binding_model) 

176 # if a relation is not a binding, we wrap the record in the 

177 # binding, we'll return the id of the binding 

178 wrap = bool(binding) 

179 value = binder.to_external(rel_id, wrap=wrap) 

180 if not value: 

181 raise MappingError("Can not find an external id for record " 

182 "%s in model %s %s wrapping" % 

183 (rel_id, binding_model, 

184 'with' if wrap else 'without')) 

185 return value 

186 return modifier

189def external_to_m2o(field, binding=None): 

190 """ A modifier intended to be used on the ``direct`` mappings. 

191 

192 For a field from a backend which is an ID, search the corresponding 

193 binding in Odoo and returns it. 

194 

195 When the field's relation is not a binding (i.e. it does not point to 

196 something like ``magento.*``), the binding model needs to be provided 

197 in the ``binding`` keyword argument. 

198 

199 Example:: 

200 

201 direct = [(external_to_m2o('country', binding='magento.res.country'), 

202 'country_id'), 

203 (external_to_m2o('country'), 'magento_country_id')] 

204 

205 :param field: name of the source field in the record 

206 :param binding: name of the binding model is the relation is not a binding 

207 """ 

208 def modifier(self, record, to_attr): 

209 if not record[field]: 

210 return False 

211 column = self.model._fields[to_attr] 

212 if column.type != 'many2one': 

213 raise ValueError('The column %s should be a Many2one, got %s' % 

214 (to_attr, type(column))) 

215 rel_id = record[field] 

216 if binding is None: 

217 binding_model = column.comodel_name 

218 else: 

219 binding_model = binding 

220 binder = self.binder_for(binding_model) 

221 # if we want the normal record, not a binding, 

222 # we ask to the binder to unwrap the binding 

223 unwrap = bool(binding) 

224 record = binder.to_internal(rel_id, unwrap=unwrap) 

225 if not record: 

226 raise MappingError("Can not find an existing %s for external " 

227 "record %s %s unwrapping" % 

228 (binding_model, rel_id, 

229 'with' if unwrap else 'without')) 

230 if isinstance(record, models.BaseModel): 

231 return record.id 

232 else: 

233 _logger.debug( 

234 'Binder for %s returned an id, ' 

235 'returning a record should be preferred.', binding_model 

236 ) 

237 return record 

238 return modifier

241def follow_m2o_relations(field): 

242 """A modifier intended to be used on ``direct`` mappings. 

243 

244 'Follows' Many2one relations and return the final field value. 

245 

246 Examples: 

247 Assuming model is ``product.product``:: 

248 

249 direct = [ 

250 (follow_m2o_relations('product_tmpl_id.categ_id.name'), 'cat')]

268class MapChild(AbstractComponent): 

269 """ MapChild is responsible to convert items. 

270 

271 Items are sub-records of a main record. 

272 In this example, the items are the records in ``lines``:: 

273 

274 sales = {'name': 'SO10', 

275 'lines': [{'product_id': 1, 'quantity': 2}, 

276 {'product_id': 2, 'quantity': 2}]} 

277 

278 A MapChild is always called from another :py:class:`Mapper` which 

279 provides a ``children`` configuration. 

280 

281 Considering the example above, the "main" :py:class:`Mapper` would 

282 returns something as follows:: 

283 

284 {'name': 'SO10', 

285 'lines': [(0, 0, {'product_id': 11, 'quantity': 2}), 

286 (0, 0, {'product_id': 12, 'quantity': 2})]} 

287 

288 A MapChild is responsible to: 

289 

290 * Find the :py:class:`Mapper` to convert the items 

291 * Possibly filter out some lines (can be done by inheriting 

292 :py:meth:`skip_item`) 

293 * Convert the items' records using the found :py:class:`Mapper` 

294 * Format the output values to the format expected by Odoo or the 

295 backend (as seen above with ``(0, 0, {values})`` 

296 

297 A MapChild can be extended like any other 

298 :py:class:`~component.core.Component`. 

299 However, it is not mandatory to explicitly create a MapChild for 

300 each children mapping, the default one will be used 

301 (:py:class:`ImportMapChild` or :py:class:`ExportMapChild`). 

302 

303 The implementation by default does not take care of the updates: if 

304 I import a sales order 2 times, the lines will be duplicated. This 

305 is not a problem as long as an importation should only support the 

306 creation (typical for sales orders). It can be implemented on a 

307 case-by-case basis by inheriting :py:meth:`get_item_values` and 

308 :py:meth:`format_items`. 

309 

310 """

330 def get_items(self, items, parent, to_attr, options): 

331 """ Returns the formatted output values of items from a main record 

332 

333 :param items: list of item records 

334 :type items: list 

335 :param parent: parent record 

336 :param to_attr: destination field (can be used for introspecting 

337 the relation) 

338 :type to_attr: str 

339 :param options: dict of options, herited from the main mapper 

340 :return: formatted output values for the item 

341

343 mapper = self._child_mapper() 

344 mapped = [] 

345 for item in items: 

346 map_record = mapper.map_record(item, parent=parent) 

347 if self.skip_item(map_record): 

348 continue 

349 item_values = self.get_item_values(map_record, to_attr, options) 

350 if item_values: 

351 mapped.append(item_values) 

352 return self.format_items(mapped)

431class Mapper(AbstractComponent): 

432 """ A Mapper translates an external record to an Odoo record and 

433 conversely. The output of a Mapper is a ``dict``. 

434 

435 3 types of mappings are supported: 

436 

437 Direct Mappings 

438 Example:: 

439 

440 direct = [('source', 'target')] 

441 

442 Here, the ``source`` field will be copied in the ``target`` field. 

443 

444 A modifier can be used in the source item. 

445 The modifier will be applied to the source field before being 

446 copied in the target field. 

447 It should be a closure function respecting this idiom:: 

448 

449 def a_function(field): 

450 ''' ``field`` is the name of the source field. 

451 

452 Naming the arg: ``field`` is required for the conversion''' 

453 def modifier(self, record, to_attr): 

454 ''' self is the current Mapper, 

455 record is the current record to map, 

456 to_attr is the target field''' 

457 return record[field] 

458 return modifier 

459 

460 And used like that:: 

461 

462 direct = [ 

463 (a_function('source'), 'target'), 

464 ] 

465 

466 A more concrete example of modifier:: 

467 

468 def convert(field, conv_type): 

469 ''' Convert the source field to a defined ``conv_type`` 

470 (ex. str) before returning it''' 

471 def modifier(self, record, to_attr): 

472 value = record[field] 

473 if not value: 

474 return None 

475 return conv_type(value) 

476 return modifier 

477 

478 And used like that:: 

479 

480 direct = [ 

481 (convert('myfield', float), 'target_field'), 

482 ] 

483 

484 More examples of modifiers: 

485 

486 * :py:func:`convert` 

487 * :py:func:`m2o_to_external` 

488 * :py:func:`external_to_m2o` 

489 

490 Method Mappings 

491 A mapping method allows to execute arbitrary code and return one 

492 or many fields:: 

493 

494 @mapping 

495 def compute_state(self, record): 

496 # compute some state, using the ``record`` or not 

497 state = 'pending' 

498 return {'state': state} 

499 

500 We can also specify that a mapping methods should be applied 

501 only when an object is created, and never applied on further 

502 updates:: 

503 

504 @only_create 

505 @mapping 

506 def default_warehouse(self, record): 

507 # get default warehouse 

508 warehouse_id = ... 

509 return {'warehouse_id': warehouse_id} 

510 

511 Submappings 

512 When a record contains sub-items, like the lines of a sales order, 

513 we can convert the children using another Mapper:: 

514 

515 children = [('items', 'line_ids', 'model.name')] 

516 

517 It allows to create the sales order and all its lines with the 

518 same call to :py:meth:`odoo.models.BaseModel.create()`. 

519 

520 When using ``children`` for items of a record, we need to create 

521 a :py:class:`Mapper` for the model of the items, and optionally a 

522 :py:class:`MapChild`. 

523 

524 Usage of a Mapper:: 

525 

526 >>> mapper = self.component(usage='mapper') 

527 >>> map_record = mapper.map_record(record) 

528 >>> values = map_record.values() 

529 >>> values = map_record.values(only_create=True) 

530 >>> values = map_record.values(fields=['name', 'street']) 

531 

532 """ 

533 

534 _name = 'base.mapper' 

535 _inherit = 'base.connector' 

536 _usage = 'mapper' 

537 

538 direct = [] # direct conversion of a field to another (from_attr, to_attr) 

539 children = [] # conversion of sub-records (from_attr, to_attr, model)

546 @classmethod 

547 def _build_mapper_component(cls): 

548 """ Build a Mapper component 

549 

550 When a Mapper component is built, we will look into every of its bases 

551 and look for methods decorated by ``@mapping`` or ``@changed_by``. We 

552 keep the definitions in a ``_map_methods`` attribute for later use by 

553 the Mapper instances. 

554 

555 The ``__bases__`` of a newly generated Component are of 2 kinds: 

556 

557 * other dynamically generated components (below 'base' and 

558 'second.mapper') 

559 * "real" Python classes applied on top of existing components (here 

560 ThirdMapper) 

561 

562 :: 

563 

564 >>> cls.__bases__ 

565 (<class 'odoo.addons.connector.tests.test_mapper.ThirdMapper'>, 

566 <class 'odoo.addons.component.core.second.mapper'>, 

567 <class 'odoo.addons.component.core.base'>) 

568 

569 This method traverses these bases, from the bottom to the top, and 

570 merges the mapping definitions. It reuses the computed definitions 

571 for the generated components (for which this code already ran), and 

572 inspect the real classes to find mapping methods. 

573 

574 """ 

575 

576 map_methods = {} 

577 for base in reversed(cls.__bases__): 

578 if hasattr(base, '_map_methods'): 

579 # this is already a dynamically generated Component, so we can 

580 # use it's existing mappings 

581 base_map_methods = base._map_methods or {} 

582 for attr_name, definition in base_map_methods.iteritems(): 

583 if attr_name in map_methods: 

584 # Update the existing @changed_by with the content 

585 # of each base (it is mutated in place). 

586 mapping_changed_by = map_methods[attr_name].changed_by 

587 mapping_changed_by.update(definition.changed_by) 

588 # keep the last value for @only_create 

589 if definition.only_create: 

590 new_definition = MappingDefinition( 

591 mapping_changed_by, 

592 definition.only_create, 

593 ) 

594 map_methods[attr_name] = new_definition 

595 else: 

596 map_methods[attr_name] = definition 

597 else: 

598 # this is a real class that needs to be applied upon 

599 # the base Components 

600 for attr_name in dir(base): 

601 attr = getattr(base, attr_name, None) 

602 if not getattr(attr, 'is_mapping', None): 

603 continue 

604 has_only_create = getattr(attr, 'only_create', False) 

605 mapping_changed_by = set(getattr(attr, 'changed_by', ())) 

606 

607 # if already existing, it has been defined in an previous 

608 # base, extend the @changed_by set 

609 if map_methods.get(attr_name) is not None: 

610 definition = map_methods[attr_name] 

611 mapping_changed_by.update(definition.changed_by) 

612 

613 # keep the last choice for only_create 

614 definition = MappingDefinition(mapping_changed_by, 

615 has_only_create) 

616 map_methods[attr_name] = definition 

617 

618 cls._map_methods = map_methods

692 def changed_by_fields(self):

694

708 def _direct_source_field_name(self, direct_entry): 

709 """ Get the mapping field name. Goes through the function modifiers. 

710 

711 Ex:: 

712 

713 [(none(convert(field_name, str)), out_field_name)] 

714 

715 It assumes that the modifier has ``field`` as first argument like:: 

716 

717 def modifier(field, args): 

718 

719 """ 

720 fieldname = direct_entry 

721 if callable(direct_entry): 

722 # Map the closure entries with variable names 

723 cells = dict(zip( 

724 direct_entry.func_code.co_freevars, 

725 (c.cell_contents for c in direct_entry.func_closure))) 

726 assert 'field' in cells, "Modifier without 'field' argument." 

727 if callable(cells['field']): 

728 fieldname = self._direct_source_field_name(cells['field']) 

729 else: 

730 fieldname = cells['field'] 

731 return fieldname

735 converted using the current Mapper.

738 :param parent: optional parent record, for items

743 def _apply(self, map_record, options=None): 

744 """ Apply the mappings on a :py:class:`MapRecord` 

745 

746 :param map_record: source record to convert 

747 :type map_record: :py:class:`MapRecord` 

748 

749 """ 

750 if options is None: 

751 options = {} 

752 with self._mapping_options(options): 

753 return self._apply_with_options(map_record) 

754 

755 def _apply_with_options(self, map_record): 

756 """ Apply the mappings on a :py:class:`MapRecord` with 

757 contextual options (the ``options`` given in 

758 :py:meth:`MapRecord.values()` are accessible in 

759 ``self.options``) 

760 

761 :param map_record: source record to convert 

762 :type map_record: :py:class:`MapRecord` 

763 

764 """ 

765 assert self.options is not None, ( 

766 "options should be defined with '_mapping_options'") 

767 _logger.debug('converting record %s to model %s', 

768 map_record.source, self.model) 

769 

770 fields = self.options.fields 

771 for_create = self.options.for_create 

772 result = {} 

773 for from_attr, to_attr in self.direct: 

774 if callable(from_attr): 

775 attr_name = self._direct_source_field_name(from_attr) 

776 else: 

777 attr_name = from_attr 

778 

779 if (not fields or attr_name in fields): 

780 value = self._map_direct(map_record.source, 

781 from_attr, 

782 to_attr) 

783 result[to_attr] = value 

784 

785 for meth, definition in self.map_methods: 

786 mapping_changed_by = definition.changed_by 

787 if (not fields or not mapping_changed_by or 

788 mapping_changed_by.intersection(fields)): 

789 if definition.only_create and not for_create: 

790 continue 

791 values = meth(map_record.source) 

792 if not values: 

793 continue 

794 if not isinstance(values, dict): 

795 raise ValueError('%s: invalid return value for the ' 

796 'mapping method %s' % (values, meth)) 

797 result.update(values) 

798 

799 for from_attr, to_attr, model_name in self.children: 

800 if (not fields or from_attr in fields): 

801 result[to_attr] = self._map_child(map_record, from_attr, 

802 to_attr, model_name) 

803 

804 return self.finalize(map_record, result) 

805 

806 def finalize(self, map_record, values): 

807 """ Called at the end of the mapping. 

808 

809 Can be used to modify the values generated by all the mappings before 

810 returning them. 

811 

812 :param map_record: source map_record 

813 :type map_record: :py:class:`MapRecord` 

814 :param values: mapped values 

815 :returns: mapped values 

816 :rtype: dict 

817 """ 

818 return values

903class MapRecord(object): 

904 """ A record prepared to be converted using a :py:class:`Mapper`. 

905 

906 MapRecord instances are prepared by :py:meth:`Mapper.map_record`. 

907 

908 Usage:: 

909 

910 >>> map_record = mapper.map_record(record) 

911 >>> output_values = map_record.values() 

912 

913 See :py:meth:`values` for more information on the available arguments. 

914 

915 """ 

916 

917 def __init__(self, mapper, source, parent=None): 

918 self._source = source 

919 self._mapper = mapper 

920 self._parent = parent 

921 self._forced_values = {} 

922 

923 @property 

924 def source(self): 

925 """ Source record to be converted """ 

926 return self._source 

927 

928 @property 

929 def parent(self): 

930 """ Parent record if the current record is an item """ 

931 return self._parent 

932 

933 def values(self, for_create=None, fields=None, **kwargs): 

934 """ Build and returns the mapped values according to the options. 

935 

936 Usage:: 

937 

938 >>> map_record = mapper.map_record(record) 

939 >>> output_values = map_record.values() 

940 

941 Creation of records 

942 When using the option ``for_create``, only the mappings decorated 

943 with ``@only_create`` will be mapped. 

944 

945 :: 

946 

947 >>> output_values = map_record.values(for_create=True) 

948 

949 Filter on fields 

950 When using the ``fields`` argument, the mappings will be 

951 filtered using either the source key in ``direct`` arguments, 

952 either the ``changed_by`` arguments for the mapping methods. 

953 

954 :: 

955 

956 >>> output_values = map_record.values( 

957 fields=['name', 'street'] 

958 ) 

959 

960 Custom options 

961 Arbitrary key and values can be defined in the ``kwargs`` 

962 arguments. They can later be used in the mapping methods 

963 using ``self.options``. 

964 

965 :: 

966 

967 >>> output_values = map_record.values(tax_include=True) 

968 

969 :param for_create: specify if only the mappings for creation 

970 (``@only_create``) should be mapped. 

971 :type for_create: boolean 

972 :param fields: filter on fields 

973 :type fields: list 

974 :param ``**kwargs``: custom options, they can later be used in the 

975 mapping methods 

976 

977 """ 

978 options = MapOptions(for_create=for_create, fields=fields, **kwargs) 

979 values = self._mapper._apply(self, options=options) 

980 values.update(self._forced_values) 

981 return values 

982 

983 def update(self, *args, **kwargs): 

984 """ Force values to be applied after a mapping. 

985 

986 Usage:: 

987 

988 >>> map_record = mapper.map_record(record) 

989 >>> map_record.update(a=1) 

990 >>> output_values = map_record.values() 

991 # output_values will at least contain {'a': 1} 

992 

993 The values assigned with ``update()`` are in any case applied, 

994 they have a greater priority than the mapping values. 

995 

996 """ 

997 self._forced_values.update(*args, **kwargs)
