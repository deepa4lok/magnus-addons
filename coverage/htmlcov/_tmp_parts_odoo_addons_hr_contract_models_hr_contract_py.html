22 def _compute_contract_id(self): 

23 """ get the lastest contract """ 

24 Contract = self.env['hr.contract'] 

25 for employee in self: 

26 employee.contract_id = Contract.search([('employee_id', '=', employee.id)], order='date_start desc', limit=1) 

27 

28 def _compute_contracts_count(self): 

29 # read_group as sudo, since contract count is displayed on form view 

30 contract_data = self.env['hr.contract'].sudo().read_group([('employee_id', 'in', self.ids)], ['employee_id'], ['employee_id']) 

31 result = dict((data['employee_id'][0], data['employee_id_count']) for data in contract_data) 

32 for employee in self: 

33 employee.contracts_count = result.get(employee.id, 0)
