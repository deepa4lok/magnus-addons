158 def test_40_res_partner_address_get(self): 

159 """ Test address_get address resolution mechanism: it should first go down through descendants, 

160 stopping when encountering another is_copmany entity, then go up, stopping again at the first 

161 is_company entity or the root ancestor and if nothing matches, it should use the provided partner 

162 itself """ 

163 res_partner = self.env['res.partner'] 

164 elmtree = res_partner.browse(res_partner.name_create('Elmtree')[0]) 

165 branch1 = res_partner.create({'name': 'Branch 1', 

166 'parent_id': elmtree.id, 

167 'is_company': True}) 

168 leaf10 = res_partner.create({'name': 'Leaf 10', 

169 'parent_id': branch1.id, 

170 'type': 'invoice'}) 

171 branch11 = res_partner.create({'name': 'Branch 11', 

172 'parent_id': branch1.id, 

173 'type': 'other'}) 

174 leaf111 = res_partner.create({'name': 'Leaf 111', 

175 'parent_id': branch11.id, 

176 'type': 'delivery'}) 

177 branch11.write({'is_company': False}) # force is_company after creating 1rst child 

178 branch2 = res_partner.create({'name': 'Branch 2', 

179 'parent_id': elmtree.id, 

180 'is_company': True}) 

181 leaf21 = res_partner.create({'name': 'Leaf 21', 

182 'parent_id': branch2.id, 

183 'type': 'delivery'}) 

184 leaf22 = res_partner.create({'name': 'Leaf 22', 

185 'parent_id': branch2.id}) 

186 leaf23 = res_partner.create({'name': 'Leaf 23', 

187 'parent_id': branch2.id, 

188 'type': 'contact'}) 

189 

190 # go up, stop at branch1 

191 self.assertEqual(leaf111.address_get(['delivery', 'invoice', 'contact', 'other']), 

192 {'delivery': leaf111.id, 

193 'invoice': leaf10.id, 

194 'contact': branch1.id, 

195 'other': branch11.id}, 'Invalid address resolution') 

196 self.assertEqual(branch11.address_get(['delivery', 'invoice', 'contact', 'other']), 

197 {'delivery': leaf111.id, 

198 'invoice': leaf10.id, 

199 'contact': branch1.id, 

200 'other': branch11.id}, 'Invalid address resolution') 

201 

202 # go down, stop at at all child companies 

203 self.assertEqual(elmtree.address_get(['delivery', 'invoice', 'contact', 'other']), 

204 {'delivery': elmtree.id, 

205 'invoice': elmtree.id, 

206 'contact': elmtree.id, 

207 'other': elmtree.id}, 'Invalid address resolution') 

208 

209 # go down through children 

210 self.assertEqual(branch1.address_get(['delivery', 'invoice', 'contact', 'other']), 

211 {'delivery': leaf111.id, 

212 'invoice': leaf10.id, 

213 'contact': branch1.id, 

214 'other': branch11.id}, 'Invalid address resolution') 

215 

216 self.assertEqual(branch2.address_get(['delivery', 'invoice', 'contact', 'other']), 

217 {'delivery': leaf21.id, 

218 'invoice': branch2.id, 

219 'contact': branch2.id, 

220 'other': branch2.id}, 'Invalid address resolution. Company is the first encountered contact, therefore default for unfound addresses.') 

221 

222 # go up then down through siblings 

223 self.assertEqual(leaf21.address_get(['delivery', 'invoice', 'contact', 'other']), 

224 {'delivery': leaf21.id, 

225 'invoice': branch2.id, 

226 'contact': branch2.id, 

227 'other': branch2.id}, 'Invalid address resolution, should scan commercial entity ancestor and its descendants') 

228 self.assertEqual(leaf22.address_get(['delivery', 'invoice', 'contact', 'other']), 

229 {'delivery': leaf21.id, 

230 'invoice': leaf22.id, 

231 'contact': leaf22.id, 

232 'other': leaf22.id}, 'Invalid address resolution, should scan commercial entity ancestor and its descendants') 

233 self.assertEqual(leaf23.address_get(['delivery', 'invoice', 'contact', 'other']), 

234 {'delivery': leaf21.id, 

235 'invoice': leaf23.id, 

236 'contact': leaf23.id, 

237 'other': leaf23.id}, 'Invalid address resolution, `default` should only override if no partner with specific type exists') 

238 

239 # empty adr_pref means only 'contact' 

240 self.assertEqual(elmtree.address_get([]), 

241 {'contact': elmtree.id}, 'Invalid address resolution, no contact means commercial entity ancestor') 

242 self.assertEqual(leaf111.address_get([]), 

243 {'contact': branch1.id}, 'Invalid address resolution, no contact means finding contact in ancestors') 

244 branch11.write({'type': 'contact'}) 

245 self.assertEqual(leaf111.address_get([]), 

246 {'contact': branch11.id}, 'Invalid address resolution, branch11 should now be contact')
