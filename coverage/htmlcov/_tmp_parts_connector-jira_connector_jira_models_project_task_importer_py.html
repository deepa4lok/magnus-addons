31 @mapping 

32 def issue_type(self, record): 

33 binder = self.binder_for('jira.issue.type') 

34 jira_type_id = record['fields']['issuetype']['id'] 

35 binding = binder.to_internal(jira_type_id) 

36 return {'jira_issue_type_id': binding.id}

38 @mapping 

39 def assignee(self, record): 

40 assignee = record['fields'].get('assignee') 

41 if not assignee: 

42 return {'user_id': False} 

43 jira_key = assignee['key'] 

44 binder = self.binder_for('jira.res.users') 

45 user = binder.to_internal(jira_key, unwrap=True) 

46 if not user: 

47 email = assignee['emailAddress'] 

48 raise MappingError( 

49 _('No user found with login "%s" or email "%s".' 

50 'You must create a user or link it manually if the ' 

51 'login/email differs.') % (jira_key, email) 

52 ) 

53 return {'user_id': user.id}

141 def _import(self, binding, **kwargs): 

142 # called at the beginning of _import because we must be sure

148 def _import_dependencies(self): 

149 """ Import the dependencies for the record""" 

150 jira_assignee = self.external_record['fields'].get('assignee') or {} 

151 jira_key = jira_assignee.get('key') 

152 self._import_dependency(jira_key,

161 jira_parent = self.external_record['fields'].get('parent') 

162 if jira_parent: 

163 jira_parent_id = jira_parent['id'] 

164 self._import_dependency(jira_parent_id, 'jira.project.task', 

165 record=jira_parent)

167 if self.jira_epic:

169 record=self.jira_epic)
