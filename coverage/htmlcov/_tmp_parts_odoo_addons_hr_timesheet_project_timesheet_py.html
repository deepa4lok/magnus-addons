25 def _hours_get(self): 

26 for task in self.sorted(key='id', reverse=True): 

27 children_hours = 0 

28 for child_task in task.child_ids: 

29 if child_task.stage_id and child_task.stage_id.fold: 

30 children_hours += child_task.effective_hours + child_task.children_hours 

31 else: 

32 children_hours += max(child_task.planned_hours, child_task.effective_hours + child_task.children_hours) 

33 

34 task.children_hours = children_hours 

35 task.effective_hours = sum(task.timesheet_ids.mapped('unit_amount')) 

36 task.remaining_hours = task.planned_hours - task.effective_hours - task.children_hours 

37 task.total_hours = max(task.planned_hours, task.effective_hours) 

38 task.total_hours_spent = task.effective_hours + task.children_hours 

39 task.delay_hours = max(-task.remaining_hours, 0.0) 

40 

41 if task.stage_id and task.stage_id.fold: 

42 task.progress = 100.0 

43 elif (task.planned_hours > 0.0): 

44 task.progress = round(100.0 * (task.effective_hours + task.children_hours) / task.planned_hours, 2) 

45 else: 

46 task.progress = 0.0
