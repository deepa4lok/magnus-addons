37 @lazy_classproperty 

38 def registries(cls): 

39 """ A mapping from database names to registries. """ 

40 size = config.get('registry_lru_size', None) 

41 if not size: 

42 # Size the LRU depending of the memory limits 

43 if os.name != 'posix': 

44 # cannot specify the memory limit soft on windows... 

45 size = 42 

46 else: 

47 # A registry takes 10MB of memory on average, so we reserve 

48 # 10Mb (registry) + 5Mb (working memory) per registry 

49 avgsz = 15 * 1024 * 1024 

50 size = int(config['limit_memory_soft'] / avgsz) 

51 return LRU(size)

53 def __new__(cls, db_name): 

54 """ Return the registry for the given database name.""" 

55 with cls._lock: 

56 try: 

57 return cls.registries[db_name] 

58 except KeyError: 

59 return cls.new(db_name) 

60 finally: 

61 # set db tracker - cleaned up at the WSGI dispatching phase in 

62 # odoo.service.wsgi_server.application 

63 threading.current_thread().dbname = db_name

65 @classmethod 

66 def new(cls, db_name, force_demo=False, status=None, update_module=False): 

67 """ Create and return a new registry for the given database name. """ 

68 with cls._lock: 

69 with odoo.api.Environment.manage(): 

70 registry = object.__new__(cls) 

71 registry.init(db_name) 

72 

73 # Initializing a registry will call general code which will in 

74 # turn call Registry() to obtain the registry being initialized. 

75 # Make it available in the registries dictionary then remove it 

76 # if an exception is raised. 

77 cls.delete(db_name) 

78 cls.registries[db_name] = registry 

79 try: 

80 registry.setup_signaling() 

81 # This should be a method on Registry 

82 try: 

83 odoo.modules.load_modules(registry._db, force_demo, status, update_module) 

84 except Exception: 

85 odoo.modules.reset_modules_state(db_name) 

86 raise 

87 except Exception: 

88 _logger.exception('Failed to load registry') 

89 del cls.registries[db_name] 

90 raise 

91 

92 # load_modules() above can replace the registry by calling 

93 # indirectly new() again (when modules have to be uninstalled). 

94 # Yeah, crazy. 

95 init_parent = registry._init_parent 

96 registry = cls.registries[db_name] 

97 registry._init_parent.update(init_parent) 

98 

99 with closing(registry.cursor()) as cr: 

100 registry.do_parent_store(cr) 

101 cr.commit() 

102 

103 registry.ready = True 

104 

105 if update_module: 

106 # only in case of update, otherwise we'll have an infinite reload loop! 

107 registry.signal_registry_change() 

108 return registry

188 def field_sequence(self): 

189 """ Return a function mapping a field to an integer. The value of a 

190 field is guaranteed to be strictly greater than the value of the 

191 field's dependencies. 

192 """ 

193 # map fields on their dependents 

194 dependents = { 

195 field: set(dep for dep, _ in model._field_triggers[field] if dep != field) 

196 for model in self.itervalues() 

197 for field in model._fields.itervalues() 

198 } 

199 # sort them topologically, and associate a sequence number to each field 

200 mapping = { 

201 field: num 

202 for num, field in enumerate(reversed(topological_sort(dependents))) 

203 } 

204 return mapping.get

233 def descendants(self, model_names, *kinds): 

234 """ Return the models corresponding to ``model_names`` and all those 

235 that inherit/inherits from them. 

236 """ 

237 assert all(kind in ('_inherit', '_inherits') for kind in kinds) 

238 funcs = [attrgetter(kind + '_children') for kind in kinds] 

239 

240 models = OrderedSet() 

241 queue = deque(model_names) 

242 while queue: 

243 model = self[queue.popleft()] 

244 models.add(model._name) 

245 for func in funcs: 

246 queue.extend(func(model)) 

247 return models

305 def init_models(self, cr, model_names, context): 

306 """ Initialize a list of models (given by their name). Call methods 

307 ``_auto_init``, ``init``, and ``_auto_end`` on each model to create 

308 or update the database tables supporting the models. 

309 

310 The ``context`` may contain the following items: 

311 - ``module``: the name of the module being installed/updated, if any; 

312 - ``update_custom_fields``: whether custom fields should be updated. 

313 """ 

314 if 'module' in context: 

315 _logger.info('module %s: creating or updating database tables', context['module']) 

316 elif context.get('models_to_check', False): 

317 _logger.info("verifying fields for every extended model") 

318 

319 context = dict(context, todo=[]) 

320 env = odoo.api.Environment(cr, SUPERUSER_ID, context) 

321 models = [env[model_name] for model_name in model_names] 

322 

323 for model in models: 

324 model._auto_init() 

325 model.init() 

326 cr.commit() 

327 

328 for model in models: 

329 model._auto_end() 

330 cr.commit() 

331 

332 for _, func, args in sorted(context['todo']): 

333 func(*args) 

334 

335 if models: 

336 models[0].recompute() 

337 cr.commit() 

338 

339 # make sure all tables are present 

340 missing = [name 

341 for name, model in env.items() 

342 if not model._abstract and not model._table_exist()] 

343 if missing: 

344 _logger.warning("Models have no table: %s.", ", ".join(missing)) 

345 # recreate missing tables following model dependencies 

346 deps = {name: model._depends for name, model in env.items()} 

347 for name in topological_sort(deps): 

348 if name in missing: 

349 _logger.info("Recreate table of model %s.", name) 

350 env[name].init() 

351 cr.commit() 

352 # check again, and log errors if tables are still missing 

353 for name, model in env.items(): 

354 if not model._abstract and not model._table_exist(): 

355 _logger.error("Model %s has no table.", name)

365 def setup_signaling(self): 

366 """ Setup the inter-process signaling on this registry. """ 

367 if self.in_test_mode(): 

368 return 

369 

370 with self.cursor() as cr: 

371 # The `base_registry_signaling` sequence indicates when the registry 

372 # must be reloaded. 

373 # The `base_cache_signaling` sequence indicates when all caches must 

374 # be invalidated (i.e. cleared). 

375 cr.execute("SELECT sequence_name FROM information_schema.sequences WHERE sequence_name='base_registry_signaling'") 

376 if not cr.fetchall(): 

377 cr.execute("CREATE SEQUENCE base_registry_signaling INCREMENT BY 1 START WITH 1") 

378 cr.execute("SELECT nextval('base_registry_signaling')") 

379 cr.execute("CREATE SEQUENCE base_cache_signaling INCREMENT BY 1 START WITH 1") 

380 cr.execute("SELECT nextval('base_cache_signaling')") 

381 

382 cr.execute(""" SELECT base_registry_signaling.last_value, 

383 base_cache_signaling.last_value 

384 FROM base_registry_signaling, base_cache_signaling""") 

385 self.registry_sequence, self.cache_sequence = cr.fetchone() 

386 _logger.debug("Multiprocess load registry signaling: [Registry: %s] [Cache: %s]", 

387 self.registry_sequence, self.cache_sequence)

389 def check_signaling(self): 

390 """ Check whether the registry has changed, and performs all necessary 

391 operations to update the registry. Return an up-to-date registry. 

392 """ 

393 if self.in_test_mode(): 

394 return self 

395 

396 with closing(self.cursor()) as cr: 

397 cr.execute(""" SELECT base_registry_signaling.last_value, 

398 base_cache_signaling.last_value 

399 FROM base_registry_signaling, base_cache_signaling""") 

400 r, c = cr.fetchone() 

401 _logger.debug("Multiprocess signaling check: [Registry - %s -> %s] [Cache - %s -> %s]", 

402 self.registry_sequence, r, self.cache_sequence, c) 

403 # Check if the model registry must be reloaded 

404 if self.registry_sequence != r: 

405 _logger.info("Reloading the model registry after database signaling.") 

406 self = Registry.new(self.db_name) 

407 # Check if the model caches must be invalidated. 

408 elif self.cache_sequence != c: 

409 _logger.info("Invalidating all model caches after database signaling.") 

410 self.clear_caches() 

411 self.cache_cleared = False 

412 self.registry_sequence = r 

413 self.cache_sequence = c 

414 

415 return self
