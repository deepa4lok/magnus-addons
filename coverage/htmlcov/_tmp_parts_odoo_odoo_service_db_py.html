120def _drop_conn(cr, db_name): 

121 # Try to terminate all other connections that might prevent 

122 # dropping the database 

123 try: 

124 # PostgreSQL 9.2 renamed pg_stat_activity.procpid to pid: 

125 # http://www.postgresql.org/docs/9.2/static/release-9-2.html#AEN110389 

126 pid_col = 'pid' if cr._cnx.server_version >= 90200 else 'procpid' 

127 

128 cr.execute("""SELECT pg_terminate_backend(%(pid_col)s) 

129 FROM pg_stat_activity 

130 WHERE datname = %%s AND 

131 %(pid_col)s != pg_backend_pid()""" % {'pid_col': pid_col}, 

132 (db_name,)) 

133 except Exception: 

134 pass

181def dump_db(db_name, stream, backup_format='zip'): 

182 """Dump database `db` into file-like object `stream` if stream is None 

183 return a file object with the dump """ 

184 

185 _logger.info('DUMP DB: %s format %s', db_name, backup_format) 

186 

187 cmd = ['pg_dump', '--no-owner'] 

188 cmd.append(db_name) 

189 

190 if backup_format == 'zip': 

191 with odoo.tools.osutil.tempdir() as dump_dir: 

192 filestore = odoo.tools.config.filestore(db_name) 

193 if os.path.exists(filestore): 

194 shutil.copytree(filestore, os.path.join(dump_dir, 'filestore')) 

195 with open(os.path.join(dump_dir, 'manifest.json'), 'w') as fh: 

196 db = odoo.sql_db.db_connect(db_name) 

197 with db.cursor() as cr: 

198 json.dump(dump_db_manifest(cr), fh, indent=4) 

199 cmd.insert(-1, '--file=' + os.path.join(dump_dir, 'dump.sql')) 

200 odoo.tools.exec_pg_command(*cmd) 

201 if stream: 

202 odoo.tools.osutil.zip_dir(dump_dir, stream, include_dir=False, fnct_sort=lambda file_name: file_name != 'dump.sql') 

203 else: 

204 t=tempfile.TemporaryFile() 

205 odoo.tools.osutil.zip_dir(dump_dir, t, include_dir=False, fnct_sort=lambda file_name: file_name != 'dump.sql') 

206 t.seek(0) 

207 return t 

208 else: 

209 cmd.insert(-1, '--format=c') 

210 stdin, stdout = odoo.tools.exec_pg_command_pipe(*cmd) 

211 if stream: 

212 shutil.copyfileobj(stdout, stream) 

213 else: 

214 return stdout

216def exp_restore(db_name, data, copy=False): 

217 def chunks(d, n=8192): 

218 for i in range(0, len(d), n): 

219 yield d[i:i+n] 

220 data_file = tempfile.NamedTemporaryFile(delete=False) 

221 try: 

222 for chunk in chunks(data): 

223 data_file.write(chunk.decode('base64')) 

224 data_file.close() 

225 restore_db(db_name, data_file.name, copy=copy) 

226 finally: 

227 os.unlink(data_file.name) 

228 return True

241 # v8 format

243 # only extract known members! 

244 filestore = [m for m in z.namelist() if m.startswith('filestore/')] 

245 z.extractall(dump_dir, ['dump.sql'] + filestore)

326def list_dbs(force=False): 

327 if not odoo.tools.config['list_db'] and not force: 

328 raise odoo.exceptions.AccessDenied() 

329 

330 if not odoo.tools.config['dbfilter'] and odoo.tools.config['db_name']: 

331 # In case --db-filter is not provided and --database is passed, Odoo will not 

332 # fetch the list of databases available on the postgres server and instead will 

333 # use the value of --database as comma seperated list of exposed databases. 

334 res = sorted(db.strip() for db in odoo.tools.config['db_name'].split(',')) 

335 return res 

336 

337 chosen_template = odoo.tools.config['db_template'] 

338 templates_list = tuple(set(['postgres', chosen_template])) 

339 db = odoo.sql_db.db_connect('postgres') 

340 with closing(db.cursor()) as cr: 

341 try: 

342 db_user = odoo.tools.config["db_user"] 

343 if not db_user and os.name == 'posix': 

344 import pwd 

345 db_user = pwd.getpwuid(os.getuid())[0] 

346 if not db_user: 

347 cr.execute("select usename from pg_user where usesysid=(select datdba from pg_database where datname=%s)", (odoo.tools.config["db_name"],)) 

348 res = cr.fetchone() 

349 db_user = res and str(res[0]) 

350 if db_user: 

351 cr.execute("select datname from pg_database where datdba=(select usesysid from pg_user where usename=%s) and not datistemplate and datallowconn and datname not in %s order by datname", (db_user, templates_list)) 

352 else: 

353 cr.execute("select datname from pg_database where not datistemplate and datallowconn and datname not in %s order by datname", (templates_list,)) 

354 res = [odoo.tools.ustr(name) for (name,) in cr.fetchall()] 

355 except Exception: 

356 res = [] 

357 res.sort() 

358 return res
