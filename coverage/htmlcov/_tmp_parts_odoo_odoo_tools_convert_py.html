69def _fix_multiple_roots(node): 

70 """ 

71 Surround the children of the ``node`` element of an XML field with a 

72 single root "data" element, to prevent having a document with multiple 

73 roots once parsed separately. 

74 

75 XML nodes should have one root only, but we'd like to support 

76 direct multiple roots in our partial documents (like inherited view architectures). 

77 As a convention we'll surround multiple root with a container "data" element, to be

574 rec_string = rec.get("string",'').encode('utf8') or 'unknown' 

575 

576 records = None 

577 eval_dict = {'ref': self.id_get} 

578 context = self.get_context(data_node, rec, eval_dict) 

579 uid = self.get_uid(data_node, rec) 

580 env = self.env(user=uid, context=context) 

581 if rec_id: 

582 records = env[rec_model].browse(self.id_get(rec_id)) 

583 elif rec_src: 

584 q = safe_eval(rec_src, eval_dict) 

585 records = env[rec_model].search(q) 

586 if rec_src_count: 

587 count = int(rec_src_count) 

588 if len(records) != count: 

589 self.assertion_report.record_failure() 

590 msg = 'assertion "%s" failed!\n' \ 

591 ' Incorrect search count:\n' \ 

592 ' expected count: %d\n' \ 

593 ' obtained count: %d\n' \ 

594 % (rec_string, count, len(records)) 

595 _logger.error(msg) 

596 return 

597 

598 assert records is not None,\ 

599 'You must give either an id or a search criteria' 

600 ref = self.id_get 

601 for record in records: 

602 globals_dict = RecordDictWrapper(record) 

603 globals_dict['floatEqual'] = self._assert_equals 

604 globals_dict['ref'] = ref 

605 globals_dict['_ref'] = ref 

606 for test in rec.findall('./test'): 

607 f_expr = test.get("expr",'').encode('utf-8') 

608 env = self.env(user=uid, context=context) 

609 expected_value = _eval_xml(self, test, env) or True

612 self.assertion_report.record_failure() 

613 msg = 'assertion "%s" failed!\n' \ 

614 ' xmltag: %s\n' \ 

615 ' expected value: %r\n' \ 

616 ' obtained value: %r\n' \ 

617 % (rec_string, etree.tostring(test), expected_value, expression_value) 

618 _logger.error(msg) 

619 return 

620 else: # all tests were successful for this assertion tag (no break) 

621 self.assertion_report.record_success()

623 def _tag_record(self, rec, data_node=None, mode=None): 

624 rec_model = rec.get("model").encode('ascii') 

625 model = self.env[rec_model] 

626 rec_id = rec.get("id",'').encode('ascii') 

627 rec_context = rec.get("context", {}) 

628 if rec_context: 

629 rec_context = safe_eval(rec_context) 

630 

631 if self.xml_filename and rec_id: 

632 rec_context['install_mode_data'] = dict( 

633 xml_file=self.xml_filename, 

634 xml_id=rec_id, 

635 model=rec_model, 

636 module=self.module 

637 ) 

638 

639 self._test_xml_id(rec_id) 

640 # in update mode, the record won't be updated if the data node explicitely 

641 # opt-out using @noupdate="1". A second check will be performed in 

642 # ir.model.data#_update() using the record's ir.model.data `noupdate` field. 

643 if self.isnoupdate(data_node) and self.mode != 'init': 

644 # check if the xml record has no id, skip 

645 if not rec_id: 

646 return None 

647 

648 if '.' in rec_id: 

649 module,rec_id2 = rec_id.split('.') 

650 else: 

651 module = self.module 

652 rec_id2 = rec_id 

653 id = self.env['ir.model.data']._update_dummy(rec_model, module, rec_id2) 

654 if id: 

655 # if the resource already exists, don't update it but store 

656 # its database id (can be useful) 

657 self.idref[rec_id] = int(id) 

658 return None 

659 elif not self.nodeattr2bool(rec, 'forcecreate', True): 

660 # if it doesn't exist and we shouldn't create it, skip it 

661 return None 

662 # else create it normally 

663 

664 res = {} 

665 for field in rec.findall('./field'): 

666 #TODO: most of this code is duplicated above (in _eval_xml)... 

667 f_name = field.get("name").encode('utf-8') 

668 f_ref = field.get("ref",'').encode('utf-8') 

669 f_search = field.get("search",'').encode('utf-8') 

670 f_model = field.get("model",'').encode('utf-8') 

671 if not f_model and f_name in model._fields: 

672 f_model = model._fields[f_name].comodel_name 

673 f_use = field.get("use",'').encode('utf-8') or 'id' 

674 f_val = False 

675 

676 if f_search: 

677 q = safe_eval(f_search, self.idref) 

678 assert f_model, 'Define an attribute model="..." in your .XML file !' 

679 # browse the objects searched 

680 s = self.env[f_model].search(q) 

681 # column definitions of the "local" object 

682 _fields = self.env[rec_model]._fields 

683 # if the current field is many2many 

684 if (f_name in _fields) and _fields[f_name].type == 'many2many': 

685 f_val = [(6, 0, map(lambda x: x[f_use], s))] 

686 elif len(s): 

687 # otherwise (we are probably in a many2one field), 

688 # take the first element of the search 

689 f_val = s[0][f_use] 

690 elif f_ref: 

691 if f_name in model._fields and model._fields[f_name].type == 'reference': 

692 val = self.model_id_get(f_ref) 

693 f_val = val[0] + ',' + str(val[1]) 

694 else: 

695 f_val = self.id_get(f_ref) 

696 else: 

697 f_val = _eval_xml(self, field, self.env) 

698 if f_name in model._fields: 

699 if model._fields[f_name].type == 'integer': 

700 f_val = int(f_val) 

701 elif model._fields[f_name].type in ['float', 'monetary']: 

702 f_val = float(f_val) 

703 elif model._fields[f_name].type == 'boolean' and isinstance(f_val, basestring): 

704 f_val = str2bool(f_val) 

705 res[f_name] = f_val 

706 

707 id = self.env(context=rec_context)['ir.model.data']._update(rec_model, self.module, res, rec_id or False, not self.isnoupdate(data_node), noupdate=self.isnoupdate(data_node), mode=self.mode) 

708 if rec_id: 

709 self.idref[rec_id] = int(id) 

710 if config.get('import_partial'): 

711 self.cr.commit() 

712 return rec_model, id

714 def _tag_template(self, el, data_node=None, mode=None): 

715 # This helper transforms a <template> element into a <record> and forwards it 

716 tpl_id = el.get('id', el.get('t-name', '')).encode('ascii') 

717 full_tpl_id = tpl_id 

718 if '.' not in full_tpl_id: 

719 full_tpl_id = '%s.%s' % (self.module, tpl_id) 

720 # set the full template name for qweb <module>.<id> 

721 if not el.get('inherit_id'): 

722 el.set('t-name', full_tpl_id) 

723 el.tag = 't' 

724 else: 

725 el.tag = 'data' 

726 el.attrib.pop('id', None) 

727 

728 record_attrs = { 

729 'id': tpl_id, 

730 'model': 'ir.ui.view', 

731 } 

732 for att in ['forcecreate', 'context']: 

733 if att in el.keys(): 

734 record_attrs[att] = el.attrib.pop(att) 

735 

736 Field = builder.E.field 

737 name = el.get('name', tpl_id) 

738 

739 record = etree.Element('record', attrib=record_attrs) 

740 record.append(Field(name, name='name')) 

741 record.append(Field(full_tpl_id, name='key')) 

742 record.append(Field("qweb", name='type')) 

743 if 'priority' in el.attrib: 

744 record.append(Field(el.get('priority'), name='priority')) 

745 if 'inherit_id' in el.attrib: 

746 record.append(Field(name='inherit_id', ref=el.get('inherit_id'))) 

747 if 'website_id' in el.attrib: 

748 record.append(Field(name='website_id', ref=el.get('website_id'))) 

749 if 'key' in el.attrib: 

750 record.append(Field(el.get('key'), name='key')) 

751 if el.get('active') in ("True", "False"): 

752 view_id = self.id_get(tpl_id, raise_if_not_found=False) 

753 if mode != "update" or not view_id: 

754 record.append(Field(name='active', eval=el.get('active'))) 

755 if el.get('customize_show') in ("True", "False"): 

756 record.append(Field(name='customize_show', eval=el.get('customize_show'))) 

757 groups = el.attrib.pop('groups', None) 

758 if groups: 

759 grp_lst = map(lambda x: "ref('%s')" % x, groups.split(',')) 

760 record.append(Field(name="groups_id", eval="[(6, 0, ["+', '.join(grp_lst)+"])]")) 

761 if el.attrib.pop('page', None) == 'True': 

762 record.append(Field(name="page", eval="True")) 

763 if el.get('primary') == 'True': 

764 # Pseudo clone mode, we'll set the t-name to the full canonical xmlid 

765 el.append( 

766 builder.E.xpath( 

767 builder.E.attribute(full_tpl_id, name='t-name'), 

768 expr=".", 

769 position="attributes", 

770 ) 

771 ) 

772 record.append(Field('primary', name='mode')) 

773 # inject complete <template> element (after changing node name) into 

774 # the ``arch`` field 

775 record.append(Field(el, name="arch", type="xml")) 

776 

777 return self._tag_record(record, data_node)

791 roots = ['openerp','data','odoo'] 

792 if de.tag not in roots: 

793 raise Exception("Root xml tag must be <openerp>, <odoo> or <data>.")

795 if rec.tag in roots:

797 elif rec.tag in self._tags:

802 exc_info = sys.exc_info() 

803 raise ParseError, (ustr(e), etree.tostring(rec).rstrip(), rec.getroottree().docinfo.URL, rec.sourceline), exc_info[2] 

804 return True

856def convert_csv_import(cr, module, fname, csvcontent, idref=None, mode='init', 

857 noupdate=False): 

858 '''Import csv file : 

859 quote: " 

860 delimiter: , 

861 encoding: utf-8''' 

862 if not idref: 

863 idref={} 

864 model = ('.'.join(fname.split('.')[:-1]).split('-'))[0] 

865 #remove folder path from model 

866 head, model = os.path.split(model) 

867 

868 input = cStringIO.StringIO(csvcontent) #FIXME 

869 reader = csv.reader(input, quotechar='"', delimiter=',') 

870 fields = reader.next() 

871 

872 if not (mode == 'init' or 'id' in fields): 

873 _logger.error("Import specification does not contain 'id' and we are in init mode, Cannot continue.") 

874 return 

875 

876 datas = [] 

877 for line in reader: 

878 if not (line and any(line)): 

879 continue 

880 try: 

881 datas.append(map(ustr, line)) 

882 except Exception: 

883 _logger.error("Cannot import the line: %s", line) 

884 

885 context = { 

886 'mode': mode, 

887 'module': module, 

888 'noupdate': noupdate, 

889 } 

890 env = odoo.api.Environment(cr, SUPERUSER_ID, context) 

891 result = env[model].load(fields, datas) 

892 if any(msg['type'] == 'error' for msg in result['messages']): 

893 # Report failed import and abort module install 

894 warning_msg = "\n".join(msg['message'] for msg in result['messages']) 

895 raise Exception(_('Module loading %s failed: file %s could not be processed:\n %s') % (module, fname, warning_msg))
