102 @api.model 

103 def process_email_queue(self, ids=None): 

104 """Send immediately queued messages, committing after each 

105 message is sent - this is not transactional and should 

106 not be called during another transaction! 

107 

108 :param list ids: optional list of emails ids to send. If passed 

109 no search is performed, and these ids are used 

110 instead. 

111 :param dict context: if a 'filters' key is present in context, 

112 this value will be used as an additional 

113 filter to further restrict the outgoing 

114 messages to send (by default all 'outgoing' 

115 messages are sent). 

116 """ 

117 if not self.ids: 

118 filters = ['&', 

119 ('state', '=', 'outgoing'), 

120 '|', 

121 ('scheduled_date', '<', datetime.datetime.now()), 

122 ('scheduled_date', '=', False)] 

123 if 'filters' in self._context: 

124 filters.extend(self._context['filters']) 

125 ids = self.search(filters).ids 

126 res = None 

127 try: 

128 # auto-commit except in testing mode 

129 auto_commit = not getattr(threading.currentThread(), 'testing', False) 

130 res = self.browse(ids).send(auto_commit=auto_commit) 

131 except Exception: 

132 _logger.exception("Failed processing mail queue") 

133 return res

135 @api.multi 

136 def _postprocess_sent_message(self, mail_sent=True): 

137 """Perform any post-processing necessary after sending ``mail`` 

138 successfully, including deleting it completely along with its 

139 attachment if the ``auto_delete`` flag of the mail was set. 

140 Overridden by subclasses for extra post-processing behaviors. 

141 

142 :param browse_record mail: the mail that was just sent 

143 :return: True 

144 """ 

145 notif_emails = self.filtered(lambda email: email.notification) 

146 if notif_emails: 

147 notifications = self.env['mail.notification'].search([ 

148 ('mail_message_id', 'in', notif_emails.mapped('mail_message_id').ids), 

149 ('is_email', '=', True)]) 

150 if mail_sent: 

151 notifications.write({ 

152 'email_status': 'sent', 

153 }) 

154 else: 

155 notifications.write({ 

156 'email_status': 'exception', 

157 }) 

158 if mail_sent: 

159 self.sudo().filtered(lambda self: self.auto_delete).unlink() 

160 return True

204 @api.multi 

205 def send(self, auto_commit=False, raise_exception=False): 

206 """ Sends the selected emails immediately, ignoring their current 

207 state (mails that have already been sent should not be passed 

208 unless they should actually be re-sent). 

209 Emails successfully delivered are marked as 'sent', and those 

210 that fail to be deliver are marked as 'exception', and the 

211 corresponding error mail is output in the server logs. 

212 

213 :param bool auto_commit: whether to force a commit of the mail status 

214 after sending each mail (meant only for scheduler processing); 

215 should never be True during normal transactions (default: False) 

216 :param bool raise_exception: whether to raise an exception if the 

217 email sending process has failed 

218 :return: True 

219 """ 

220 IrMailServer = self.env['ir.mail_server'] 

221 

222 for mail_id in self.ids: 

223 try: 

224 mail = self.browse(mail_id) 

225 # TDE note: remove me when model_id field is present on mail.message - done here to avoid doing it multiple times in the sub method 

226 if mail.model: 

227 model = self.env['ir.model'].sudo().search([('model', '=', mail.model)])[0] 

228 else: 

229 model = None 

230 if model: 

231 mail = mail.with_context(model_name=model.name) 

232 

233 # load attachment binary data with a separate read(), as prefetching all 

234 # `datas` (binary field) could bloat the browse cache, triggerring 

235 # soft/hard mem limits with temporary data. 

236 attachments = [(a['datas_fname'], base64.b64decode(a['datas'])) 

237 for a in mail.attachment_ids.sudo().read(['datas_fname', 'datas'])] 

238 

239 # specific behavior to customize the send email for notified partners 

240 email_list = [] 

241 if mail.email_to: 

242 email_list.append(mail.send_get_email_dict()) 

243 for partner in mail.recipient_ids: 

244 email_list.append(mail.send_get_email_dict(partner=partner)) 

245 

246 # headers 

247 headers = {} 

248 bounce_alias = self.env['ir.config_parameter'].get_param("mail.bounce.alias") 

249 catchall_domain = self.env['ir.config_parameter'].get_param("mail.catchall.domain") 

250 if bounce_alias and catchall_domain: 

251 if mail.model and mail.res_id: 

252 headers['Return-Path'] = '%s+%d-%s-%d@%s' % (bounce_alias, mail.id, mail.model, mail.res_id, catchall_domain) 

253 else: 

254 headers['Return-Path'] = '%s+%d@%s' % (bounce_alias, mail.id, catchall_domain) 

255 if mail.headers: 

256 try: 

257 headers.update(safe_eval(mail.headers)) 

258 except Exception: 

259 pass 

260 

261 # Writing on the mail object may fail (e.g. lock on user) which 

262 # would trigger a rollback *after* actually sending the email. 

263 # To avoid sending twice the same email, provoke the failure earlier 

264 mail.write({ 

265 'state': 'exception', 

266 'failure_reason': _('Error without exception. Probably due do sending an email without computed recipients.'), 

267 }) 

268 mail_sent = False 

269 

270 # build an RFC2822 email.message.Message object and send it without queuing 

271 res = None 

272 for email in email_list: 

273 msg = IrMailServer.build_email( 

274 email_from=mail.email_from, 

275 email_to=email.get('email_to'), 

276 subject=mail.subject, 

277 body=email.get('body'), 

278 body_alternative=email.get('body_alternative'), 

279 email_cc=tools.email_split(mail.email_cc), 

280 reply_to=mail.reply_to, 

281 attachments=attachments, 

282 message_id=mail.message_id, 

283 references=mail.references, 

284 object_id=mail.res_id and ('%s-%s' % (mail.res_id, mail.model)), 

285 subtype='html', 

286 subtype_alternative='plain', 

287 headers=headers) 

288 try: 

289 res = IrMailServer.send_email(msg, mail_server_id=mail.mail_server_id.id) 

290 except AssertionError as error: 

291 if error.message == IrMailServer.NO_VALID_RECIPIENT: 

292 # No valid recipient found for this particular 

293 # mail item -> ignore error to avoid blocking 

294 # delivery to next recipients, if any. If this is 

295 # the only recipient, the mail will show as failed. 

296 _logger.info("Ignoring invalid recipients for mail.mail %s: %s", 

297 mail.message_id, email.get('email_to')) 

298 else: 

299 raise 

300 if res: 

301 mail.write({'state': 'sent', 'message_id': res, 'failure_reason': False}) 

302 mail_sent = True 

303 

304 # /!\ can't use mail.state here, as mail.refresh() will cause an error 

305 # see revid:odo@openerp.com-20120622152536-42b2s28lvdv3odyr in 6.1 

306 if mail_sent: 

307 _logger.info('Mail with ID %r and Message-Id %r successfully sent', mail.id, mail.message_id) 

308 mail._postprocess_sent_message(mail_sent=mail_sent) 

309 except MemoryError: 

310 # prevent catching transient MemoryErrors, bubble up to notify user or abort cron job 

311 # instead of marking the mail as failed 

312 _logger.exception( 

313 'MemoryError while processing mail with ID %r and Msg-Id %r. Consider raising the --limit-memory-hard startup option', 

314 mail.id, mail.message_id) 

315 raise 

316 except psycopg2.Error: 

317 # If an error with the database occurs, chances are that the cursor is unusable. 

318 # This will lead to an `psycopg2.InternalError` being raised when trying to write 

319 # `state`, shadowing the original exception and forbid a retry on concurrent 

320 # update. Let's bubble it. 

321 raise 

322 except Exception as e: 

323 failure_reason = tools.ustr(e) 

324 _logger.exception('failed sending mail (id: %s) due to %s', mail.id, failure_reason) 

325 mail.write({'state': 'exception', 'failure_reason': failure_reason}) 

326 mail._postprocess_sent_message(mail_sent=False) 

327 if raise_exception: 

328 if isinstance(e, AssertionError): 

329 # get the args of the original error, wrap into a value and throw a MailDeliveryException 

330 # that is an except_orm, with name and value as arguments 

331 value = '. '.join(e.args) 

332 raise MailDeliveryException(_("Mail Delivery Failed"), value) 

333 raise 

334 

335 if auto_commit is True: 

336 self._cr.commit() 

337 return True
