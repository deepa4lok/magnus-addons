19class MyOption (optparse.Option, object): 

20 """ optparse Option with two additional attributes. 

21 

22 The list of command line options (getopt.Option) is used to create the 

23 list of the configuration file options. When reading the file, and then 

24 reading the command line arguments, we don't want optparse.parse results 

25 to override the configuration file values. But if we provide default 

26 values to optparse, optparse will return them and we can't know if they 

27 were really provided by the user or not. A solution is to not use 

28 optparse's default attribute, but use a custom one (that will be copied 

29 to create the default values of the configuration file). 

30 

31 """ 

32 def __init__(self, *opts, **attrs): 

33 self.my_default = attrs.pop('my_default', None) 

34 super(MyOption, self).__init__(*opts, **attrs)

49def _deduplicate_loggers(loggers): 

50 """ Avoid saving multiple logging levels for the same loggers to a save 

51 file, that just takes space and the list can potentially grow unbounded 

52 if for some odd reason people use :option`--save`` all the time. 

53 """ 

54 # dict(iterable) -> the last item of iterable for any given key wins, 

55 # which is what we want and expect. Output order should not matter as 

56 # there are no duplicates within the output sequence 

57 return ( 

58 '{}:{}'.format(logger, level) 

59 for logger, level in dict(it.split(':') for it in loggers).iteritems() 

60 )

495 ad_paths = [] 

496 for path in value.split(','): 

497 path = path.strip() 

498 res = os.path.abspath(os.path.expanduser(path)) 

499 if not os.path.isdir(res):

507 def load(self): 

508 p = ConfigParser.ConfigParser() 

509 try: 

510 p.read([self.rcfile]) 

511 for (name,value) in p.items('options'): 

512 if value=='True' or value=='true': 

513 value = True 

514 if value=='False' or value=='false': 

515 value = False 

516 self.options[name] = value 

517 #parse the other sections, as well 

518 for sec in p.sections(): 

519 if sec == 'options': 

520 continue 

521 if not self.misc.has_key(sec): 

522 self.misc[sec]= {} 

523 for (name, value) in p.items(sec): 

524 if value=='True' or value=='true': 

525 value = True 

526 if value=='False' or value=='false': 

527 value = False 

528 self.misc[sec][name] = value 

529 except IOError: 

530 pass 

531 except ConfigParser.NoSectionError: 

532 pass

534 def save(self): 

535 p = ConfigParser.ConfigParser() 

536 loglevelnames = dict(zip(self._LOGLEVELS.values(), self._LOGLEVELS.keys())) 

537 p.add_section('options') 

538 for opt in sorted(self.options.keys()): 

539 if opt in ('version', 'language', 'translate_out', 'translate_in', 'overwrite_existing_translations', 'init', 'update'): 

540 continue 

541 if opt in self.blacklist_for_save: 

542 continue 

543 if opt in ('log_level',): 

544 p.set('options', opt, loglevelnames.get(self.options[opt], self.options[opt])) 

545 elif opt == 'log_handler': 

546 p.set('options', opt, ','.join(_deduplicate_loggers(self.options[opt]))) 

547 else: 

548 p.set('options', opt, self.options[opt]) 

549 

550 for sec in sorted(self.misc.keys()): 

551 p.add_section(sec) 

552 for opt in sorted(self.misc[sec].keys()): 

553 p.set(sec,opt,self.misc[sec][opt]) 

554 

555 # try to create the directories and write the file 

556 try: 

557 rc_exists = os.path.exists(self.rcfile) 

558 if not rc_exists and not os.path.exists(os.path.dirname(self.rcfile)): 

559 os.makedirs(os.path.dirname(self.rcfile)) 

560 try: 

561 p.write(file(self.rcfile, 'w')) 

562 if not rc_exists: 

563 os.chmod(self.rcfile, 0600) 

564 except IOError: 

565 sys.stderr.write("ERROR: couldn't write the config file\n") 

566 

567 except OSError: 

568 # what to do if impossible? 

569 sys.stderr.write("ERROR: couldn't create the config directory\n")
