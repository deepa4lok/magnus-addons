21__doc__ = """The module for manipulating texts and their attributes. 

22 

23Pychart supports extensive sets of attributes in texts. All attributes 

24are specified via "escape sequences", starting from letter "/". For 

25example, the below examples draws string "Hello" using a 12-point font 

26at 60-degree angle: 

27 

28/12/a60{}Hello 

29 

30List of attributes: 

31 

32/hA 

33 Specifies horizontal alignment of the text. A is one of L (left 

34 alignment), R (right alignment), or C (center alignment). 

35/vA 

36 Specifies vertical alignment of the text. A is one of "B" 

37 (bottom), "T" (top), " M" (middle). 

38 

39/F{FONT} 

40 Switch to FONT font family. 

41/T 

42 Shorthand of /F{Times-Roman}. 

43/H 

44 Shorthand of /F{Helvetica}. 

45/C 

46 Shorthand of /F{Courier}. 

47/B 

48 Shorthand of /F{Bookman-Demi}. 

49/A 

50 Shorthand of /F{AvantGarde-Book}. 

51/P 

52 Shorthand of /F{Palatino}. 

53/S 

54 Shorthand of /F{Symbol}. 

55/b 

56 Switch to bold typeface. 

57/i 

58 Switch to italic typeface. 

59/o 

60 Switch to oblique typeface. 

61/DD 

62 Set font size to DD points. 

63 

64 /20{}2001 space odyssey! 

65 

66/cDD 

67 Set gray-scale to 0.DD. Gray-scale of 00 means black, 99 means white. 

68 

69//, /{, /} 

70 Display `/', `{', or `}'. 

71  

72{ ... } 

73 Limit the effect of escape sequences. For example, the below 

74 example draws "Foo" at 12pt, "Bar" at 8pt, and "Baz" at 12pt. 

75 

76 /12Foo{/8Bar}Baz 

77\n 

78 Break the line. 

79"""

92def _intern_afm(font, text): 

93 global _undefined_font_warned 

94 

95 r = afm.dir.afm.get(font, None) 

96 if r: return r 

97 

98 font2 = _font_aliases.get(font, None) 

99 if font2: 

100 r = afm.dir.afm.get(font2, None) 

101 if r: return r 

102 

103 try: 

104 exec("import pychart.afm.%s" % re.sub("-", "_", font)) 

105 return afm.dir.afm[font] 

106 except: 

107 if not font2 and not _undefined_font_warned.has_key(font): 

108 pychart_util.warn('Warning: unknown font "%s" while parsing "%s"' 

109 % (font, text)) 

110 _undefined_font_warned[font] = 1 

111 

112 if font2: 

113 try: 

114 exec('import pychart.afm.%s' % re.sub('-', '_', font2)) 

115 return afm.dir.afm[font2] 

116 except: 

117 if not _undefined_font_warned.has_key(font): 

118 pychart_util.warn('Warning: unknown font "%s" while parsing "%s"' % (font, text)) 

119 _undefined_font_warned[font] = 1 

120 return None 

121def line_width(font, size, text): 

122 table = _intern_afm(font, text) 

123 if not table: 

124 return 0 

125 

126 width = 0 

127 for ch in text: 

128 code = ord(ch) 

129 if code < len(table): 

130 width += table[code] 

131 else: 

132 # Invalid char. Make up a number. 

133 width += table[0] 

134 

135 width = float(width) * size / 1000.0 

136 return width

360def unaligned_get_dimension(text): 

361 """Return the bounding box of the text, assuming that the left-bottom corner 

362 of the first letter of the text is at (0, 0). This procedure ignores 

363 /h, /v, and /a directives when calculating the BB; it just returns the 

364 alignment specifiers as a part of the return value. The return value is a 

365 tuple (width, height, halign, valign, angle).""" 

366 

367 xmax = 0 

368 ymax = 0 

369 ymax = 0 

370 angle = None 

371 halign = None 

372 valign = None 

373 itr = text_iterator(None) 

374 for line in unicode(text).split('\n'): 

375 cur_height = 0 

376 cur_width = 0 

377 itr.reset(line) 

378 while 1: 

379 elem = itr.next() 

380 if not elem: 

381 break 

382 (font, size, line_height, color, new_h, new_v, new_a, chunk) = elem 

383 if halign != None and new_h != halign: 

384 raise FontException('Only one "/h" can appear in a string.', 

385 unicode(text)) 

386 if valign != None and new_v != valign: 

387 raise FontException('Only one "/v" can appear in a string.', 

388 unicode(text)) 

389 if angle != None and new_a != angle: 

390 raise FontException('Only one "/a" can appear in a string.', 

391 unicode(text)) 

392 halign = new_h 

393 valign = new_v 

394 angle = new_a 

395 cur_width += line_width(font, size, chunk) 

396 cur_height = max(cur_height, line_height) 

397 xmax = max(cur_width, xmax) 

398 ymax += cur_height 

399 return (xmax, ymax, 

400 halign or theme.default_font_halign, 

401 valign or theme.default_font_valign, 

402 angle or theme.default_font_angle)

404def get_dimension(text): 

405 """Return the bounding box of the <text>, 

406 assuming that the left-bottom corner 

407 of the first letter of the text is at (0, 0). This procedure ignores 

408 /h, /v, and /a directives when calculating the boundingbox; it just returns the 

409 alignment specifiers as a part of the return value. The return value is a 

410 tuple (width, height, halign, valign, angle).""" 

411 (xmax, ymax, halign, valign, angle) = unaligned_get_dimension(text) 

412 xmin = ymin = 0 

413 if halign == 'C': 

414 xmin = -xmax / 2.0 

415 xmax = xmax / 2.0 

416 elif halign == 'R': 

417 xmin = -xmax 

418 xmax = 0 

419 if valign == 'M': 

420 ymin = -ymax / 2.0 

421 ymax = ymax / 2.0 

422 elif valign == 'T': 

423 ymin = -ymax 

424 ymax = 0 

425 if angle != 0: 

426 (x0, y0) = pychart_util.rotate(xmin, ymin, angle) 

427 (x1, y1) = pychart_util.rotate(xmax, ymin, angle) 

428 (x2, y2) = pychart_util.rotate(xmin, ymax, angle) 

429 (x3, y3) = pychart_util.rotate(xmax, ymax, angle) 

430 xmax = max(x0, x1, x2, x3) 

431 xmin = min(x0, x1, x2, x3) 

432 ymax = max(y0, y1, y2, y3) 

433 ymin = min(y0, y1, y2, y3) 

434 return (xmin, xmax, ymin, ymax) 

435 return (xmin, xmax, ymin, ymax)

461def quotemeta(text): 

462 """Quote letters with special meanings in pychart so that <text> will display 

463 as-is when passed to canvas.show().  

464 

465>>> font.quotemeta("foo/bar") 

466"foo//bar" 

467""" 

468 text = re.sub(r'/', '//', text) 

469 text = re.sub(r'\\{', '/{', text) 

470 text = re.sub(r'\\}', '/}', text) 

471 return text
