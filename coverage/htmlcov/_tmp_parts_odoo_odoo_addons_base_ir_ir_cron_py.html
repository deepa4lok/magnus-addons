101 @api.model 

102 def _callback(self, model_name, method_name, args, job_id): 

103 """ Run the method associated to a given job 

104 

105 It takes care of logging and exception handling. 

106 

107 :param model_name: model name on which the job method is located. 

108 :param method_name: name of the method to call when this job is processed. 

109 :param args: arguments of the method (without the usual self, cr, uid). 

110 :param job_id: job id. 

111 """ 

112 try: 

113 args = str2tuple(args) 

114 if self.pool != self.pool.check_signaling(): 

115 # the registry has changed, reload self in the new registry 

116 self.env.reset() 

117 self = self.env()[self._name] 

118 if model_name in self.env: 

119 model = self.env[model_name] 

120 if hasattr(model, method_name): 

121 log_depth = (None if _logger.isEnabledFor(logging.DEBUG) else 1) 

122 odoo.netsvc.log(_logger, logging.DEBUG, 'cron.object.execute', (self._cr.dbname, self._uid, '*', model_name, method_name)+tuple(args), depth=log_depth) 

123 start_time = False 

124 if _logger.isEnabledFor(logging.DEBUG): 

125 start_time = time.time() 

126 getattr(model, method_name)(*args) 

127 if start_time and _logger.isEnabledFor(logging.DEBUG): 

128 end_time = time.time() 

129 _logger.debug('%.3fs (%s, %s)', end_time - start_time, model_name, method_name) 

130 self.pool.signal_caches_change() 

131 else: 

132 _logger.warning("Method '%s.%s' does not exist.", model_name, method_name) 

133 else: 

134 _logger.warning("Model %r does not exist.", model_name) 

135 except Exception, e: 

136 _logger.exception("Call of self.env[%r].%s(*%r) failed in Job #%s", 

137 model_name, method_name, args, job_id) 

138 self._handle_callback_exception(model_name, method_name, args, job_id, e)

140 @classmethod 

141 def _process_job(cls, job_cr, job, cron_cr): 

142 """ Run a given job taking care of the repetition. 

143 

144 :param job_cr: cursor to use to execute the job, safe to commit/rollback 

145 :param job: job to be run (as a dictionary). 

146 :param cron_cr: cursor holding lock on the cron job row, to use to update the next exec date, 

147 must not be committed/rolled back! 

148 """ 

149 try: 

150 with api.Environment.manage(): 

151 cron = api.Environment(job_cr, job['user_id'], {})[cls._name] 

152 # Use the user's timezone to compare and compute datetimes, 

153 # otherwise unexpected results may appear. For instance, adding 

154 # 1 month in UTC to July 1st at midnight in GMT+2 gives July 30 

155 # instead of August 1st! 

156 now = fields.Datetime.context_timestamp(cron, datetime.now()) 

157 nextcall = fields.Datetime.context_timestamp(cron, fields.Datetime.from_string(job['nextcall'])) 

158 numbercall = job['numbercall'] 

159 

160 ok = False 

161 while nextcall < now and numbercall: 

162 if numbercall > 0: 

163 numbercall -= 1 

164 if not ok or job['doall']: 

165 cron._callback(job['model'], job['function'], job['args'], job['id']) 

166 if numbercall: 

167 nextcall += _intervalTypes[job['interval_type']](job['interval_number']) 

168 ok = True 

169 addsql = '' 

170 if not numbercall: 

171 addsql = ', active=False' 

172 cron_cr.execute("UPDATE ir_cron SET nextcall=%s, numbercall=%s"+addsql+" WHERE id=%s", 

173 (fields.Datetime.to_string(nextcall.astimezone(pytz.UTC)), numbercall, job['id'])) 

174 cron.invalidate_cache() 

175 

176 finally: 

177 job_cr.commit() 

178 cron_cr.commit()

180 @classmethod 

181 def _process_jobs(cls, db_name): 

182 """ Try to process all cron jobs. 

183 

184 This selects in database all the jobs that should be processed. It then 

185 tries to lock each of them and, if it succeeds, run the cron job (if it 

186 doesn't succeed, it means the job was already locked to be taken care 

187 of by another thread) and return. 

188 

189 :raise BadVersion: if the version is different from the worker's 

190 :raise BadModuleState: if modules are to install/upgrade/remove 

191 """ 

192 db = odoo.sql_db.db_connect(db_name) 

193 threading.current_thread().dbname = db_name 

194 try: 

195 with db.cursor() as cr: 

196 # Make sure the database has the same version as the code of 

197 # base and that no module must be installed/upgraded/removed 

198 cr.execute("SELECT latest_version FROM ir_module_module WHERE name=%s", ['base']) 

199 (version,) = cr.fetchone() 

200 cr.execute("SELECT COUNT(*) FROM ir_module_module WHERE state LIKE %s", ['to %']) 

201 (changes,) = cr.fetchone() 

202 if version is None: 

203 raise BadModuleState() 

204 elif version != BASE_VERSION: 

205 raise BadVersion() 

206 # Careful to compare timestamps with 'UTC' - everything is UTC as of v6.1. 

207 cr.execute("""SELECT * FROM ir_cron 

208 WHERE numbercall != 0 

209 AND active AND nextcall <= (now() at time zone 'UTC') 

210 ORDER BY priority""") 

211 jobs = cr.dictfetchall() 

212 

213 if changes: 

214 if not jobs: 

215 raise BadModuleState() 

216 # nextcall is never updated if the cron is not executed, 

217 # it is used as a sentinel value to check whether cron jobs 

218 # have been locked for a long time (stuck) 

219 parse = fields.Datetime.from_string 

220 oldest = min([parse(job['nextcall']) for job in jobs]) 

221 if datetime.now() - oldest > MAX_FAIL_TIME: 

222 odoo.modules.reset_modules_state(db_name) 

223 else: 

224 raise BadModuleState() 

225 

226 for job in jobs: 

227 lock_cr = db.cursor() 

228 try: 

229 # Try to grab an exclusive lock on the job row from within the task transaction 

230 # Restrict to the same conditions as for the search since the job may have already 

231 # been run by an other thread when cron is running in multi thread 

232 lock_cr.execute("""SELECT * 

233 FROM ir_cron 

234 WHERE numbercall != 0 

235 AND active 

236 AND nextcall <= (now() at time zone 'UTC') 

237 AND id=%s 

238 FOR UPDATE NOWAIT""", 

239 (job['id'],), log_exceptions=False) 

240 

241 locked_job = lock_cr.fetchone() 

242 if not locked_job: 

243 _logger.debug("Job `%s` already executed by another process/thread. skipping it", job['name']) 

244 continue 

245 # Got the lock on the job row, run its code 

246 _logger.info('Starting job `%s`.', job['name']) 

247 job_cr = db.cursor() 

248 try: 

249 registry = odoo.registry(db_name) 

250 registry[cls._name]._process_job(job_cr, job, lock_cr) 

251 except Exception: 

252 _logger.exception('Unexpected exception while processing cron job %r', job) 

253 finally: 

254 job_cr.close() 

255 

256 except psycopg2.OperationalError, e: 

257 if e.pgcode == '55P03': 

258 # Class 55: Object not in prerequisite state; 55P03: lock_not_available 

259 _logger.debug('Another process/thread is already busy executing job `%s`, skipping it.', job['name']) 

260 continue 

261 else: 

262 # Unexpected OperationalError 

263 raise 

264 finally: 

265 # we're exiting due to an exception while acquiring the lock 

266 lock_cr.close() 

267 

268 finally: 

269 if hasattr(threading.current_thread(), 'dbname'): 

270 del threading.current_thread().dbname

272 @classmethod 

273 def _acquire_job(cls, db_name): 

274 """ Try to process all cron jobs. 

275 

276 This selects in database all the jobs that should be processed. It then 

277 tries to lock each of them and, if it succeeds, run the cron job (if it 

278 doesn't succeed, it means the job was already locked to be taken care 

279 of by another thread) and return. 

280 

281 This method hides most exceptions related to the database's version, the 

282 modules' state, and such. 

283 """ 

284 try: 

285 cls._process_jobs(db_name) 

286 except BadVersion: 

287 _logger.warning('Skipping database %s as its base version is not %s.', db_name, BASE_VERSION) 

288 except BadModuleState: 

289 _logger.warning('Skipping database %s because of modules to install/upgrade/remove.', db_name) 

290 except psycopg2.ProgrammingError as e: 

291 if e.pgcode == '42P01': 

292 # Class 42 — Syntax Error or Access Rule Violation; 42P01: undefined_table 

293 # The table ir_cron does not exist; this is probably not an OpenERP database. 

294 _logger.warning('Tried to poll an undefined table on database %s.', db_name) 

295 else: 

296 raise 

297 except Exception: 

298 _logger.warning('Exception in cron:', exc_info=True)

300 @api.multi 

301 def _try_lock(self): 

302 """Try to grab a dummy exclusive write-lock to the rows with the given ids, 

303 to make sure a following write() or unlink() will not block due 

304 to a process currently executing those cron tasks""" 

305 try: 

306 self._cr.execute("""SELECT id FROM "%s" WHERE id IN %%s FOR UPDATE NOWAIT""" % self._table, 

307 [tuple(self.ids)], log_exceptions=False) 

308 except psycopg2.OperationalError: 

309 self._cr.rollback() # early rollback to allow translations to work for the user feedback 

310 raise UserError(_("Record cannot be modified right now: " 

311 "This cron task is currently being executed and may not be modified " 

312 "Please try again in a few minutes"))
