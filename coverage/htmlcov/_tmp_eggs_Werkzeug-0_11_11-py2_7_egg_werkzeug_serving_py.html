1# -*- coding: utf-8 -*- 

2""" 

3 werkzeug.serving 

4 ~~~~~~~~~~~~~~~~ 

5 

6 There are many ways to serve a WSGI application. While you're developing 

7 it you usually don't want a full blown webserver like Apache but a simple 

8 standalone one. From Python 2.5 onwards there is the `wsgiref`_ server in 

9 the standard library. If you're using older versions of Python you can 

10 download the package from the cheeseshop. 

11 

12 However there are some caveats. Sourcecode won't reload itself when 

13 changed and each time you kill the server using ``^C`` you get an 

14 `KeyboardInterrupt` error. While the latter is easy to solve the first 

15 one can be a pain in the ass in some situations. 

16 

17 The easiest way is creating a small ``start-myproject.py`` that runs the 

18 application:: 

19 

20 #!/usr/bin/env python 

21 # -*- coding: utf-8 -*- 

22 from myproject import make_app 

23 from werkzeug.serving import run_simple 

24 

25 app = make_app(...) 

26 run_simple('localhost', 8080, app, use_reloader=True) 

27 

28 You can also pass it a `extra_files` keyword argument with a list of 

29 additional files (like configuration files) you want to observe. 

30 

31 For bigger applications you should consider using `werkzeug.script` 

32 instead of a simple start file. 

33 

34 

35 :copyright: (c) 2014 by the Werkzeug Team, see AUTHORS for more details. 

36 :license: BSD, see LICENSE for more details. 

37"""

91 def make_environ(self): 

92 request_url = url_parse(self.path)

94 def shutdown_server(): 

95 self.server.shutdown_signal = True 

96 

97 url_scheme = self.server.ssl_context is None and 'http' or 'https' 

98 path_info = url_unquote(request_url.path) 

99 

100 environ = { 

101 'wsgi.version': (1, 0), 

102 'wsgi.url_scheme': url_scheme, 

103 'wsgi.input': self.rfile, 

104 'wsgi.errors': sys.stderr, 

105 'wsgi.multithread': self.server.multithread, 

106 'wsgi.multiprocess': self.server.multiprocess, 

107 'wsgi.run_once': False, 

108 'werkzeug.server.shutdown': shutdown_server, 

109 'SERVER_SOFTWARE': self.server_version, 

110 'REQUEST_METHOD': self.command, 

111 'SCRIPT_NAME': '', 

112 'PATH_INFO': wsgi_encoding_dance(path_info), 

113 'QUERY_STRING': wsgi_encoding_dance(request_url.query), 

114 'CONTENT_TYPE': self.headers.get('Content-Type', ''), 

115 'CONTENT_LENGTH': self.headers.get('Content-Length', ''), 

116 'REMOTE_ADDR': self.address_string(), 

117 'REMOTE_PORT': self.port_integer(), 

118 'SERVER_NAME': self.server.server_address[0], 

119 'SERVER_PORT': str(self.server.server_address[1]), 

120 'SERVER_PROTOCOL': self.request_version 

121 } 

122 

123 for key, value in self.headers.items(): 

124 key = 'HTTP_' + key.upper().replace('-', '_') 

125 if key not in ('HTTP_CONTENT_TYPE', 'HTTP_CONTENT_LENGTH'): 

126 environ[key] = value 

127 

128 if request_url.scheme and request_url.netloc: 

129 environ['HTTP_HOST'] = request_url.netloc 

130 

131 return environ

133 def run_wsgi(self): 

134 if self.headers.get('Expect', '').lower().strip() == '100-continue': 

135 self.wfile.write(b'HTTP/1.1 100 Continue\r\n\r\n') 

136 

137 self.environ = environ = self.make_environ() 

138 headers_set = [] 

139 headers_sent = [] 

140 

141 def write(data): 

142 assert headers_set, 'write() before start_response' 

143 if not headers_sent: 

144 status, response_headers = headers_sent[:] = headers_set 

145 try: 

146 code, msg = status.split(None, 1) 

147 except ValueError: 

148 code, msg = status, "" 

149 self.send_response(int(code), msg) 

150 header_keys = set() 

151 for key, value in response_headers: 

152 self.send_header(key, value) 

153 key = key.lower() 

154 header_keys.add(key) 

155 if 'content-length' not in header_keys: 

156 self.close_connection = True 

157 self.send_header('Connection', 'close') 

158 if 'server' not in header_keys: 

159 self.send_header('Server', self.version_string()) 

160 if 'date' not in header_keys: 

161 self.send_header('Date', self.date_time_string()) 

162 self.end_headers() 

163 

164 assert isinstance(data, bytes), 'applications must write bytes' 

165 self.wfile.write(data) 

166 self.wfile.flush() 

167 

168 def start_response(status, response_headers, exc_info=None): 

169 if exc_info: 

170 try: 

171 if headers_sent: 

172 reraise(*exc_info) 

173 finally: 

174 exc_info = None 

175 elif headers_set: 

176 raise AssertionError('Headers already set') 

177 headers_set[:] = [status, response_headers] 

178 return write 

179 

180 def execute(app): 

181 application_iter = app(environ, start_response) 

182 try: 

183 for data in application_iter: 

184 write(data) 

185 if not headers_sent: 

186 write(b'') 

187 finally: 

188 if hasattr(application_iter, 'close'): 

189 application_iter.close() 

190 application_iter = None 

191 

192 try: 

193 execute(self.server.app) 

194 except (socket.error, socket.timeout) as e: 

195 self.connection_dropped(e, environ) 

196 except Exception: 

197 if self.server.passthrough_errors: 

198 raise 

199 from werkzeug.debug.tbtools import get_current_traceback 

200 traceback = get_current_traceback(ignore_system_exceptions=True) 

201 try: 

202 # if we haven't yet sent the headers but they are set 

203 # we roll back to be able to set them again. 

204 if not headers_sent: 

205 del headers_set[:] 

206 execute(InternalServerError()) 

207 except Exception: 

208 pass 

209 self.server.log('error', 'Error on request:\n%s', 

210 traceback.plaintext)

212 def handle(self): 

213 """Handles a request ignoring dropped connections.""" 

214 rv = None 

215 try:

218 self.connection_dropped(e) 

219 except Exception:

221 raise

290def generate_adhoc_ssl_pair(cn=None): 

291 from random import random 

292 crypto = _get_openssl_crypto_module() 

293 

294 # pretty damn sure that this is not actually accepted by anyone 

295 if cn is None: 

296 cn = '*' 

297 

298 cert = crypto.X509() 

299 cert.set_serial_number(int(random() * sys.maxsize)) 

300 cert.gmtime_adj_notBefore(0) 

301 cert.gmtime_adj_notAfter(60 * 60 * 24 * 365) 

302 

303 subject = cert.get_subject() 

304 subject.CN = cn 

305 subject.O = 'Dummy Certificate' 

306 

307 issuer = cert.get_issuer() 

308 issuer.CN = 'Untrusted Authority' 

309 issuer.O = 'Self-Signed' 

310 

311 pkey = crypto.PKey() 

312 pkey.generate_key(crypto.TYPE_RSA, 1024) 

313 cert.set_pubkey(pkey) 

314 cert.sign(pkey, 'md5') 

315 

316 return cert, pkey

319def make_ssl_devcert(base_path, host=None, cn=None): 

320 """Creates an SSL key for development. This should be used instead of 

321 the ``'adhoc'`` key which generates a new cert on each server start. 

322 It accepts a path for where it should store the key and cert and 

323 either a host or CN. If a host is given it will use the CN 

324 ``*.host/CN=host``. 

325 

326 For more information see :func:`run_simple`. 

327 

328 .. versionadded:: 0.9 

329 

330 :param base_path: the path to the certificate and key. The extension 

331 ``.crt`` is added for the certificate, ``.key`` is 

332 added for the key. 

333 :param host: the name of the host. This can be used as an alternative 

334 for the `cn`. 

335 :param cn: the `CN` to use. 

336 """ 

337 from OpenSSL import crypto 

338 if host is not None: 

339 cn = '*.%s/CN=%s' % (host, host) 

340 cert, pkey = generate_adhoc_ssl_pair(cn=cn) 

341 

342 cert_file = base_path + '.crt' 

343 pkey_file = base_path + '.key' 

344 

345 with open(cert_file, 'wb') as f: 

346 f.write(crypto.dump_certificate(crypto.FILETYPE_PEM, cert)) 

347 with open(pkey_file, 'wb') as f: 

348 f.write(crypto.dump_privatekey(crypto.FILETYPE_PEM, pkey)) 

349 

350 return cert_file, pkey_file

353def generate_adhoc_ssl_context(): 

354 """Generates an adhoc SSL context for the development server.""" 

355 crypto = _get_openssl_crypto_module() 

356 import tempfile 

357 import atexit 

358 

359 cert, pkey = generate_adhoc_ssl_pair() 

360 cert_handle, cert_file = tempfile.mkstemp() 

361 pkey_handle, pkey_file = tempfile.mkstemp() 

362 atexit.register(os.remove, pkey_file) 

363 atexit.register(os.remove, cert_file) 

364 

365 os.write(cert_handle, crypto.dump_certificate(crypto.FILETYPE_PEM, cert)) 

366 os.write(pkey_handle, crypto.dump_privatekey(crypto.FILETYPE_PEM, pkey)) 

367 os.close(cert_handle) 

368 os.close(pkey_handle) 

369 ctx = load_ssl_context(cert_file, pkey_file) 

370 return ctx

562def run_simple(hostname, port, application, use_reloader=False, 

563 use_debugger=False, use_evalex=True, 

564 extra_files=None, reloader_interval=1, 

565 reloader_type='auto', threaded=False, 

566 processes=1, request_handler=None, static_files=None, 

567 passthrough_errors=False, ssl_context=None): 

568 """Start a WSGI application. Optional features include a reloader, 

569 multithreading and fork support. 

570 

571 This function has a command-line interface too:: 

572 

573 python -m werkzeug.serving --help 

574 

575 .. versionadded:: 0.5 

576 `static_files` was added to simplify serving of static files as well 

577 as `passthrough_errors`. 

578 

579 .. versionadded:: 0.6 

580 support for SSL was added. 

581 

582 .. versionadded:: 0.8 

583 Added support for automatically loading a SSL context from certificate 

584 file and private key. 

585 

586 .. versionadded:: 0.9 

587 Added command-line interface. 

588 

589 .. versionadded:: 0.10 

590 Improved the reloader and added support for changing the backend 

591 through the `reloader_type` parameter. See :ref:`reloader` 

592 for more information. 

593 

594 :param hostname: The host for the application. eg: ``'localhost'`` 

595 :param port: The port for the server. eg: ``8080`` 

596 :param application: the WSGI application to execute 

597 :param use_reloader: should the server automatically restart the python 

598 process if modules were changed? 

599 :param use_debugger: should the werkzeug debugging system be used? 

600 :param use_evalex: should the exception evaluation feature be enabled? 

601 :param extra_files: a list of files the reloader should watch 

602 additionally to the modules. For example configuration 

603 files. 

604 :param reloader_interval: the interval for the reloader in seconds. 

605 :param reloader_type: the type of reloader to use. The default is 

606 auto detection. Valid values are ``'stat'`` and 

607 ``'watchdog'``. See :ref:`reloader` for more 

608 information. 

609 :param threaded: should the process handle each request in a separate 

610 thread? 

611 :param processes: if greater than 1 then handle each request in a new process 

612 up to this maximum number of concurrent processes. 

613 :param request_handler: optional parameter that can be used to replace 

614 the default one. You can use this to replace it 

615 with a different 

616 :class:`~BaseHTTPServer.BaseHTTPRequestHandler` 

617 subclass. 

618 :param static_files: a dict of paths for static files. This works exactly 

619 like :class:`SharedDataMiddleware`, it's actually 

620 just wrapping the application in that middleware before 

621 serving. 

622 :param passthrough_errors: set this to `True` to disable the error catching. 

623 This means that the server will die on errors but 

624 it can be useful to hook debuggers in (pdb etc.) 

625 :param ssl_context: an SSL context for the connection. Either an 

626 :class:`ssl.SSLContext`, a tuple in the form 

627 ``(cert_file, pkey_file)``, the string ``'adhoc'`` if 

628 the server should automatically create one, or ``None`` 

629 to disable SSL (which is the default). 

630 """ 

631 if use_debugger: 

632 from werkzeug.debug import DebuggedApplication 

633 application = DebuggedApplication(application, use_evalex) 

634 if static_files: 

635 from werkzeug.wsgi import SharedDataMiddleware 

636 application = SharedDataMiddleware(application, static_files)
