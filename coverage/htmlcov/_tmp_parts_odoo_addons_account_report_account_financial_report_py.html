18 '''Returns a dictionary with key=the ID of a record and value = the level of this

20 for report in self:

26 def _get_children_by_order(self): 

27 '''returns a recordset of all the children computed recursively, and sorted by sequence. Ready for the printing''' 

28 res = self 

29 children = self.search([('parent_id', 'in', self.ids)], order='sequence ASC') 

30 if children: 

31 for child in children: 

32 res += child._get_children_by_order() 

33 return res
