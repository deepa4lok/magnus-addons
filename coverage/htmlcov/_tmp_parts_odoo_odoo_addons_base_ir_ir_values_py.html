24class IrValues(models.Model): 

25 """Holds internal model-specific action bindings and user-defined default 

26 field values. definitions. This is a legacy internal model, mixing 

27 two different concepts, and will likely be updated or replaced in a 

28 future version by cleaner, separate models. You should not depend 

29 explicitly on it. 

30 

31 The purpose of each ``ir.values`` entry depends on its type, defined 

32 by the ``key`` column: 

33 

34 * 'default': user-defined default values, used when creating new 

35 records of this model: 

36 * 'action': binding of an action to a particular *action slot* of 

37 this model, making the action easily available in the user 

38 interface for this model. 

39 

40 The ``key2`` column acts as a qualifier, further refining the type 

41 of the entry. The possible values are: 

42 

43 * for 'default' entries: an optional condition restricting the 

44 cases where this particular default value will be applicable, 

45 or ``False`` for no condition 

46 * for 'action' entries: the ``key2`` qualifier is one of the available 

47 action slots, defining how this action can be invoked: 

48 

49 * ``'client_print_multi'`` for report printing actions that will 

50 be available on views displaying items from this model 

51 * ``'client_action_multi'`` for assistants (wizards) actions 

52 that will be available in views displaying objects of this model 

53 * ``'client_action_relate'`` for links towards related documents 

54 that should be available in views displaying objects of this model 

55 * ``'tree_but_open'`` for actions that will be triggered when 

56 double-clicking an item from this model in a hierarchical tree view 

57 

58 Each entry is specific to a model (``model`` column), and for ``'actions'`` 

59 type, may even be made specific to a given record of that model when the 

60 ``res_id`` column contains a record ID (``False`` means it's global for 

61 all records). 

62 

63 The content of the entry is defined by the ``value`` column, which may either 

64 contain an arbitrary value, or a reference string defining the action that 

65 should be executed. 

66 

67 .. rubric:: Usage: default values 

68  

69 The ``'default'`` entries are usually defined manually by the 

70 users, and set by their UI clients calling :meth:`~.set_default`. 

71 These default values are then automatically used by the 

72 ORM every time a new record is about to be created, i.e. when 

73 :meth:`~odoo.models.Model.default_get` 

74 or :meth:`~odoo.models.Model.create` are called. 

75 

76 .. rubric:: Usage: action bindings 

77 

78 Business applications will usually bind their actions during 

79 installation, and Odoo UI clients will apply them as defined, 

80 based on the list of actions included in the result of 

81 :meth:`~odoo.models.Model.fields_view_get`, 

82 or directly returned by explicit calls to :meth:`~.get_actions`. 

83 """

122 @api.depends('key', 'value') 

123 def _value_unpickle(self): 

124 for record in self: 

125 value = record.value 

126 if record.key == 'default' and value: 

127 # default values are pickled on the fly 

128 with tools.ignore(Exception): 

129 value = str(pickle.loads(value)) 

130 record.value_unpickle = value

132 def _value_pickle(self): 

133 context = dict(self._context) 

134 context.pop(self.CONCURRENCY_CHECK_FIELD, None) 

135 for record in self.with_context(context): 

136 value = record.value_unpickle 

137 # Only char-like fields should be written directly. Other types should be converted to 

138 # their appropriate type first. 

139 if record.model in self.env and record.name in self.env[record.model]._fields: 

140 field = self.env[record.model]._fields[record.name] 

141 if field.type not in ['char', 'text', 'html', 'selection']: 

142 value = literal_eval(value) 

143 if record.key == 'default': 

144 value = pickle.dumps(value) 

145 record.value = value

182 def set_default(self, model, field_name, value, for_all_users=True, company_id=False, condition=False): 

183 """Defines a default value for the given model and field_name. Any previous 

184 default for the same scope (model, field_name, value, for_all_users, company_id, condition) 

185 will be replaced and lost in the process. 

186 

187 Defaults can be later retrieved via :meth:`~.get_defaults`, which will return 

188 the highest priority default for any given field. Defaults that are more specific 

189 have a higher priority, in the following order (highest to lowest): 

190 

191 * specific to user and company 

192 * specific to user only 

193 * specific to company only 

194 * global to everyone 

195 

196 :param string model: model name 

197 :param string field_name: field name to which the default applies 

198 :param value: the default field value to set 

199 :type value: any serializable Python value 

200 :param bool for_all_users: whether the default should apply to everybody or only 

201 the user calling the method 

202 :param int company_id: optional ID of the company to which the default should 

203 apply. If omitted, the default will be global. If True 

204 is passed, the current user's company will be used. 

205 :param string condition: optional condition specification that can be used to 

206 restrict the applicability of the default values 

207 (e.g. based on another field's value). This is an 

208 opaque string as far as the API is concerned, but client 

209 stacks typically use single-field conditions in the 

210 form ``'key=stringified_value'``. 

211 (Currently, the condition is trimmed to 200 characters, 

212 so values that share the same first 200 characters always 

213 match) 

214 :return: the newly created ir.values entry 

215 """ 

216 if isinstance(value, unicode): 

217 value = value.encode('utf8') 

218 if company_id is True: 

219 # should be company-specific, need to get company id 

220 company_id = self.env.user.company_id.id 

221 

222 # remove existing defaults for the same scope 

223 search_criteria = [ 

224 ('key', '=', 'default'), 

225 ('key2', '=', condition and condition[:200]), 

226 ('model', '=', model), 

227 ('name', '=', field_name), 

228 ('user_id', '=', False if for_all_users else self._uid), 

229 ('company_id', '=', company_id) 

230 ] 

231 self.search(search_criteria).unlink() 

232 

233 return self.create({ 

234 'name': field_name, 

235 'value': pickle.dumps(value), 

236 'model': model, 

237 'key': 'default', 

238 'key2': condition and condition[:200], 

239 'user_id': False if for_all_users else self._uid, 

240 'company_id': company_id, 

241 })

259 @api.model 

260 def get_defaults(self, model, condition=False): 

261 """Returns any default values that are defined for the current model and user, 

262 (and match ``condition``, if specified), previously registered via 

263 :meth:`~.set_default`. 

264 

265 Defaults are global to a model, not field-specific, but an optional 

266 ``condition`` can be provided to restrict matching default values 

267 to those that were defined for the same condition (usually based 

268 on another field's value). 

269 

270 Default values also have priorities depending on whom they apply 

271 to: only the highest priority value will be returned for any 

272 field. See :meth:`~.set_default` for more details. 

273 

274 :param string model: model name 

275 :param string condition: optional condition specification that can be used to 

276 restrict the applicability of the default values 

277 (e.g. based on another field's value). This is an 

278 opaque string as far as the API is concerned, but client 

279 stacks typically use single-field conditions in the 

280 form ``'key=stringified_value'``. 

281 (Currently, the condition is trimmed to 200 characters, 

282 so values that share the same first 200 characters always 

283 match) 

284 :return: list of default values tuples of the form ``(id, field_name, value)`` 

285 (``id`` is the ID of the default entry, usually irrelevant) 

286 """ 

287 # use a direct SQL query for performance reasons, 

288 # this is called very often 

289 query = """ SELECT v.id, v.name, v.value FROM ir_values v 

290 LEFT JOIN res_users u ON (v.user_id = u.id) 

291 WHERE v.key = %%s AND v.model = %%s 

292 AND (v.user_id = %%s OR v.user_id IS NULL) 

293 AND (v.company_id IS NULL OR 

294 v.company_id = (SELECT company_id FROM res_users WHERE id = %%s) 

295 ) 

296 %s 

297 ORDER BY v.user_id, u.company_id""" 

298 params = ('default', model, self._uid, self._uid) 

299 if condition: 

300 query = query % 'AND v.key2 = %s' 

301 params += (condition[:200],) 

302 else: 

303 query = query % 'AND v.key2 IS NULL' 

304 self._cr.execute(query, params) 

305 

306 # keep only the highest priority default for each field 

307 defaults = {} 

308 for row in self._cr.dictfetchall(): 

309 value = pickle.loads(row['value'].encode('utf-8')) 

310 defaults.setdefault(row['name'], (row['id'], row['name'], value)) 

311 return defaults.values()

325 def set_action(self, name, action_slot, model, action, res_id=False): 

326 """Binds an the given action to the given model's action slot - for later 

327 retrieval via :meth:`~.get_actions`. Any existing binding of the same action 

328 to the same slot is first removed, allowing an update of the action's name. 

329 See the class description for more details about the various action 

330 slots: :class:`~ir_values`. 

331 

332 :param string name: action label, usually displayed by UI client 

333 :param string action_slot: the action slot to which the action should be 

334 bound to - one of ``client_action_multi``, 

335 ``client_print_multi``, ``client_action_relate``, 

336 ``tree_but_open``. 

337 :param string model: model name 

338 :param string action: action reference, in the form ``'model,id'`` 

339 :param int res_id: optional record id - will bind the action only to a 

340 specific record of the model, not all records. 

341 :return: the newly created ir.values entry 

342 """ 

343 assert isinstance(action, basestring) and ',' in action, \ 

344 'Action definition must be an action reference, e.g. "ir.actions.act_window,42"' 

345 assert action_slot in ACTION_SLOTS, \ 

346 'Action slot (%s) must be one of: %r' % (action_slot, ACTION_SLOTS) 

347 

348 # remove existing action definition of same slot and value 

349 search_criteria = [ 

350 ('key', '=', 'action'), 

351 ('key2', '=', action_slot), 

352 ('model', '=', model), 

353 ('res_id', '=', res_id or 0), # int field -> NULL == 0 

354 ('value', '=', action), 

355 ] 

356 self.search(search_criteria).unlink() 

357 

358 return self.create({ 

359 'key': 'action', 

360 'key2': action_slot, 

361 'model': model, 

362 'res_id': res_id, 

363 'name': name, 

364 'value': action, 

365 })

369 def get_actions(self, action_slot, model, res_id=False): 

370 """Retrieves the list of actions bound to the given model's action slot. 

371 See the class description for more details about the various action 

372 slots: :class:`~.ir_values`. 

373 

374 :param string action_slot: the action slot to which the actions should be 

375 bound to - one of ``client_action_multi``, 

376 ``client_print_multi``, ``client_action_relate``, 

377 ``tree_but_open``. 

378 :param string model: model name 

379 :param int res_id: optional record id - will bind the action only to a 

380 specific record of the model, not all records. 

381 :return: list of action tuples of the form ``(id, name, action_def)``, 

382 where ``id`` is the ID of the default entry, ``name`` is the 

383 action label, and ``action_def`` is a dict containing the 

384 action definition as obtained by calling 

385 :meth:`~odoo.models.Model.read` on the action record. 

386 """ 

387 assert action_slot in ACTION_SLOTS, 'Illegal action slot value: %s' % action_slot 

388 # use a direct SQL query for performance reasons, 

389 # this is called very often 

390 query = """ SELECT v.id, v.name, v.value FROM ir_values v 

391 WHERE v.key = %s AND v.key2 = %s AND v.model = %s 

392 AND (v.res_id = %s OR v.res_id IS NULL OR v.res_id = 0) 

393 ORDER BY v.id """ 

394 self._cr.execute(query, ('action', action_slot, model, res_id or None)) 

395 

396 # map values to their corresponding action record 

397 actions = [] 

398 for id, name, value in self._cr.fetchall(): 

399 if not value: 

400 continue # skip if undefined 

401 action_model, action_id = value.split(',') 

402 if action_model not in self.env: 

403 continue # unknown model? skip it! 

404 action = self.env[action_model].browse(int(action_id)) 

405 actions.append((id, name, action)) 

406 

407 # process values and their action 

408 results = {} 

409 for id, name, action in actions: 

410 fields = [field for field in action._fields if field not in EXCLUDED_FIELDS] 

411 # FIXME: needs cleanup 

412 try: 

413 action_def = { 

414 field: action._fields[field].convert_to_read(action[field], action) 

415 for field in fields 

416 } 

417 if action._name in ('ir.actions.report.xml', 'ir.actions.act_window'): 

418 if action.groups_id and not action.groups_id & self.env.user.groups_id: 

419 if name == 'Menuitem': 

420 raise AccessError(_('You do not have the permission to perform this operation!!!')) 

421 continue 

422 # keep only the last action registered for each action name 

423 results[name] = (id, name, action_def) 

424 except (AccessError, MissingError): 

425 continue 

426 return sorted(results.values())
