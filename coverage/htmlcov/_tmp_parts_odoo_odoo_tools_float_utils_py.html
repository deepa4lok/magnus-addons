6def _float_check_precision(precision_digits=None, precision_rounding=None): 

7 assert (precision_digits is not None or precision_rounding is not None) and \ 

8 not (precision_digits and precision_rounding),\ 

9 "exactly one of precision_digits and precision_rounding must be specified" 

10 if precision_digits is not None: 

11 return 10 ** -precision_digits 

12 return precision_rounding

14def float_round(value, precision_digits=None, precision_rounding=None, rounding_method='HALF-UP'): 

15 """Return ``value`` rounded to ``precision_digits`` decimal digits, 

16 minimizing IEEE-754 floating point representation errors, and applying 

17 the tie-breaking rule selected with ``rounding_method``, by default 

18 HALF-UP (away from zero). 

19 Precision must be given by ``precision_digits`` or ``precision_rounding``, 

20 not both! 

21 

22 :param float value: the value to round 

23 :param int precision_digits: number of fractional digits to round to. 

24 :param float precision_rounding: decimal number representing the minimum 

25 non-zero value at the desired precision (for example, 0.01 for a  

26 2-digit precision). 

27 :param rounding_method: the rounding method used: 'HALF-UP' or 'UP', the first 

28 one rounding up to the closest number with the rule that number>=0.5 is  

29 rounded up to 1, and the latest one always rounding up. 

30 :return: rounded float 

31 """ 

32 rounding_factor = _float_check_precision(precision_digits=precision_digits, 

33 precision_rounding=precision_rounding) 

34 if rounding_factor == 0 or value == 0: return 0.0 

35 

36 # NORMALIZE - ROUND - DENORMALIZE 

37 # In order to easily support rounding to arbitrary 'steps' (e.g. coin values), 

38 # we normalize the value before rounding it as an integer, and de-normalize 

39 # after rounding: e.g. float_round(1.3, precision_rounding=.5) == 1.5 

40 

41 # TIE-BREAKING: HALF-UP (for normal rounding) 

42 # We want to apply HALF-UP tie-breaking rules, i.e. 0.5 rounds away from 0. 

43 # Due to IEE754 float/double representation limits, the approximation of the 

44 # real value may be slightly below the tie limit, resulting in an error of 

45 # 1 unit in the last place (ulp) after rounding. 

46 # For example 2.675 == 2.6749999999999998. 

47 # To correct this, we add a very small epsilon value, scaled to the 

48 # the order of magnitude of the value, to tip the tie-break in the right 

49 # direction. 

50 # Credit: discussion with OpenERP community members on bug 882036 

51 

52 normalized_value = value / rounding_factor # normalize 

53 epsilon_magnitude = math.log(abs(normalized_value), 2) 

54 epsilon = 2**(epsilon_magnitude-53) 

55 if rounding_method == 'HALF-UP': 

56 normalized_value += cmp(normalized_value,0) * epsilon 

57 rounded_value = round(normalized_value) # round to integer 

58 

59 # TIE-BREAKING: UP (for ceiling operations) 

60 # When rounding the value up, we instead subtract the epsilon value 

61 # as the the approximation of the real value may be slightly *above* the 

62 # tie limit, this would result in incorrectly rounding up to the next number 

63 # The math.ceil operation is applied on the absolute value in order to 

64 # round "away from zero" and not "towards infinity", then the sign is 

65 # restored. 

66 

67 elif rounding_method == 'UP': 

68 sign = cmp(normalized_value, 0) 

69 normalized_value -= sign*epsilon 

70 rounded_value = math.ceil(abs(normalized_value))*sign # ceil to integer 

71 

72 result = rounded_value * rounding_factor # de-normalize 

73 return result

75def float_is_zero(value, precision_digits=None, precision_rounding=None): 

76 """Returns true if ``value`` is small enough to be treated as 

77 zero at the given precision (smaller than the corresponding *epsilon*). 

78 The precision (``10**-precision_digits`` or ``precision_rounding``) 

79 is used as the zero *epsilon*: values less than that are considered 

80 to be zero. 

81 Precision must be given by ``precision_digits`` or ``precision_rounding``, 

82 not both!  

83 

84 Warning: ``float_is_zero(value1-value2)`` is not equivalent to 

85 ``float_compare(value1,value2) == 0``, as the former will round after 

86 computing the difference, while the latter will round before, giving 

87 different results for e.g. 0.006 and 0.002 at 2 digits precision.  

88 

89 :param int precision_digits: number of fractional digits to round to. 

90 :param float precision_rounding: decimal number representing the minimum 

91 non-zero value at the desired precision (for example, 0.01 for a  

92 2-digit precision). 

93 :param float value: value to compare with the precision's zero 

94 :return: True if ``value`` is considered zero 

95 """ 

96 epsilon = _float_check_precision(precision_digits=precision_digits, 

97 precision_rounding=precision_rounding) 

98 return abs(float_round(value, precision_rounding=epsilon)) < epsilon

100def float_compare(value1, value2, precision_digits=None, precision_rounding=None): 

101 """Compare ``value1`` and ``value2`` after rounding them according to the 

102 given precision. A value is considered lower/greater than another value 

103 if their rounded value is different. This is not the same as having a 

104 non-zero difference! 

105 Precision must be given by ``precision_digits`` or ``precision_rounding``, 

106 not both! 

107 

108 Example: 1.432 and 1.431 are equal at 2 digits precision, 

109 so this method would return 0 

110 However 0.006 and 0.002 are considered different (this method returns 1) 

111 because they respectively round to 0.01 and 0.0, even though 

112 0.006-0.002 = 0.004 which would be considered zero at 2 digits precision. 

113 

114 Warning: ``float_is_zero(value1-value2)`` is not equivalent to  

115 ``float_compare(value1,value2) == 0``, as the former will round after 

116 computing the difference, while the latter will round before, giving 

117 different results for e.g. 0.006 and 0.002 at 2 digits precision.  

118 

119 :param int precision_digits: number of fractional digits to round to. 

120 :param float precision_rounding: decimal number representing the minimum 

121 non-zero value at the desired precision (for example, 0.01 for a  

122 2-digit precision). 

123 :param float value1: first value to compare 

124 :param float value2: second value to compare 

125 :return: (resp.) -1, 0 or 1, if ``value1`` is (resp.) lower than, 

126 equal to, or greater than ``value2``, at the given precision. 

127 """ 

128 rounding_factor = _float_check_precision(precision_digits=precision_digits, 

129 precision_rounding=precision_rounding) 

130 value1 = float_round(value1, precision_rounding=rounding_factor) 

131 value2 = float_round(value2, precision_rounding=rounding_factor) 

132 delta = value1 - value2 

133 if float_is_zero(delta, precision_rounding=rounding_factor): return 0 

134 return -1 if delta < 0.0 else 1

136def float_repr(value, precision_digits): 

137 """Returns a string representation of a float with the 

138 the given number of fractional digits. This should not be 

139 used to perform a rounding operation (this is done via 

140 :meth:`~.float_round`), but only to produce a suitable 

141 string representation for a float. 

142 

143 :param int precision_digits: number of fractional digits to 

144 include in the output 

145 """ 

146 # Can't use str() here because it seems to have an intrisic 

147 # rounding to 12 significant digits, which causes a loss of 

148 # precision. e.g. str(123456789.1234) == str(123456789.123)!! 

149 return ("%%.%sf" % precision_digits) % value

153class float_precision(float): 

154 """ A class for float values that carry precision digits. This is a thin 

155 layer on top of ``float``, and the precision digits are not propagated 

156 to the result of arithmetic operations. This class is used when 

157 converting monetary values to cache, and for serializing them to the 

158 database. 

159 """ 

160 __slots__ = ['precision_digits'] 

161 

162 def __new__(cls, value, precision_digits): 

163 obj = super(float_precision, cls).__new__(cls, value) 

164 obj.precision_digits = precision_digits 

165 return obj 

166 

167 def float_repr(self): 

168 return _float_repr(self, self.precision_digits)

178 def try_round(amount, expected, precision_digits=3): 

179 global count, errors; count += 1 

180 result = float_repr(float_round(amount, precision_digits=precision_digits),

186 # Extended float range test, inspired by Cloves Almeida's test on bug #882036. 

187 fractions = [.0, .015, .01499, .675, .67499, .4555, .4555, .45555] 

188 expecteds = ['.00', '.02', '.01', '.68', '.67', '.46', '.456', '.4556'] 

189 precisions = [2, 2, 2, 2, 2, 2, 3, 4] 

190 for magnitude in range(7): 

191 for i in xrange(len(fractions)): 

192 frac, exp, prec = fractions[i], expecteds[i], precisions[i] 

193 for sign in [-1,1]: 

194 for x in xrange(0,10000,97): 

195 n = x * 10**magnitude 

196 f = sign * (n + frac) 

197 f_exp = ('-' if f != 0 and sign == -1 else '') + str(n) + exp 

198 try_round(f, f_exp, precision_digits=prec)
