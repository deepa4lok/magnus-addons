94 @api.model 

95 def _get_fpos_by_region(self, country_id=False, state_id=False, zipcode=False, vat_required=False): 

96 if not country_id: 

97 return False 

98 base_domain = [('auto_apply', '=', True), ('vat_required', '=', vat_required)] 

99 if self.env.context.get('force_company'): 

100 base_domain.append(('company_id', '=', self.env.context.get('force_company'))) 

101 null_state_dom = state_domain = [('state_ids', '=', False)] 

102 null_zip_dom = zip_domain = [('zip_from', '=', 0), ('zip_to', '=', 0)] 

103 null_country_dom = [('country_id', '=', False), ('country_group_id', '=', False)] 

104 

105 if zipcode and zipcode.isdigit(): 

106 zipcode = int(zipcode) 

107 zip_domain = [('zip_from', '<=', zipcode), ('zip_to', '>=', zipcode)] 

108 else: 

109 zipcode = 0 

110 

111 if state_id: 

112 state_domain = [('state_ids', '=', state_id)] 

113 

114 domain_country = base_domain + [('country_id', '=', country_id)] 

115 domain_group = base_domain + [('country_group_id.country_ids', '=', country_id)] 

116 

117 # Build domain to search records with exact matching criteria 

118 fpos = self.search(domain_country + state_domain + zip_domain, limit=1) 

119 # return records that fit the most the criteria, and fallback on less specific fiscal positions if any can be found 

120 if not fpos and state_id: 

121 fpos = self.search(domain_country + null_state_dom + zip_domain, limit=1) 

122 if not fpos and zipcode: 

123 fpos = self.search(domain_country + state_domain + null_zip_dom, limit=1) 

124 if not fpos and state_id and zipcode: 

125 fpos = self.search(domain_country + null_state_dom + null_zip_dom, limit=1) 

126 

127 # fallback: country group with no state/zip range 

128 if not fpos: 

129 fpos = self.search(domain_group + null_state_dom + null_zip_dom, limit=1) 

130 

131 if not fpos: 

132 # Fallback on catchall (no country, no group) 

133 fpos = self.search(base_domain + null_country_dom, limit=1) 

134 return fpos or False

136 @api.model 

137 def get_fiscal_position(self, partner_id, delivery_id=None): 

138 if not partner_id: 

139 return False 

140 # This can be easily overriden to apply more complex fiscal rules 

141 PartnerObj = self.env['res.partner'] 

142 partner = PartnerObj.browse(partner_id) 

143 

144 # if no delivery use invoicing 

145 if delivery_id: 

146 delivery = PartnerObj.browse(delivery_id) 

147 else: 

148 delivery = partner 

149 

150 # partner manually set fiscal position always win 

151 if delivery.property_account_position_id or partner.property_account_position_id: 

152 return delivery.property_account_position_id.id or partner.property_account_position_id.id 

153 

154 # First search only matching VAT positions 

155 vat_required = bool(partner.vat) 

156 fp = self._get_fpos_by_region(delivery.country_id.id, delivery.state_id.id, delivery.zip, vat_required) 

157 

158 # Then if VAT required found no match, try positions that do not require it 

159 if not fp and vat_required: 

160 fp = self._get_fpos_by_region(delivery.country_id.id, delivery.state_id.id, delivery.zip, False) 

161 

162 return fp.id if fp else False

229 @api.multi 

230 def _asset_difference_search(self, account_type, operator, operand): 

231 if operator not in ('<', '=', '>', '>=', '<='): 

232 return [] 

233 if type(operand) not in (float, int): 

234 return [] 

235 sign = 1 

236 if account_type == 'payable': 

237 sign = -1 

238 res = self._cr.execute(''' 

239 SELECT partner.id 

240 FROM res_partner partner 

241 LEFT JOIN account_move_line aml ON aml.partner_id = partner.id 

242 RIGHT JOIN account_account acc ON aml.account_id = acc.id 

243 WHERE acc.internal_type = %s 

244 AND NOT acc.deprecated 

245 GROUP BY partner.id 

246 HAVING %s * COALESCE(SUM(aml.amount_residual), 0) ''' + operator + ''' %s''', (account_type, sign, operand)) 

247 res = self._cr.fetchall() 

248 if not res: 

249 return [('id', '=', '0')] 

250 return [('id', 'in', map(itemgetter(0), res))]
