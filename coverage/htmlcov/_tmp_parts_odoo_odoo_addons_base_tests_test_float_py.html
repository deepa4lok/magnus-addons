95 # Extended float range test, inspired by Cloves Almeida's test on bug #882036. 

96 fractions = [.0, .015, .01499, .675, .67499, .4555, .4555, .45555] 

97 expecteds = ['.00', '.02', '.01', '.68', '.67', '.46', '.456', '.4556'] 

98 precisions = [2, 2, 2, 2, 2, 2, 3, 4] 

99 # Note: max precision for double floats is 53 bits of precision or 

100 # 17 significant decimal digits 

101 for magnitude in range(7): 

102 for i in xrange(len(fractions)): 

103 frac, exp, prec = fractions[i], expecteds[i], precisions[i] 

104 for sign in [-1,1]: 

105 for x in xrange(0,10000,97): 

106 n = x * 10**magnitude 

107 f = sign * (n + frac) 

108 f_exp = ('-' if f != 0 and sign == -1 else '') + str(n) + exp 

109 try_round(f, f_exp, digits=prec)
