35def _parse_style_str(s): 

36 """ 

37 Take an SVG 'style' attribute string like 'stroke:none;fill:black' 

38 and parse it into a dictionary like {'stroke' : 'none', 'fill' : 'black'}. 

39 """ 

40 styledict = {} 

41 if s : 

42 # parses L -> R so later keys overwrite earlier ones 

43 for keyval in s.split(';'): 

44 l = keyval.strip().split(':') 

45 if l and len(l) == 2: styledict[l[0].strip()] = l[1].strip() 

46 return styledict

48def _make_style_str(styledict): 

49 """ 

50 Make an SVG style string from the dictionary. See also _parse_style_str also. 

51 """ 

52 s = '' 

53 for key in styledict.keys(): 

54 s += "%s:%s;"%(key, styledict[key]) 

55 return s

57def _protect_current_children(elt): 

58 # If elt is a group, check to see whether there are any non-comment 

59 # children, and if so, create a new group to hold attributes 

60 # to avoid affecting previous children. Return either the current 

61 # elt or the newly generated group. 

62 if (elt.nodeName == 'g') : 

63 for kid in elt.childNodes : 

64 if kid.nodeType != Comment.nodeType: 

65 g = elt.ownerDocument.createElement('g') 

66 g.setAttribute('auto','') 

67 if _comment_p: 

68 g.appendChild(g.ownerDocument.createComment 

69 ('auto-generated group')) 

70 elt.appendChild(g) 

71 elt = g 

72 break 

73 return elt

101 def __update_style(self, **addstyledict): 

102 elt = _protect_current_children(self.__cur_element) 

103 

104 # fetch the current styles for this node 

105 my_style_dict = _parse_style_str(elt.getAttribute('style')) 

106 

107 # concat all ancestor style strings to get default styles for this node 

108 parent, s = elt.parentNode, '' 

109 while parent.nodeType != Document.nodeType : 

110 # prepend parent str so later keys will override earlier ones 

111 s = parent.getAttribute('style') + s 

112 parent = parent.parentNode 

113 default_style_dict = _parse_style_str(s) 

114 

115 # Do some pre-processing on the caller-supplied add'l styles 

116 # Convert '_' to '-' so caller can specify style tags as python 

117 # variable names, eg. stroke_width => stroke-width. 

118 # Also convert all RHS values to strs 

119 for key in addstyledict.keys(): 

120 k = re.sub('_','-',key) 

121 addstyledict[k] = str(addstyledict[key]) # all vals => strs 

122 if (k != key) : del addstyledict[key] 

123 

124 for k in addstyledict.keys() : 

125 if (my_style_dict.has_key(k) or # need to overwrite it 

126 (not default_style_dict.has_key(k)) or # need to set it 

127 default_style_dict[k] != addstyledict[k]) : # need to override it 

128 my_style_dict[k] = addstyledict[k] 

129 

130 s = _make_style_str(my_style_dict) 

131 if s : elt.setAttribute('style',s) 

132 

133 self.__cur_element = elt

161 def path_arc(self, x, y, radius, ratio, start_angle, end_angle): 

162 # mimic PS 'arc' given radius, yr/xr (=eccentricity), start and 

163 # end angles. PS arc draws from CP (if exists) to arc start, 

164 # then draws arc in counterclockwise dir from start to end 

165 # SVG provides an arc command that draws a segment of an 

166 # ellipse (but not a full circle) given these args: 

167 # A xr yr rotate majorArcFlag counterclockwiseFlag xe ye 

168 # We don't use rotate(=0) and flipped axes => all arcs are clockwise 

169 

170 if (self.__cur_element.nodeName != 'path') : 

171 raise OverflowError, "No path for path_arc" 

172 

173 self.comment('x=%g, y=%g, r=%g, :=%g, %g-%g' 

174 % (x,y,radius,ratio,start_angle,end_angle)) 

175 

176 xs = x+radius*math.cos(2*math.pi/360.*start_angle) 

177 ys = y+ratio*radius*math.sin(2*math.pi/360.*start_angle) 

178 xe = x+radius*math.cos(2*math.pi/360.*end_angle) 

179 ye = y+ratio*radius*math.sin(2*math.pi/360.*end_angle) 

180 if (end_angle < start_angle) : # make end bigger than start 

181 while end_angle <= start_angle: # '<=' so 360->0 becomes 360->720 

182 end_angle += 360 

183 full_circ = (end_angle - start_angle >= 360) # draw a full circle? 

184 

185 d = self.__cur_element.getAttribute('d') 

186 d += ' %s %g %g' % (d and 'L' or 'M',xs,-ys) # draw from CP, if exists 

187 if (radius > 0) : # skip, eg. 0-radius 'rounded' corners which blowup 

188 if (full_circ) : 

189 # If we're drawing a full circle, move to the end coord 

190 # and draw half a circle to the reflected xe,ye 

191 d += ' M %g %g A %g %g 0 1 0 %g %g'%(xe,-ye, 

192 radius,radius*ratio, 

193 2*x-xe,-(2*y-ye)) 

194 # Draw arc from the CP (either reflected xe,ye for full circle else 

195 # xs,ys) to the end coord - note with full_circ the 

196 # 'bigArcFlag' value is moot, with exactly 180deg left to draw 

197 d += ' A %g %g 0 %d 0 %g %g' % (radius,radius*ratio, 

198 end_angle-start_angle>180, 

199 xe,-ye) 

200 self.__cur_element.setAttribute('d',d.strip())

273 # PyChart constructs a postscript font name, for example: 

274 # 

275 # Helvetica Helvetica-Bold Helvetica-Oblique Helvetica-BoldOblique 

276 # Helvetica-Narrow Times-Roman Times-Italic 

277 # Symbol Palatino-Roman Bookman-Demi Courier AvantGarde-Book 

278 # 

279 # We need to deconstruct this to get the font-family (the 

280 # piece before the '-'), and other characteristics. 

281 # Note that 'Courier' seems to correspond to SVGs 'CourierNew' 

282 # and that the SVG Symbol font is Unicode where the ascii text 

283 # 'Symbol' doesn't create greek characters like 'Sigma ...' - 

284 # should really pass a unicode string, or provide translation 

285 # 

286 # SVG defines: 

287 # font-style = normal (aka roman) | italic | oblique 

288 # font-weight = normal | bold (aka demi?) 

289 # font-stretch = normal | wider | narrower | ultra-condensed | 

290 # extra-condensed | condensed | semi-condensed | 

291 # semi-expanded | expanded | extra-expanded | ultra-expanded 

292 # ('narrow' seems to correspond to 'condensed') 

293 font_name, modifiers = re.match(r'([^-]*)(-.*)?', font_name).groups() 

294 if font_name == 'Courier' : font_name = 'CourierNew' 

295 font_style = font_weight = font_stretch = 'normal' 

296 if modifiers : 

297 if re.search('Italic',modifiers) : font_style = 'italic' 

298 elif re.search('Oblique',modifiers) : font_style = 'oblique' 

299 if re.search('Bold|Demi',modifiers) : font_weight = 'bold' 

300 if re.search('Narrow',modifiers) : font_stretch = 'condensed' 

301 #! translate ascii symbol font chars -> unicode (see www.unicode.org) 

302 #! http://www.unicode.org/Public/MAPPINGS/VENDORS/ADOBE/symbol.txt 

303 #! but xml Text element writes unicode chars as '?' to XML file... 

304 string = re.sub(r'\\([()])',r'\1',string) # unescape brackets 

305 self.__update_style(fill=_svgcolor(color), 

306 stroke='none', 

307 font_family=font_name, 

308 font_size=size, 

309 font_style=font_style, 

310 font_weight=font_weight, 

311 font_stretch=font_stretch) 

312 self.__cur_element.appendChild(self.__doc.createTextNode(string.encode('utf-8')))

359 def push_transformation(self, baseloc, scale, angle, in_text=0): 

360 #? in_text arg appears to always be ignored 

361 

362 # In some cases this gets called after newpath, with 

363 # corresonding pop_transformation called after the path is 

364 # finalized so we check specifically for that, and generate 

365 # an enclosing group to hold the incomplete path element 

366 # We could add the transform directly to the path element 

367 # (like we do with line-style etc) but that makes it harder 

368 # to handle the closing 'pop' and might lead to inconsitency 

369 # with PostScript if the closing pop doesn't come right after 

370 # the path element 

371 

372 elt = self.__cur_element 

373 if elt.nodeName == 'g': 

374 elt = None 

375 elif (elt.nodeName == 'path' and not elt.hasAttribute('d')) : 

376 g = elt.parentNode 

377 g.removeChild(elt) 

378 self.__cur_element = g 

379 else: 

380 raise ValueError, "Illegal placement of push_transformation" 

381 

382 t = '' 

383 if baseloc : 

384 t += 'translate(%g,%g) '%(baseloc[0],-baseloc[1]) 

385 if angle : 

386 t += 'rotate(%g) '%-angle 

387 if scale : 

388 t += 'scale(%g,%g) '%tuple(scale) 

389 

390 self.gsave() 

391 self.__cur_element.setAttribute('transform',t.strip()) 

392 if elt: # elt has incomplete 'path' or None 

393 self.__cur_element.appendChild(elt) 

394 self.__cur_element = elt

410 # The close() method finalizes the SVG document and flattens the 

411 # DOM document to XML text to the specified file (or stdout) 

412 def close(self): 

413 basecanvas.T.close(self) 

414 self.grestore() # matching the gsave in __init__ 

415 if (self.__cur_element.nodeName != 'svg') : 

416 raise ValueError, "Incomplete document at close!" 

417 

418 # Don't bother to output an empty document - this can happen 

419 # when we get close()d immediately by theme reinit 

420 if (len(self.__svg.childNodes[-1].childNodes) == 0) : 

421 return 

422 

423 fp, need_close = self.open_output(self.__out_fname) 

424 bbox = theme.adjust_bounding_box([self.__xmin, self.__ymin, 

425 self.__xmax, self.__ymax]) 

426 self.__svg.setAttribute('viewBox','%g %g %g %g' 

427 % (xscale(bbox[0]), 

428 -yscale(bbox[3]), 

429 xscale(bbox[2])-xscale(bbox[0]), 

430 yscale(bbox[3])-yscale(bbox[1]))) 

431 self.__svg.setAttribute('xmlns','http://www.w3.org/2000/svg') 

432 self.__svg.setAttribute('xmlns:xlink','http://www.w3.org/1999/xlink')

434 self.__doc.writexml(fp,'',' ','\n') 

435 if need_close: 

436 fp.close()
