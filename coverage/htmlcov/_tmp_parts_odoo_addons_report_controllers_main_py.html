22 @route([ 

23 '/report/<converter>/<reportname>', 

24 '/report/<converter>/<reportname>/<docids>', 

25 ], type='http', auth='user', website=True) 

26 def report_routes(self, reportname, docids=None, converter=None, **data): 

27 report_obj = request.env['report'] 

28 context = dict(request.env.context) 

29 

30 if docids: 

31 docids = [int(i) for i in docids.split(',')] 

32 if data.get('options'): 

33 data.update(json.loads(data.pop('options'))) 

34 if data.get('context'): 

35 # Ignore 'lang' here, because the context in data is the one from the webclient *but* if 

36 # the user explicitely wants to change the lang, this mechanism overwrites it. 

37 data['context'] = json.loads(data['context']) 

38 if data['context'].get('lang'): 

39 del data['context']['lang'] 

40 context.update(data['context']) 

41 if converter == 'html': 

42 html = report_obj.with_context(context).get_html(docids, reportname, data=data) 

43 return request.make_response(html) 

44 elif converter == 'pdf': 

45 pdf = report_obj.with_context(context).get_pdf(docids, reportname, data=data) 

46 pdfhttpheaders = [('Content-Type', 'application/pdf'), ('Content-Length', len(pdf))] 

47 return request.make_response(pdf, headers=pdfhttpheaders) 

48 else: 

49 raise exceptions.HTTPException(description='Converter %s not implemented.' % converter)

54 @route(['/report/barcode', '/report/barcode/<type>/<path:value>'], type='http', auth="public") 

55 def report_barcode(self, type, value, width=600, height=100, humanreadable=0): 

56 """Contoller able to render barcode images thanks to reportlab. 

57 Samples: 

58 <img t-att-src="'/report/barcode/QR/%s' % o.name"/> 

59 <img t-att-src="'/report/barcode/?type=%s&amp;value=%s&amp;width=%s&amp;height=%s' % 

60 ('QR', o.name, 200, 200)"/> 

61 

62 :param type: Accepted types: 'Codabar', 'Code11', 'Code128', 'EAN13', 'EAN8', 'Extended39', 

63 'Extended93', 'FIM', 'I2of5', 'MSI', 'POSTNET', 'QR', 'Standard39', 'Standard93', 

64 'UPCA', 'USPS_4State' 

65 :param humanreadable: Accepted values: 0 (default) or 1. 1 will insert the readable value 

66 at the bottom of the output image 

67 """ 

68 try: 

69 barcode = request.env['report'].barcode(type, value, width=width, height=height, humanreadable=humanreadable) 

70 except (ValueError, AttributeError): 

71 raise exceptions.HTTPException(description='Cannot convert into barcode.') 

72 

73 return request.make_response(barcode, headers=[('Content-Type', 'image/png')])

77 """This function is used by 'qwebactionmanager.js' in order to trigger the download of

79 

80 :param data: a javascript array JSON.stringified containg report internal url ([0]) and 

81 type [1] 

82 :returns: Response with a filetoken cookie and an attachment header

84 requestcontent = json.loads(data)

88 reportname = url.split('/report/pdf/')[1].split('?')[0]

90 docids = None

92 reportname, docids = reportname.split('/') 

93 

94 if docids: 

95 # Generic report:

97 else: 

98 # Particular report: 

99 data = url_decode(url.split('?')[1]).items() # decoding the args represented in JSON 

100 response = self.report_routes(reportname, converter='pdf', **dict(data)) 

101 

102 report = request.env['report']._get_report_from_name(reportname) 

103 filename = "%s.%s" % (report.name, "pdf") 

104 if docids: 

105 ids = [int(x) for x in docids.split(",")] 

106 obj = request.env[report.model].browse(ids) 

107 if report.print_report_name and not len(obj) > 1: 

108 filename = safe_eval(report.print_report_name, {'object': obj, 'time': time}) 

109 response.headers.add('Content-Disposition', content_disposition(filename)) 

110 response.set_cookie('fileToken', token) 

111 return response 

112 elif type == 'controller': 

113 reqheaders = Headers(request.httprequest.headers) 

114 response = Client(request.httprequest.app, BaseResponse).get(url, headers=reqheaders, follow_redirects=True) 

115 response.set_cookie('fileToken', token) 

116 return response 

117 else: 

118 return 

119 except Exception, e: 

120 se = _serialize_exception(e) 

121 error = {
