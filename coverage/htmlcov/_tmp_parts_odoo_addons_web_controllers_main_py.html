100def ensure_db(redirect='/web/database/selector'): 

101 # This helper should be used in web client auth="none" routes 

102 # if those routes needs a db to work with. 

103 # If the heuristics does not find any database, then the users will be 

104 # redirected to db selector or any url specified by `redirect` argument. 

105 # If the db is taken out of a query parameter, it will be checked against 

106 # `http.db_filter()` in order to ensure it's legit and thus avoid db 

107 # forgering that could lead to xss attacks. 

108 db = request.params.get('db') and request.params.get('db').strip() 

109 

110 # Ensure db is legit 

111 if db and db not in http.db_filter([db]): 

112 db = None 

113 

114 if db and not request.session.db: 

115 # User asked a specific database on a new session. 

116 # That mean the nodb router has been used to find the route 

117 # Depending on installed module in the database, the rendering of the page 

118 # may depend on data injected by the database route dispatcher. 

119 # Thus, we redirect the user to the same page but with the session cookie set. 

120 # This will force using the database route dispatcher... 

121 r = request.httprequest 

122 url_redirect = r.base_url 

123 if r.query_string: 

124 # Can't use werkzeug.wrappers.BaseRequest.url with encoded hashes: 

125 # https://github.com/amigrave/werkzeug/commit/b4a62433f2f7678c234cdcac6247a869f90a7eb7 

126 url_redirect += '?' + r.query_string 

127 response = werkzeug.utils.redirect(url_redirect, 302) 

128 request.session.db = db 

129 abort_and_redirect(url_redirect) 

130 

131 # if db not provided, use the session one 

132 if not db and request.session.db and http.db_filter([request.session.db]): 

133 db = request.session.db 

134 

135 # if no database provided and no database in session, use monodb 

136 if not db: 

137 db = db_monodb(request.httprequest) 

138 

139 # if no db can be found til here, send to the database selector 

140 # the database selector will redirect to database manager if needed 

141 if not db: 

142 werkzeug.exceptions.abort(werkzeug.utils.redirect(redirect, 303)) 

143 

144 # always switch the session to the computed db 

145 if db != request.session.db: 

146 request.session.logout() 

147 abort_and_redirect(request.httprequest.url) 

148 

149 request.session.db = db

191def concat_xml(file_list): 

192 """Concatenate xml files 

193 

194 :param list(str) file_list: list of files to check 

195 :returns: (concatenation_result, checksum) 

196 :rtype: (str, str) 

197 """ 

198 checksum = hashlib.new('sha1') 

199 if not file_list: 

200 return '', checksum.hexdigest() 

201 

202 root = None 

203 for fname in file_list: 

204 with open(fname, 'rb') as fp: 

205 contents = fp.read() 

206 checksum.update(contents) 

207 fp.seek(0) 

208 xml = ElementTree.parse(fp).getroot() 

209 

210 if root is None: 

211 root = ElementTree.Element(xml.tag) 

212 #elif root.tag != xml.tag: 

213 # raise ValueError("Root tags missmatch: %r != %r" % (root.tag, xml.tag)) 

214 

215 for child in xml.getchildren(): 

216 root.append(child) 

217 return ElementTree.tostring(root, 'utf-8'), checksum.hexdigest()

312def generate_views(action): 

313 """ 

314 While the server generates a sequence called "views" computing dependencies 

315 between a bunch of stuff for views coming directly from the database 

316 (the ``ir.actions.act_window model``), it's also possible for e.g. buttons 

317 to return custom view dictionaries generated on the fly. 

318 

319 In that case, there is no ``views`` key available on the action. 

320 

321 Since the web client relies on ``action['views']``, generate it here from 

322 ``view_mode`` and ``view_id``. 

323 

324 Currently handles two different cases: 

325 

326 * no view_id, multiple view_mode 

327 * single view_id, single view_mode 

328 

329 :param dict action: action descriptor dictionary to generate a views key for 

330 """ 

331 view_id = action.get('view_id') or False 

332 if isinstance(view_id, (list, tuple)): 

333 view_id = view_id[0] 

334 

335 # providing at least one view mode is a requirement, not an option 

336 view_modes = action['view_mode'].split(',') 

337 

338 if len(view_modes) > 1: 

339 if view_id: 

340 raise ValueError('Non-db action dictionaries should provide ' 

341 'either multiple view modes or a single view ' 

342 'mode and an optional view id.\n\n Got view ' 

343 'modes %r and view id %r for action %r' % ( 

344 view_modes, view_id, action)) 

345 action['views'] = [(False, mode) for mode in view_modes] 

346 return 

347 action['views'] = [(view_id, view_modes[0])]

349def fix_view_modes(action): 

350 """ For historical reasons, Odoo has weird dealings in relation to 

351 view_mode and the view_type attribute (on window actions): 

352 

353 * one of the view modes is ``tree``, which stands for both list views 

354 and tree views 

355 * the choice is made by checking ``view_type``, which is either 

356 ``form`` for a list view or ``tree`` for an actual tree view 

357 

358 This methods simply folds the view_type into view_mode by adding a 

359 new view mode ``list`` which is the result of the ``tree`` view_mode 

360 in conjunction with the ``form`` view_type. 

361 

362 TODO: this should go into the doc, some kind of "peculiarities" section 

363 

364 :param dict action: an action descriptor 

365 :returns: nothing, the action is modified in place 

366 """ 

367 if not action.get('views'): 

368 generate_views(action) 

369 

370 if action.pop('view_type', 'form') != 'form': 

371 return action 

372 

373 if 'view_mode' in action: 

374 action['view_mode'] = ','.join( 

375 mode if mode != 'tree' else 'list' 

376 for mode in action['view_mode'].split(',')) 

377 action['views'] = [ 

378 [id, mode if mode != 'tree' else 'list'] 

379 for id, mode in action['views'] 

380 ] 

381 

382 return action

1269 def fields_info(self, model, export_fields): 

1270 info = {} 

1271 fields = self.fields_get(model) 

1272 if ".id" in export_fields: 

1273 fields['.id'] = fields.get('id', {'string': 'ID'}) 

1274 

1275 # To make fields retrieval more efficient, fetch all sub-fields of a 

1276 # given field at the same time. Because the order in the export list is 

1277 # arbitrary, this requires ordering all sub-fields of a given field 

1278 # together so they can be fetched at the same time 

1279 # 

1280 # Works the following way: 

1281 # * sort the list of fields to export, the default sorting order will 

1282 # put the field itself (if present, for xmlid) and all of its 

1283 # sub-fields right after it 

1284 # * then, group on: the first field of the path (which is the same for 

1285 # a field and for its subfields and the length of splitting on the 

1286 # first '/', which basically means grouping the field on one side and 

1287 # all of the subfields on the other. This way, we have the field (for 

1288 # the xmlid) with length 1, and all of the subfields with the same 

1289 # base but a length "flag" of 2 

1290 # * if we have a normal field (length 1), just add it to the info 

1291 # mapping (with its string) as-is 

1292 # * otherwise, recursively call fields_info via graft_subfields. 

1293 # all graft_subfields does is take the result of fields_info (on the 

1294 # field's model) and prepend the current base (current field), which 

1295 # rebuilds the whole sub-tree for the field 

1296 # 

1297 # result: because we're not fetching the fields_get for half the 

1298 # database models, fetching a namelist with a dozen fields (including 

1299 # relational data) falls from ~6s to ~300ms (on the leads model). 

1300 # export lists with no sub-fields (e.g. import_compatible lists with 

1301 # no o2m) are even more efficient (from the same 6s to ~170ms, as 

1302 # there's a single fields_get to execute) 

1303 for (base, length), subfields in itertools.groupby( 

1304 sorted(export_fields), 

1305 lambda field: (field.split('/', 1)[0], len(field.split('/', 1)))): 

1306 subfields = list(subfields) 

1307 if length == 2: 

1308 # subfields is a seq of $base/*rest, and not loaded yet 

1309 info.update(self.graft_subfields( 

1310 fields[base]['relation'], base, fields[base]['string'], 

1311 subfields 

1312 )) 

1313 elif base in fields: 

1314 info[base] = fields[base]['string'] 

1315 

1316 return info
