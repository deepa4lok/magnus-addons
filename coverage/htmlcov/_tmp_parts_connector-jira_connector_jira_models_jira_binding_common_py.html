11class JiraBinding(models.AbstractModel): 

12 """ Abstract Model for the Bindings. 

13 

14 All the models used as bindings between Jira and Odoo 

15 (``jira.product.product``, ...) should ``_inherit`` it. 

16 """ 

17 _name = 'jira.binding' 

18 _inherit = 'external.binding' 

19 _description = 'Jira Binding (abstract)' 

20 

21 # odoo-side id must be declared in concrete model 

22 # odoo_id = fields.Many2one(...) 

23 backend_id = fields.Many2one( 

24 comodel_name='jira.backend', 

25 string='Jira Backend', 

26 required=True, 

27 ondelete='restrict', 

28 ) 

29 external_id = fields.Char(string='ID on Jira', index=True) 

30 

31 _sql_constraints = [ 

32 ('jira_binding_uniq', 'unique(backend_id, external_id)', 

33 "A binding already exists for this Jira record"), 

34 ]

36 @job(default_channel='root.connector_jira.import') 

37 @api.model 

38 def import_batch(self, backend, from_date=None, to_date=None): 

39 """ Prepare import of a batch of records """ 

40 with backend.get_environment(self._name) as connector_env: 

41 importer = connector_env.get_connector_unit(BatchImporter) 

42 importer.run(from_date=from_date, to_date=to_date) 

43 

44 @job(default_channel='root.connector_jira.import') 

45 @api.model 

46 def import_record(self, backend, external_id, force=False): 

47 """ Import a record """ 

48 with backend.get_environment(self._name) as connector_env: 

49 importer = connector_env.get_connector_unit(JiraImporter) 

50 importer.run(external_id, force=force) 

51 

52 @job(default_channel='root.connector_jira.import') 

53 @api.model 

54 def delete_record(self, backend, external_id): 

55 """ Delete a record on Odoo """ 

56 with backend.get_environment(self._name) as connector_env: 

57 importer = connector_env.get_connector_unit(JiraDeleter) 

58 importer.run(external_id) 

59 

60 @job(default_channel='root.connector_jira.export') 

61 @related_action(action='related_action_unwrap_binding')

67 exporter.run(self, fields=fields)
