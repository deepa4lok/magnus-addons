7class ProcurementOrder(models.Model):

10 task_id = fields.Many2one('project.task', 'Task', copy=False)

12 def _is_procurement_task(self): 

13 return self.product_id.type == 'service' and self.product_id.track_service == 'task'

15 @api.multi 

16 def _assign(self): 

17 self.ensure_one() 

18 res = super(ProcurementOrder, self)._assign() 

19 if not res: 

20 # if there isn't any specific procurement.rule defined for the product, we may want to create a task 

21 return self._is_procurement_task() 

22 return res 

23 

24 @api.multi 

25 def _run(self): 

26 self.ensure_one() 

27 if self._is_procurement_task() and not self.task_id: 

28 # If the SO was confirmed, cancelled, set to draft then confirmed, avoid creating a new 

29 # task. 

30 if self.sale_line_id: 

31 existing_task = self.env['project.task'].search( 

32 [('sale_line_id', '=', self.sale_line_id.id)] 

33 ) 

34 if existing_task: 

35 return existing_task 

36 # create a task for the procurement 

37 return self._create_service_task() 

38 return super(ProcurementOrder, self)._run()

40 def _convert_qty_company_hours(self): 

41 company_time_uom_id = self.env.user.company_id.project_time_mode_id 

42 if self.product_uom.id != company_time_uom_id.id and self.product_uom.category_id.id == company_time_uom_id.category_id.id: 

43 planned_hours = self.product_uom._compute_quantity(self.product_qty, company_time_uom_id) 

44 else: 

45 planned_hours = self.product_qty 

46 return planned_hours

48 def _get_project(self): 

49 Project = self.env['project.project'] 

50 project = self.product_id.with_context(force_company=self.company_id.id).project_id 

51 if not project and self.sale_line_id: 

52 # find the project corresponding to the analytic account of the sales order 

53 account = self.sale_line_id.order_id.project_id 

54 if not account: 

55 self.sale_line_id.order_id._create_analytic_account() 

56 account = self.sale_line_id.order_id.project_id 

57 project = Project.search([('analytic_account_id', '=', account.id)], limit=1) 

58 if not project: 

59 project_id = account.project_create({'name': account.name, 'use_tasks': True}) 

60 project = Project.browse(project_id) 

61 return project

63 def _create_service_task(self): 

64 project = self._get_project() 

65 planned_hours = self._convert_qty_company_hours() 

66 task = self.env['project.task'].create({ 

67 'name': '%s:%s' % (self.origin or '', self.product_id.name), 

68 'date_deadline': self.date_planned, 

69 'planned_hours': planned_hours, 

70 'remaining_hours': planned_hours, 

71 'partner_id': self.sale_line_id.order_id.partner_id.id or self.partner_dest_id.id, 

72 'user_id': self.env.uid, 

73 'procurement_id': self.id, 

74 'description': self.name + '<br/>', 

75 'project_id': project.id, 

76 'company_id': self.company_id.id, 

77 }) 

78 self.write({'task_id': task.id}) 

79 

80 msg_body = _("Task Created (%s): <a href=# data-oe-model=project.task data-oe-id=%d>%s</a>") % (self.product_id.name, task.id, task.name) 

81 self.message_post(body=msg_body) 

82 if self.sale_line_id.order_id:

85 task.message_post(body=task_msg)
