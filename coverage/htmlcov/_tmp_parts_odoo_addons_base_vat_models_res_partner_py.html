164 def check_vat_ch(self, vat): 

165 ''' 

166 Check Switzerland VAT number. 

167 ''' 

168 # VAT number in Switzerland will change between 2011 and 2013 

169 # http://www.estv.admin.ch/mwst/themen/00154/00589/01107/index.html?lang=fr 

170 # Old format is "TVA 123456" we will admit the user has to enter ch before the number 

171 # Format will becomes such as "CHE-999.999.99C TVA" 

172 # Both old and new format will be accepted till end of 2013 

173 # Accepted format are: (spaces are ignored) 

174 # CH TVA ###### 

175 # CH IVA ###### 

176 # CH MWST ####### 

177 # 

178 # CHE#########MWST 

179 # CHE#########TVA 

180 # CHE#########IVA 

181 # CHE-###.###.### MWST 

182 # CHE-###.###.### TVA 

183 # CHE-###.###.### IVA 

184 # 

185 if self.__check_vat_ch_re1.match(vat): 

186 return True 

187 match = self.__check_vat_ch_re2.match(vat) 

188 if match: 

189 # For new TVA numbers, do a mod11 check 

190 num = filter(lambda s: s.isdigit(), match.group(1)) # get the digits only 

191 factor = (5, 4, 3, 2, 7, 6, 5, 4) 

192 csum = sum([int(num[i]) * factor[i] for i in range(8)]) 

193 check = (11 - (csum % 11)) % 11 

194 return check == int(num[8]) 

195 return False

197 def _ie_check_char(self, vat): 

198 vat = vat.zfill(8) 

199 extra = 0 

200 if vat[7] not in ' W': 

201 if vat[7].isalpha(): 

202 extra = 9 * (ord(vat[7]) - 64) 

203 else: 

204 # invalid 

205 return -1 

206 checksum = extra + sum((8-i) * int(x) for i, x in enumerate(vat[:7])) 

207 return 'WABCDEFGHIJKLMNOPQRSTUV'[checksum % 23]

209 def check_vat_ie(self, vat): 

210 """ Temporary Ireland VAT validation to support the new format 

211 introduced in January 2013 in Ireland, until upstream is fixed. 

212 TODO: remove when fixed upstream""" 

213 if len(vat) not in (8, 9) or not vat[2:7].isdigit():

234 def check_vat_mx(self, vat): 

235 ''' Mexican VAT verification 

236 

237 Verificar RFC México 

238 ''' 

239 # we convert to 8-bit encoding, to help the regex parse only bytes 

240 vat = ustr(vat).encode('iso8859-1') 

241 m = self.__check_vat_mx_re.match(vat) 

242 if not m: 

243 #No valid format 

244 return False 

245 try: 

246 ano = int(m.group('ano')) 

247 if ano > 30: 

248 ano = 1900 + ano 

249 else: 

250 ano = 2000 + ano 

251 datetime.date(ano, int(m.group('mes')), int(m.group('dia'))) 

252 except ValueError: 

253 return False 

254 

255 # Valid format and valid date 

256 return True

260 def check_vat_no(self, vat): 

261 """ 

262 Check Norway VAT number.See http://www.brreg.no/english/coordination/number.html 

263 """ 

264 if len(vat) == 12 and vat.upper().endswith('MVA'): 

265 vat = vat[:-3] # Strictly speaking we should enforce the suffix MVA but... 

266 

267 if len(vat) != 9: 

268 return False 

269 try: 

270 int(vat) 

271 except ValueError: 

272 return False 

273 

274 sum = (3 * int(vat[0])) + (2 * int(vat[1])) + \ 

275 (7 * int(vat[2])) + (6 * int(vat[3])) + \ 

276 (5 * int(vat[4])) + (4 * int(vat[5])) + \ 

277 (3 * int(vat[6])) + (2 * int(vat[7])) 

278 

279 check = 11 - (sum % 11) 

280 if check == 11: 

281 check = 0 

282 if check == 10: 

283 # 10 is not a valid check digit for an organization number 

284 return False 

285 return check == int(vat[8])

288 def check_vat_pe(self, vat): 

289 

290 vat_type, vat = vat and len(vat) >= 2 and (vat[0], vat[1:]) or (False, False) 

291 

292 if vat_type and vat_type.upper() == 'D': 

293 # DNI 

294 return True 

295 elif vat_type and vat_type.upper() == 'R': 

296 # verify RUC 

297 factor = '5432765432' 

298 sum = 0 

299 dig_check = False 

300 if len(vat) != 11: 

301 return False 

302 try: 

303 int(vat) 

304 except ValueError: 

305 return False 

306 

307 for f in range(0, 10): 

308 sum += int(factor[f]) * int(vat[f]) 

309 

310 subtraction = 11 - (sum % 11) 

311 if subtraction == 10: 

312 dig_check = 0 

313 elif subtraction == 11: 

314 dig_check = 1 

315 else: 

316 dig_check = subtraction 

317 

318 return int(vat[10]) == dig_check

323 def check_vat_tr(self, vat): 

324 

325 if not (10 <= len(vat) <= 11): 

326 return False 

327 try: 

328 int(vat) 

329 except ValueError: 

330 return False 

331 

332 # check vat number (vergi no) 

333 if len(vat) == 10: 

334 sum = 0 

335 check = 0 

336 for f in range(0, 9): 

337 c1 = (int(vat[f]) + (9-f)) % 10 

338 c2 = (c1 * (2 ** (9-f))) % 9 

339 if (c1 != 0) and (c2 == 0): 

340 c2 = 9 

341 sum += c2 

342 if sum % 10 == 0: 

343 check = 0 

344 else: 

345 check = 10 - (sum % 10) 

346 return int(vat[9]) == check 

347 

348 # check personal id (tc kimlik no) 

349 if len(vat) == 11: 

350 c1a = 0 

351 c1b = 0 

352 c2 = 0 

353 for f in range(0, 9, 2): 

354 c1a += int(vat[f]) 

355 for f in range(1, 9, 2): 

356 c1b += int(vat[f]) 

357 c1 = ((7 * c1a) - c1b) % 10 

358 for f in range(0, 10): 

359 c2 += int(vat[f]) 

360 c2 = c2 % 10 

361 return int(vat[9]) == c1 and int(vat[10]) == c2
