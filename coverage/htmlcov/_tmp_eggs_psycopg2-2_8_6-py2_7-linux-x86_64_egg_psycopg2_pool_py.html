28import psycopg2 

29from psycopg2 import extensions as _ext

32class PoolError(psycopg2.Error): 

33 pass

36class AbstractConnectionPool(object): 

37 """Generic key-based pooling code.""" 

38 

39 def __init__(self, minconn, maxconn, *args, **kwargs): 

40 """Initialize the connection pool. 

41 

42 New 'minconn' connections are created immediately calling 'connfunc' 

43 with given parameters. The connection pool will support a maximum of 

44 about 'maxconn' connections. 

45 """ 

46 self.minconn = int(minconn) 

47 self.maxconn = int(maxconn) 

48 self.closed = False 

49 

50 self._args = args 

51 self._kwargs = kwargs 

52 

53 self._pool = [] 

54 self._used = {} 

55 self._rused = {} # id(conn) -> key map 

56 self._keys = 0 

57 

58 for i in range(self.minconn): 

59 self._connect() 

60 

61 def _connect(self, key=None): 

62 """Create a new connection and assign it to 'key' if not None.""" 

63 conn = psycopg2.connect(*self._args, **self._kwargs) 

64 if key is not None: 

65 self._used[key] = conn 

66 self._rused[id(conn)] = key 

67 else: 

68 self._pool.append(conn) 

69 return conn 

70 

71 def _getkey(self): 

72 """Return a new unique key.""" 

73 self._keys += 1 

74 return self._keys 

75 

76 def _getconn(self, key=None): 

77 """Get a free connection and assign it to 'key' if not None.""" 

78 if self.closed: 

79 raise PoolError("connection pool is closed") 

80 if key is None: 

81 key = self._getkey() 

82 

83 if key in self._used: 

84 return self._used[key] 

85 

86 if self._pool: 

87 self._used[key] = conn = self._pool.pop() 

88 self._rused[id(conn)] = key 

89 return conn 

90 else: 

91 if len(self._used) == self.maxconn: 

92 raise PoolError("connection pool exhausted") 

93 return self._connect(key) 

94 

95 def _putconn(self, conn, key=None, close=False): 

96 """Put away a connection.""" 

97 if self.closed: 

98 raise PoolError("connection pool is closed") 

99 

100 if key is None: 

101 key = self._rused.get(id(conn)) 

102 if key is None: 

103 raise PoolError("trying to put unkeyed connection") 

104 

105 if len(self._pool) < self.minconn and not close: 

106 # Return the connection into a consistent state before putting 

107 # it back into the pool 

108 if not conn.closed: 

109 status = conn.info.transaction_status 

110 if status == _ext.TRANSACTION_STATUS_UNKNOWN: 

111 # server connection lost 

112 conn.close() 

113 elif status != _ext.TRANSACTION_STATUS_IDLE: 

114 # connection in error or in transaction 

115 conn.rollback() 

116 self._pool.append(conn) 

117 else: 

118 # regular idle connection 

119 self._pool.append(conn) 

120 # If the connection is closed, we just discard it. 

121 else: 

122 conn.close() 

123 

124 # here we check for the presence of key because it can happen that a 

125 # thread tries to put back a connection after a call to close 

126 if not self.closed or key in self._used: 

127 del self._used[key] 

128 del self._rused[id(conn)] 

129 

130 def _closeall(self): 

131 """Close all connections. 

132 

133 Note that this can lead to some code fail badly when trying to use 

134 an already closed connection. If you call .closeall() make sure 

135 your code can deal with it. 

136 """ 

137 if self.closed: 

138 raise PoolError("connection pool is closed") 

139 for conn in self._pool + list(self._used.values()): 

140 try: 

141 conn.close() 

142 except Exception: 

143 pass 

144 self.closed = True

147class SimpleConnectionPool(AbstractConnectionPool):

150 getconn = AbstractConnectionPool._getconn

155class ThreadedConnectionPool(AbstractConnectionPool): 

156 """A connection pool that works with the threading module.""" 

157 

158 def __init__(self, minconn, maxconn, *args, **kwargs): 

159 """Initialize the threading lock.""" 

160 import threading 

161 AbstractConnectionPool.__init__( 

162 self, minconn, maxconn, *args, **kwargs) 

163 self._lock = threading.Lock() 

164 

165 def getconn(self, key=None): 

166 """Get a free connection and assign it to 'key' if not None.""" 

167 self._lock.acquire() 

168 try: 

169 return self._getconn(key) 

170 finally: 

171 self._lock.release() 

172 

173 def putconn(self, conn=None, key=None, close=False): 

174 """Put away an unused connection.""" 

175 self._lock.acquire() 

176 try: 

177 self._putconn(conn, key, close) 

178 finally: 

179 self._lock.release() 

180 

181 def closeall(self): 

182 """Close all connections (even the one currently in use.)""" 

183 self._lock.acquire() 

184 try: 

185 self._closeall() 

186 finally: 

187 self._lock.release()
