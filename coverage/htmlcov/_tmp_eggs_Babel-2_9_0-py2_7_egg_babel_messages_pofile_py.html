22def unescape(string): 

23 r"""Reverse `escape` the given string. 

24 

25 >>> print(unescape('"Say:\\n \\"hello, world!\\"\\n"')) 

26 Say: 

27 "hello, world!" 

28 <BLANKLINE> 

29 

30 :param string: the string to unescape 

31 """ 

32 def replace_escapes(match): 

33 m = match.group(1) 

34 if m == 'n': 

35 return '\n' 

36 elif m == 't': 

37 return '\t' 

38 elif m == 'r': 

39 return '\r' 

40 # m is \ or " 

41 return m 

42 return re.compile(r'\\([\\trn"])').sub(replace_escapes, string[1:-1])

45def denormalize(string): 

46 r"""Reverse the normalization done by the `normalize` function. 

47 

48 >>> print(denormalize(r'''"" 

49 ... "Say:\n" 

50 ... " \"hello, world!\"\n"''')) 

51 Say: 

52 "hello, world!" 

53 <BLANKLINE> 

54 

55 >>> print(denormalize(r'''"" 

56 ... "Say:\n" 

57 ... " \"Lorem ipsum dolor sit " 

58 ... "amet, consectetur adipisicing" 

59 ... " elit, \"\n"''')) 

60 Say: 

61 "Lorem ipsum dolor sit amet, consectetur adipisicing elit, " 

62 <BLANKLINE> 

63 

64 :param string: the string to denormalize 

65 """ 

66 if '\n' in string: 

67 escaped_lines = string.splitlines() 

68 if string.startswith('""'): 

69 escaped_lines = escaped_lines[1:] 

70 lines = map(unescape, escaped_lines) 

71 return ''.join(lines) 

72 else: 

73 return unescape(string)

132class PoFileParser(object): 

133 """Support class to read messages from a ``gettext`` PO (portable object) file 

134 and add them to a `Catalog` 

135 

136 See `read_po` for simple cases. 

137 """ 

138 

139 _keywords = [ 

140 'msgid', 

141 'msgstr', 

142 'msgctxt', 

143 'msgid_plural', 

144 ] 

145 

146 def __init__(self, catalog, ignore_obsolete=False, abort_invalid=False): 

147 self.catalog = catalog 

148 self.ignore_obsolete = ignore_obsolete 

149 self.counter = 0 

150 self.offset = 0 

151 self.abort_invalid = abort_invalid 

152 self._reset_message_state() 

153 

154 def _reset_message_state(self): 

155 self.messages = [] 

156 self.translations = [] 

157 self.locations = [] 

158 self.flags = [] 

159 self.user_comments = [] 

160 self.auto_comments = [] 

161 self.context = None 

162 self.obsolete = False 

163 self.in_msgid = False 

164 self.in_msgstr = False 

165 self.in_msgctxt = False 

166 

167 def _add_message(self): 

168 """ 

169 Add a message to the catalog based on the current parser state and 

170 clear the state ready to process the next message. 

171 """ 

172 self.translations.sort() 

173 if len(self.messages) > 1: 

174 msgid = tuple([m.denormalize() for m in self.messages]) 

175 else: 

176 msgid = self.messages[0].denormalize() 

177 if isinstance(msgid, (list, tuple)): 

178 string = ['' for _ in range(self.catalog.num_plurals)] 

179 for idx, translation in self.translations: 

180 if idx >= self.catalog.num_plurals: 

181 self._invalid_pofile(u"", self.offset, "msg has more translations than num_plurals of catalog") 

182 continue 

183 string[idx] = translation.denormalize() 

184 string = tuple(string) 

185 else: 

186 string = self.translations[0][1].denormalize() 

187 if self.context: 

188 msgctxt = self.context.denormalize() 

189 else: 

190 msgctxt = None 

191 message = Message(msgid, string, list(self.locations), set(self.flags), 

192 self.auto_comments, self.user_comments, lineno=self.offset + 1, 

193 context=msgctxt) 

194 if self.obsolete: 

195 if not self.ignore_obsolete: 

196 self.catalog.obsolete[msgid] = message 

197 else: 

198 self.catalog[msgid] = message 

199 self.counter += 1 

200 self._reset_message_state() 

201 

202 def _finish_current_message(self): 

203 if self.messages: 

204 self._add_message() 

205 

206 def _process_message_line(self, lineno, line, obsolete=False): 

207 if line.startswith('"'): 

208 self._process_string_continuation_line(line, lineno) 

209 else: 

210 self._process_keyword_line(lineno, line, obsolete) 

211 

212 def _process_keyword_line(self, lineno, line, obsolete=False): 

213 

214 for keyword in self._keywords: 

215 try: 

216 if line.startswith(keyword) and line[len(keyword)] in [' ', '[']: 

217 arg = line[len(keyword):] 

218 break 

219 except IndexError: 

220 self._invalid_pofile(line, lineno, "Keyword must be followed by a string") 

221 else: 

222 self._invalid_pofile(line, lineno, "Start of line didn't match any expected keyword.") 

223 return 

224 

225 if keyword in ['msgid', 'msgctxt']: 

226 self._finish_current_message() 

227 

228 self.obsolete = obsolete 

229 

230 # The line that has the msgid is stored as the offset of the msg 

231 # should this be the msgctxt if it has one? 

232 if keyword == 'msgid': 

233 self.offset = lineno 

234 

235 if keyword in ['msgid', 'msgid_plural']: 

236 self.in_msgctxt = False 

237 self.in_msgid = True 

238 self.messages.append(_NormalizedString(arg)) 

239 

240 elif keyword == 'msgstr': 

241 self.in_msgid = False 

242 self.in_msgstr = True 

243 if arg.startswith('['): 

244 idx, msg = arg[1:].split(']', 1) 

245 self.translations.append([int(idx), _NormalizedString(msg)]) 

246 else: 

247 self.translations.append([0, _NormalizedString(arg)]) 

248 

249 elif keyword == 'msgctxt': 

250 self.in_msgctxt = True 

251 self.context = _NormalizedString(arg) 

252 

253 def _process_string_continuation_line(self, line, lineno): 

254 if self.in_msgid: 

255 s = self.messages[-1] 

256 elif self.in_msgstr: 

257 s = self.translations[-1][1] 

258 elif self.in_msgctxt: 

259 s = self.context 

260 else: 

261 self._invalid_pofile(line, lineno, "Got line starting with \" but not in msgid, msgstr or msgctxt") 

262 return 

263 s.append(line) 

264 

265 def _process_comment(self, line): 

266 

267 self._finish_current_message() 

268 

269 if line[1:].startswith(':'): 

270 for location in line[2:].lstrip().split(): 

271 pos = location.rfind(':') 

272 if pos >= 0: 

273 try: 

274 lineno = int(location[pos + 1:]) 

275 except ValueError: 

276 continue 

277 self.locations.append((location[:pos], lineno)) 

278 else: 

279 self.locations.append((location, None)) 

280 elif line[1:].startswith(','): 

281 for flag in line[2:].lstrip().split(','): 

282 self.flags.append(flag.strip()) 

283 elif line[1:].startswith('.'): 

284 # These are called auto-comments 

285 comment = line[2:].strip() 

286 if comment: # Just check that we're not adding empty comments 

287 self.auto_comments.append(comment) 

288 else: 

289 # These are called user comments 

290 self.user_comments.append(line[1:].strip())

292 def parse(self, fileobj): 

293 """ 

294 Reads from the file-like object `fileobj` and adds any po file 

295 units found in it to the `Catalog` supplied to the constructor. 

296 """ 

297 

298 for lineno, line in enumerate(fileobj): 

299 line = line.strip() 

300 if not isinstance(line, text_type): 

301 line = line.decode(self.catalog.charset) 

302 if not line: 

303 continue 

304 if line.startswith('#'): 

305 if line[1:].startswith('~'): 

306 self._process_message_line(lineno, line[2:].lstrip(), obsolete=True) 

307 else: 

308 self._process_comment(line) 

309 else: 

310 self._process_message_line(lineno, line) 

311 

312 self._finish_current_message() 

313 

314 # No actual messages found, but there was some info in comments, from which 

315 # we'll construct an empty header message 

316 if not self.counter and (self.flags or self.user_comments or self.auto_comments): 

317 self.messages.append(_NormalizedString(u'""')) 

318 self.translations.append([0, _NormalizedString(u'""')]) 

319 self._add_message() 

320 

321 def _invalid_pofile(self, line, lineno, msg): 

322 assert isinstance(line, text_type) 

323 if self.abort_invalid: 

324 raise PoFileError(msg, self.catalog, line, lineno) 

325 print("WARNING:", msg) 

326 # `line` is guaranteed to be unicode so u"{}"-interpolating would always 

327 # succeed, but on Python < 2 if not in a TTY, `sys.stdout.encoding` 

328 # is `None`, unicode may not be printable so we `repr()` to ASCII. 

329 print(u"WARNING: Problem on line {0}: {1}".format(lineno + 1, repr(line)))

332def read_po(fileobj, locale=None, domain=None, ignore_obsolete=False, charset=None, abort_invalid=False): 

333 """Read messages from a ``gettext`` PO (portable object) file from the given 

334 file-like object and return a `Catalog`. 

335 

336 >>> from datetime import datetime 

337 >>> from babel._compat import StringIO 

338 >>> buf = StringIO(''' 

339 ... #: main.py:1 

340 ... #, fuzzy, python-format 

341 ... msgid "foo %(name)s" 

342 ... msgstr "quux %(name)s" 

343 ... 

344 ... # A user comment 

345 ... #. An auto comment 

346 ... #: main.py:3 

347 ... msgid "bar" 

348 ... msgid_plural "baz" 

349 ... msgstr[0] "bar" 

350 ... msgstr[1] "baaz" 

351 ... ''') 

352 >>> catalog = read_po(buf) 

353 >>> catalog.revision_date = datetime(2007, 4, 1) 

354 

355 >>> for message in catalog: 

356 ... if message.id: 

357 ... print((message.id, message.string)) 

358 ... print(' ', (message.locations, sorted(list(message.flags)))) 

359 ... print(' ', (message.user_comments, message.auto_comments)) 

360 (u'foo %(name)s', u'quux %(name)s') 

361 ([(u'main.py', 1)], [u'fuzzy', u'python-format']) 

362 ([], []) 

363 ((u'bar', u'baz'), (u'bar', u'baaz')) 

364 ([(u'main.py', 3)], []) 

365 ([u'A user comment'], [u'An auto comment']) 

366 

367 .. versionadded:: 1.0 

368 Added support for explicit charset argument. 

369 

370 :param fileobj: the file-like object to read the PO file from 

371 :param locale: the locale identifier or `Locale` object, or `None` 

372 if the catalog is not bound to a locale (which basically 

373 means it's a template) 

374 :param domain: the message domain 

375 :param ignore_obsolete: whether to ignore obsolete messages in the input 

376 :param charset: the character set of the catalog. 

377 :param abort_invalid: abort read if po file is invalid 

378 """ 

379 catalog = Catalog(locale=locale, domain=domain, charset=charset) 

380 parser = PoFileParser(catalog, ignore_obsolete, abort_invalid=abort_invalid) 

381 parser.parse(fileobj) 

382 return catalog

392def escape(string): 

393 r"""Escape the given string so that it can be included in double-quoted 

394 strings in ``PO`` files. 

395 

396 >>> escape('''Say: 

397 ... "hello, world!" 

398 ... ''') 

399 '"Say:\\n \\"hello, world!\\"\\n"' 

400 

401 :param string: the string to escape 

402 """ 

403 return '"%s"' % string.replace('\\', '\\\\') \ 

404 .replace('\t', '\\t') \ 

405 .replace('\r', '\\r') \ 

406 .replace('\n', '\\n') \ 

407 .replace('\"', '\\"')

410def normalize(string, prefix='', width=76): 

411 r"""Convert a string into a format that is appropriate for .po files. 

412 

413 >>> print(normalize('''Say: 

414 ... "hello, world!" 

415 ... ''', width=None)) 

416 "" 

417 "Say:\n" 

418 " \"hello, world!\"\n" 

419 

420 >>> print(normalize('''Say: 

421 ... "Lorem ipsum dolor sit amet, consectetur adipisicing elit, " 

422 ... ''', width=32)) 

423 "" 

424 "Say:\n" 

425 " \"Lorem ipsum dolor sit " 

426 "amet, consectetur adipisicing" 

427 " elit, \"\n" 

428 

429 :param string: the string to normalize 

430 :param prefix: a string that should be prepended to every line 

431 :param width: the maximum line width; use `None`, 0, or a negative number 

432 to completely disable line wrapping 

433 """ 

434 if width and width > 0: 

435 prefixlen = len(prefix) 

436 lines = [] 

437 for line in string.splitlines(True): 

438 if len(escape(line)) + prefixlen > width: 

439 chunks = WORD_SEP.split(line) 

440 chunks.reverse() 

441 while chunks: 

442 buf = [] 

443 size = 2 

444 while chunks: 

445 l = len(escape(chunks[-1])) - 2 + prefixlen 

446 if size + l < width: 

447 buf.append(chunks.pop()) 

448 size += l 

449 else: 

450 if not buf: 

451 # handle long chunks by putting them on a 

452 # separate line 

453 buf.append(chunks.pop()) 

454 break 

455 lines.append(u''.join(buf)) 

456 else: 

457 lines.append(line) 

458 else: 

459 lines = string.splitlines(True) 

460 

461 if len(lines) <= 1: 

462 return escape(string) 

463 

464 # Remove empty trailing line 

465 if lines and not lines[-1]: 

466 del lines[-1] 

467 lines[-1] += '\n' 

468 return u'""\n' + u'\n'.join([(prefix + escape(line)) for line in lines])

471def write_po(fileobj, catalog, width=76, no_location=False, omit_header=False, 

472 sort_output=False, sort_by_file=False, ignore_obsolete=False, 

473 include_previous=False, include_lineno=True): 

474 r"""Write a ``gettext`` PO (portable object) template file for a given 

475 message catalog to the provided file-like object. 

476 

477 >>> catalog = Catalog() 

478 >>> catalog.add(u'foo %(name)s', locations=[('main.py', 1)], 

479 ... flags=('fuzzy',)) 

480 <Message...> 

481 >>> catalog.add((u'bar', u'baz'), locations=[('main.py', 3)]) 

482 <Message...> 

483 >>> from babel._compat import BytesIO 

484 >>> buf = BytesIO() 

485 >>> write_po(buf, catalog, omit_header=True) 

486 >>> print(buf.getvalue().decode("utf8")) 

487 #: main.py:1 

488 #, fuzzy, python-format 

489 msgid "foo %(name)s" 

490 msgstr "" 

491 <BLANKLINE> 

492 #: main.py:3 

493 msgid "bar" 

494 msgid_plural "baz" 

495 msgstr[0] "" 

496 msgstr[1] "" 

497 <BLANKLINE> 

498 <BLANKLINE> 

499 

500 :param fileobj: the file-like object to write to 

501 :param catalog: the `Catalog` instance 

502 :param width: the maximum line width for the generated output; use `None`, 

503 0, or a negative number to completely disable line wrapping 

504 :param no_location: do not emit a location comment for every message 

505 :param omit_header: do not include the ``msgid ""`` entry at the top of the 

506 output 

507 :param sort_output: whether to sort the messages in the output by msgid 

508 :param sort_by_file: whether to sort the messages in the output by their 

509 locations 

510 :param ignore_obsolete: whether to ignore obsolete messages and not include 

511 them in the output; by default they are included as 

512 comments 

513 :param include_previous: include the old msgid as a comment when 

514 updating the catalog 

515 :param include_lineno: include line number in the location comment 

516 """ 

517 def _normalize(key, prefix=''): 

518 return normalize(key, prefix=prefix, width=width) 

519 

520 def _write(text): 

521 if isinstance(text, text_type): 

522 text = text.encode(catalog.charset, 'backslashreplace') 

523 fileobj.write(text) 

524 

525 def _write_comment(comment, prefix=''): 

526 # xgettext always wraps comments even if --no-wrap is passed; 

527 # provide the same behaviour 

528 if width and width > 0: 

529 _width = width 

530 else: 

531 _width = 76 

532 for line in wraptext(comment, _width): 

533 _write('#%s %s\n' % (prefix, line.strip())) 

534 

535 def _write_message(message, prefix=''): 

536 if isinstance(message.id, (list, tuple)): 

537 if message.context: 

538 _write('%smsgctxt %s\n' % (prefix, 

539 _normalize(message.context, prefix))) 

540 _write('%smsgid %s\n' % (prefix, _normalize(message.id[0], prefix))) 

541 _write('%smsgid_plural %s\n' % ( 

542 prefix, _normalize(message.id[1], prefix) 

543 )) 

544 

545 for idx in range(catalog.num_plurals): 

546 try: 

547 string = message.string[idx] 

548 except IndexError: 

549 string = '' 

550 _write('%smsgstr[%d] %s\n' % ( 

551 prefix, idx, _normalize(string, prefix) 

552 )) 

553 else: 

554 if message.context: 

555 _write('%smsgctxt %s\n' % (prefix, 

556 _normalize(message.context, prefix))) 

557 _write('%smsgid %s\n' % (prefix, _normalize(message.id, prefix))) 

558 _write('%smsgstr %s\n' % ( 

559 prefix, _normalize(message.string or '', prefix) 

560 )) 

561 

562 sort_by = None 

563 if sort_output: 

564 sort_by = "message" 

565 elif sort_by_file: 

566 sort_by = "location" 

567 

568 for message in _sort_messages(catalog, sort_by=sort_by): 

569 if not message.id: # This is the header "message" 

570 if omit_header: 

571 continue 

572 comment_header = catalog.header_comment 

573 if width and width > 0: 

574 lines = [] 

575 for line in comment_header.splitlines(): 

576 lines += wraptext(line, width=width, 

577 subsequent_indent='# ') 

578 comment_header = u'\n'.join(lines) 

579 _write(comment_header + u'\n') 

580 

581 for comment in message.user_comments: 

582 _write_comment(comment) 

583 for comment in message.auto_comments: 

584 _write_comment(comment, prefix='.') 

585 

586 if not no_location: 

587 locs = [] 

588 

589 # sort locations by filename and lineno. 

590 # if there's no <int> as lineno, use `-1`. 

591 # if no sorting possible, leave unsorted. 

592 # (see issue #606) 

593 try: 

594 locations = sorted(message.locations, 

595 key=lambda x: (x[0], isinstance(x[1], int) and x[1] or -1)) 

596 except TypeError: # e.g. "TypeError: unorderable types: NoneType() < int()" 

597 locations = message.locations 

598 

599 for filename, lineno in locations: 

600 if lineno and include_lineno: 

601 locs.append(u'%s:%d' % (filename.replace(os.sep, '/'), lineno)) 

602 else: 

603 locs.append(u'%s' % filename.replace(os.sep, '/')) 

604 _write_comment(' '.join(locs), prefix=':') 

605 if message.flags: 

606 _write('#%s\n' % ', '.join([''] + sorted(message.flags))) 

607 

608 if message.previous_id and include_previous: 

609 _write_comment('msgid %s' % _normalize(message.previous_id[0]), 

610 prefix='|') 

611 if len(message.previous_id) > 1: 

612 _write_comment('msgid_plural %s' % _normalize( 

613 message.previous_id[1] 

614 ), prefix='|') 

615 

616 _write_message(message) 

617 _write('\n') 

618 

619 if not ignore_obsolete: 

620 for message in _sort_messages( 

621 catalog.obsolete.values(), 

622 sort_by=sort_by 

623 ): 

624 for comment in message.user_comments: 

625 _write_comment(comment) 

626 _write_message(message, prefix='#~ ') 

627 _write('\n')

630def _sort_messages(messages, sort_by): 

631 """ 

632 Sort the given message iterable by the given criteria. 

633 

634 Always returns a list. 

635 

636 :param messages: An iterable of Messages. 

637 :param sort_by: Sort by which criteria? Options are `message` and `location`. 

638 :return: list[Message] 

639 """ 

640 messages = list(messages) 

641 if sort_by == "message": 

642 messages.sort() 

643 elif sort_by == "location": 

644 messages.sort(key=lambda m: m.locations) 

645 return messages
