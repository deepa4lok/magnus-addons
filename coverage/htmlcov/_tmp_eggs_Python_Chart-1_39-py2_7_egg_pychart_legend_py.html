130 x = self.loc[0]

133 nr_rows = min(self.nr_rows, len(entries)) 

134 nr_cols = (len(entries)-1) / nr_rows + 1

136 ymin = y 

137 max_label_width = [0] * nr_cols 

138 max_sample_width = [0] * nr_cols 

139 heights = [0] * nr_rows 

140 

141 for i in range(len(entries)): 

142 l = entries[i] 

143 (col, row) = divmod(i, nr_rows) 

144 max_label_width[col] = max(l.label_width(), max_label_width[col]) 

145 max_sample_width[col] = max(l.sample_width(), max_sample_width[col]) 

146 heights[row] = max(l.height(), heights[row])

148 for h in heights: 

149 y += h 

150 y += self.inter_row_sep * (nr_rows - 1) 

151 ymax = y

153 tot_width = self.inter_col_sep * (nr_cols -1)
