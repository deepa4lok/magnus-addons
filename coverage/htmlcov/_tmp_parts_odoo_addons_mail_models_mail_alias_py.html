17def remove_accents(input_str): 

18 """Suboptimal-but-better-than-nothing way to replace accented 

19 latin letters by an ASCII equivalent. Will obviously change the 

20 meaning of input_str and work only for some cases""" 

21 input_str = ustr(input_str) 

22 nkfd_form = unicodedata.normalize('NFKD', input_str) 

23 return u''.join([c for c in nkfd_form if not unicodedata.combining(c)])

26class Alias(models.Model): 

27 """A Mail Alias is a mapping of an email address with a given OpenERP Document 

28 model. It is used by OpenERP's mail gateway when processing incoming emails 

29 sent to the system. If the recipient address (To) of the message matches 

30 a Mail Alias, the message will be either processed following the rules 

31 of that alias. If the message is a reply it will be attached to the 

32 existing discussion on the corresponding record, otherwise a new 

33 record of the corresponding model will be created. 

34 

35 This is meant to be used in combination with a catch-all email configuration 

36 on the company's mail server, so that as soon as a new mail.alias is 

37 created, it becomes immediately usable and OpenERP will accept email for it. 

38 """

101 @api.model 

102 def create(self, vals): 

103 """ Creates an email.alias record according to the values provided in ``vals``, 

104 with 2 alterations: the ``alias_name`` value may be suffixed in order to 

105 make it unique (and certain unsafe characters replaced), and 

106 he ``alias_model_id`` value will set to the model ID of the ``model_name`` 

107 context value, if provided. 

108 """ 

109 model_name = self._context.get('alias_model_name') 

110 parent_model_name = self._context.get('alias_parent_model_name') 

111 if vals.get('alias_name'): 

112 vals['alias_name'] = self._clean_and_make_unique(vals.get('alias_name')) 

113 if model_name: 

114 model = self.env['ir.model'].search([('model', '=', model_name)]) 

115 vals['alias_model_id'] = model.id 

116 if parent_model_name: 

117 model = self.env['ir.model'].search([('model', '=', parent_model_name)]) 

118 vals['alias_parent_model_id'] = model.id 

119 return super(Alias, self).create(vals)

128 @api.multi

131 mail catchall domain if exists from config otherwise "New Alias".

134 res = []

144 @api.model 

145 def _find_unique(self, name, alias_ids=False): 

146 """Find a unique alias name similar to ``name``. If ``name`` is 

147 already taken, make a variant by adding an integer suffix until 

148 an unused alias is found. 

149 """ 

150 sequence = None 

151 while True: 

152 new_name = "%s%s" % (name, sequence) if sequence is not None else name 

153 domain = [('alias_name', '=', new_name)] 

154 if alias_ids: 

155 domain += [('id', 'not in', alias_ids)] 

156 if not self.search(domain): 

157 break 

158 sequence = (sequence + 1) if sequence else 2 

159 return new_name

161 @api.model 

162 def _clean_and_make_unique(self, name, alias_ids=False): 

163 # when an alias name appears to already be an email, we keep the local part only 

164 name = remove_accents(name).lower().split('@')[0] 

165 name = re.sub(r'[^\w+.]+', '-', name) 

166 return self._find_unique(name, alias_ids=alias_ids)
