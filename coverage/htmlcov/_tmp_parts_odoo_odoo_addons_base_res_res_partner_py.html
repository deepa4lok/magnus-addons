51 for format_pattern, format_class in ADDRESS_FORMAT_CLASSES.iteritems(): 

52 if format_pattern in address_format:

55 # add address format class to address block 

56 address_node.attrib['class'] += ' ' + format_class 

57 if format_class.startswith('o_zip'):

60 if zip_fields and city_fields: 

61 # move zip field before city field 

62 city_fields[0].addprevious(zip_fields[0]) 

63 arch = etree.tostring(doc) 

64 break

441 @api.multi 

442 def _fields_sync(self, values): 

443 """ Sync commercial fields and address fields from company and to children after create/update, 

444 just as if those were all modeled as fields.related to the parent """ 

445 # 1. From UPSTREAM: sync from parent 

446 if values.get('parent_id') or values.get('type') == 'contact': 

447 # 1a. Commercial fields: sync if parent changed 

448 if values.get('parent_id'): 

449 self._commercial_sync_from_company() 

450 # 1b. Address fields: sync if parent or use_parent changed *and* both are now set  

451 if self.parent_id and self.type == 'contact': 

452 onchange_vals = self.onchange_parent_id().get('value', {}) 

453 self.update_address(onchange_vals) 

454 

455 # 2. To DOWNSTREAM: sync children 

456 if self.child_ids: 

457 # 2a. Commercial Fields: sync if commercial entity 

458 if self.commercial_partner_id == self: 

459 commercial_fields = self._commercial_fields() 

460 if any(field in values for field in commercial_fields): 

461 self._commercial_sync_to_children() 

462 for child in self.child_ids.filtered(lambda c: not c.is_company): 

463 if child.commercial_partner_id != self.commercial_partner_id : 

464 self._commercial_sync_to_children() 

465 break 

466 # 2b. Address fields: sync if address changed 

467 address_fields = self._address_fields() 

468 if any(field in values for field in address_fields): 

469 contacts = self.child_ids.filtered(lambda c: c.type == 'contact') 

470 contacts.update_address(values)

491 @api.multi 

492 def write(self, vals): 

493 # res.partner must only allow to set the company_id of a partner if it 

494 # is the same as the company of all users that inherit from this partner 

495 # (this is to allow the code from res_users to write to the partner!) or 

496 # if setting the company_id to False (this is compatible with any user 

497 # company) 

498 if vals.get('website'): 

499 vals['website'] = self._clean_website(vals['website']) 

500 if vals.get('parent_id'): 

501 vals['company_name'] = False 

502 if vals.get('company_id'): 

503 company = self.env['res.company'].browse(vals['company_id']) 

504 for partner in self: 

505 if partner.user_ids: 

506 companies = set(user.company_id for user in partner.user_ids) 

507 if len(companies) > 1 or company not in companies: 

508 raise UserError(_("You can not change the company as the partner/user has multiple user linked with different companies.")) 

509 tools.image_resize_images(vals) 

510 

511 result = True 

512 # To write in SUPERUSER on field is_company and avoid access rights problems. 

513 if 'is_company' in vals and self.user_has_groups('base.group_partner_manager') and not self.env.uid == SUPERUSER_ID: 

514 result = super(Partner, self.sudo()).write({'is_company': vals.get('is_company')}) 

515 del vals['is_company'] 

516 result = result and super(Partner, self).write(vals) 

517 for partner in self: 

518 if any(u.has_group('base.group_user') for u in partner.user_ids if u != self.env.user): 

519 self.env['res.users'].check_access_rights('write') 

520 partner._fields_sync(vals)

642 def name_search(self, name, args=None, operator='ilike', limit=100): 

643 if args is None: 

644 args = [] 

645 if name and operator in ('=', 'ilike', '=ilike', 'like', '=like'): 

646 self.check_access_rights('read') 

647 where_query = self._where_calc(args) 

648 self._apply_ir_rules(where_query, 'read') 

649 from_clause, where_clause, where_clause_params = where_query.get_sql() 

650 from_str = from_clause if from_clause else 'res_partner' 

651 where_str = where_clause and (" WHERE %s AND " % where_clause) or ' WHERE ' 

652 

653 # search on the name of the contacts and of its company 

654 search_name = name 

655 if operator in ('ilike', 'like'): 

656 search_name = '%%%s%%' % name 

657 if operator in ('=ilike', '=like'): 

658 operator = operator[1:] 

659 

660 unaccent = get_unaccent_wrapper(self.env.cr) 

661 

662 query = """SELECT res_partner.id 

663 FROM {from_str} 

664 {where} ({email} {operator} {percent} 

665 OR {display_name} {operator} {percent} 

666 OR {reference} {operator} {percent}) 

667 -- don't panic, trust postgres bitmap 

668 ORDER BY {display_name} {operator} {percent} desc, 

669 {display_name} 

670 """.format(from_str=from_str, 

671 where=where_str, 

672 operator=operator, 

673 email=unaccent('res_partner.email'), 

674 display_name=unaccent('res_partner.display_name'),

679 if limit: 

680 query += ' limit %s'

722 @api.multi 

723 def address_get(self, adr_pref=None): 

724 """ Find contacts/addresses of the right type(s) by doing a depth-first-search 

725 through descendants within company boundaries (stop at entities flagged ``is_company``) 

726 then continuing the search at the ancestors that are within the same company boundaries. 

727 Defaults to partners of type ``'default'`` when the exact type is not found, or to the 

728 provided partner itself if no type ``'default'`` is found either. """ 

729 adr_pref = set(adr_pref or []) 

730 if 'contact' not in adr_pref: 

731 adr_pref.add('contact') 

732 result = {} 

733 visited = set() 

734 for partner in self: 

735 current_partner = partner 

736 while current_partner: 

737 to_scan = [current_partner] 

738 # Scan descendants, DFS 

739 while to_scan: 

740 record = to_scan.pop(0) 

741 visited.add(record) 

742 if record.type in adr_pref and not result.get(record.type): 

743 result[record.type] = record.id 

744 if len(result) == len(adr_pref): 

745 return result 

746 to_scan = [c for c in record.child_ids 

747 if c not in visited 

748 if not c.is_company] + to_scan 

749 

750 # Continue scanning at ancestor if current_partner is not a commercial entity 

751 if current_partner.is_company or not current_partner.parent_id: 

752 break 

753 current_partner = current_partner.parent_id 

754 

755 # default to type 'contact' or the partner itself 

756 default = result.get('contact', self.id or False) 

757 for adr_type in adr_pref: 

758 result[adr_type] = result.get(adr_type) or default 

759 return result
