44 @api.multi 

45 def _compute_date_ranges(self): 

46 self.ensure_one() 

47 vals = rrule(freq=self.unit_of_time, interval=self.duration_count, 

48 dtstart=fields.Date.from_string(self.date_start), 

49 count=self.count+1) 

50 vals = list(vals) 

51 date_ranges = [] 

52 count_digits = len(unicode(self.count)) 

53 for idx, dt_start in enumerate(vals[:-1]): 

54 date_start = fields.Date.to_string(dt_start.date()) 

55 # always remove 1 day for the date_end since range limits are 

56 # inclusive 

57 dt_end = vals[idx+1].date() - relativedelta(days=1) 

58 date_end = fields.Date.to_string(dt_end) 

59 date_ranges.append({ 

60 'name': '%s%0*d' % ( 

61 self.name_prefix, count_digits, idx + 1), 

62 'date_start': date_start, 

63 'date_end': date_end, 

64 'type_id': self.type_id.id, 

65 'company_id': self.company_id.id, 

66 'parent_id': self.parent_id.id}) 

67 return date_ranges
