138class TransactionCase(BaseCase): 

139 """ TestCase in which each test method is run in its own transaction, 

140 and with its own cursor. The transaction is rolled back and the cursor 

141 is closed after each test. 

142 """ 

143 

144 def setUp(self): 

145 self.registry = odoo.registry(get_db_name()) 

146 #: current transaction's cursor 

147 self.cr = self.cursor() 

148 self.uid = odoo.SUPERUSER_ID 

149 #: :class:`~odoo.api.Environment` for the current test case 

150 self.env = api.Environment(self.cr, self.uid, {}) 

151 

152 @self.addCleanup 

153 def reset(): 

154 # rollback and close the cursor, and reset the environments 

155 self.registry.clear_caches() 

156 self.env.reset() 

157 self.cr.rollback() 

158 self.cr.close() 

159 

160 self.patch(type(self.env['res.partner']), '_get_gravatar_image', lambda *a: False)

162 def patch(self, obj, key, val): 

163 """ Do the patch ``setattr(obj, key, val)``, and prepare cleanup. """ 

164 old = getattr(obj, key) 

165 setattr(obj, key, val) 

166 self.addCleanup(setattr, obj, key, old)

168 def patch_order(self, model, order): 

169 """ Patch the order of the given model (name), and prepare cleanup. """ 

170 self.patch(type(self.env[model]), '_order', order)

196class SavepointCase(SingleTransactionCase): 

197 """ Similar to :class:`SingleTransactionCase` in that all test methods 

198 are run in a single transaction *but* each test case is run inside a 

199 rollbacked savepoint (sub-transaction). 

200 

201 Useful for test cases containing fast tests but with significant database 

202 setup common to all cases (complex in-db test data): :meth:`~.setUpClass` 

203 can be used to generate db test data once, then all test cases use the 

204 same data without influencing one another but without having to recreate 

205 the test data either. 

206 """ 

207 def setUp(self): 

208 self._savepoint_id = next(savepoint_seq) 

209 self.cr.execute('SAVEPOINT test_%d' % self._savepoint_id) 

210 def tearDown(self): 

211 self.cr.execute('ROLLBACK TO SAVEPOINT test_%d' % self._savepoint_id) 

212 self.env.clear() 

213 self.registry.clear_caches()

298 def phantom_poll(self, phantom, timeout): 

299 """ Phantomjs Test protocol. 

300 

301 Use console.log in phantomjs to output test results: 

302 

303 - for a success: console.log("ok") 

304 - for an error: console.log("error") 

305 

306 Other lines are relayed to the test log. 

307 

308 """ 

309 t0 = datetime.now() 

310 td = timedelta(seconds=timeout) 

311 buf = bytearray() 

312 while True: 

313 # timeout 

314 self.assertLess(datetime.now() - t0, td, 

315 "PhantomJS tests should take less than %s seconds" % timeout) 

316 

317 # read a byte 

318 try: 

319 ready, _, _ = select.select([phantom.stdout], [], [], 0.5) 

320 except select.error, e: 

321 # In Python 2, select.error has no relation to IOError or 

322 # OSError, and no errno/strerror/filename, only a pair of 

323 # unnamed arguments (matching errno and strerror) 

324 err, _ = e.args 

325 if err == errno.EINTR: 

326 continue 

327 raise 

328 

329 if ready: 

330 s = phantom.stdout.read(1) 

331 if not s: 

332 break 

333 buf.append(s) 

334 

335 # process lines 

336 if '\n' in buf and (not buf.startswith('<phantomLog>') or '</phantomLog>' in buf): 

337 if buf.startswith('<phantomLog>'): 

338 line = buf[12:buf.index('</phantomLog>')] 

339 buf = bytearray() 

340 else: 

341 line, buf = buf.split('\n', 1) 

342 line = str(line) 

343 

344 lline = line.lower() 

345 if lline.startswith(("error", "server application error")): 

346 try: 

347 # when errors occur the execution stack may be sent as a JSON 

348 prefix = lline.index('error') + 6 

349 _logger.error("phantomjs: %s", pformat(json.loads(line[prefix:]))) 

350 except ValueError: 

351 line_ = line.split('\n\n') 

352 _logger.error("phantomjs: %s", line_[0]) 

353 # The second part of the log is for debugging 

354 if len(line_) > 1: 

355 _logger.info("phantomjs: \n%s", line.split('\n\n', 1)[1]) 

356 pass 

357 break 

358 elif lline.startswith("warning"): 

359 _logger.warn("phantomjs: %s", line) 

360 else: 

361 _logger.info("phantomjs: %s", line) 

362 

363 if line == "ok": 

364 return True

366 def phantom_run(self, cmd, timeout): 

367 _logger.info('phantom_run executing %s', ' '.join(cmd)) 

368 

369 ls_glob = os.path.expanduser('~/.qws/share/data/Ofi Labs/PhantomJS/http_%s_%s.*' % (HOST, PORT)) 

370 ls_glob2 = os.path.expanduser('~/.local/share/Ofi Labs/PhantomJS/http_%s_%s.*' % (HOST, PORT)) 

371 for i in (glob.glob(ls_glob) + glob.glob(ls_glob2)): 

372 _logger.info('phantomjs unlink localstorage %s', i) 

373 os.unlink(i) 

374 try: 

375 phantom = subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=None) 

376 except OSError: 

377 raise unittest.SkipTest("PhantomJS not found") 

378 result = False 

379 try: 

380 result = self.phantom_poll(phantom, timeout) 

381 finally: 

382 # kill phantomjs if phantom.exit() wasn't called in the test 

383 if phantom.poll() is None: 

384 phantom.terminate() 

385 phantom.wait() 

386 self._wait_remaining_requests() 

387 # we ignore phantomjs return code as we kill it as soon as we have ok 

388 _logger.info("phantom_run execution finished") 

389 self.assertTrue( 

390 result, 

391 "PhantomJS test completed without reporting success; " 

392 "the log may contain errors or hints.")

394 def _wait_remaining_requests(self): 

395 t0 = int(time.time()) 

396 for thread in threading.enumerate(): 

397 if thread.name.startswith('odoo.service.http.request.'): 

398 thread.join_retry_count = 10 

399 while thread.isAlive(): 

400 # Need a busyloop here as thread.join() masks signals 

401 # and would prevent the forced shutdown. 

402 thread.join(0.05) 

403 thread.join_retry_count -= 1 

404 if thread.join_retry_count < 0: 

405 _logger.warning("Stop waiting for thread %s handling request for url %s", 

406 thread.name, thread.url) 

407 break 

408 time.sleep(0.5) 

409 t1 = int(time.time()) 

410 if t0 != t1: 

411 _logger.info('remaining requests') 

412 odoo.tools.misc.dumpstacks() 

413 t0 = t1
