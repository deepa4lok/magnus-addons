40 values = {

49 @api.model 

50 def prepare_backend_url(self, action_xml_id, view_type='list', module_name=None): 

51 """ prepare the backend url to the given action, or to the given module view. 

52 :param action_xml_id : the xml id of the action to redirect to 

53 :param view_type : the view type to display when redirecting (form, kanban, list, ...) 

54 :param module_name : the name of the module to display (if action_xml_id is 'open_module_tree'), or 

55 to redirect to if the action is not found. 

56 :returns url : the url to the correct page 

57 """ 

58 params = dict(view_type=view_type) 

59 # setting the action 

60 action = self.env.ref(action_xml_id, False) 

61 if action: 

62 params['action'] = action.id 

63 else: 

64 params['model'] = 'ir.module.module' 

65 # setting the module 

66 if module_name: 

67 module = self.env['ir.module.module'].sudo().search([('name', '=', module_name)], limit=1) 

68 if module: 

69 params['id'] = module.id 

70 else: 

71 return "#show_enterprise" 

72 return "/web#%s" % (urlencode(params),)
