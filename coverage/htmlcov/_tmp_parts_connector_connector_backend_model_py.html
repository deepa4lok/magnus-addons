79class ExternalBinding(models.AbstractModel): 

80 """ An abstract model for bindings to external records. 

81 

82 An external binding is a binding between a backend and Odoo. For 

83 example, for a partner, it could be ``magento.res.partner`` or for a 

84 product, ``magento.product``. 

85 

86 The final model, will be an ``_inherits`` of the Odoo model and 

87 will ``_inherit`` this model. 

88 

89 It will have a relation to the record (via ``_inherits``) and to the 

90 concrete backend model (``magento.backend`` for instance). 

91 

92 It will also contains all the data relative to the backend for the 

93 record. 

94 

95 It needs to implements at least these fields: 

96 

97 odoo_id 

98 The many2one to the record it links (used by ``_inherits``). 

99 

100 backend_id 

101 The many2one to the backend (for instance ``magento.backend``). 

102 

103 external_id 

104 The ID on the backend. 

105 

106 sync_date 

107 Last date of synchronization 

108 

109 

110 The definition of the field relations is to be done in the 

111 concrete classes because the relations themselves do not exist in 

112 this addon. 

113 

114 For example, for a ``res.partner.category`` from Magento, I would have 

115 (this is a consolidation of all the columns from the abstract models, 

116 in ``magentoerpconnect`` you would not find that):: 

117 

118 class MagentoResPartnerCategory(models.Model): 

119 _name = 'magento.res.partner.category' 

120 

121 _inherits = {'res.partner.category': 'odoo_id'} 

122 

123 odoo_id = fields.Many2one(comodel_name='res.partner.category', 

124 string='Partner Category', 

125 required=True, 

126 ondelete='cascade') 

127 backend_id = fields.Many2one( 

128 comodel_name='magento.backend', 

129 string='Magento Backend', 

130 required=True, 

131 ondelete='restrict') 

132 external_id = fields.Char(string='ID on Magento') 

133 tax_class_id = fields.Integer(string='Tax Class ID') 

134 

135 _sql_constraints = [ 

136 ('magento_uniq', 'unique(backend_id, magento_id)', 

137 'Partner Tag with same ID on Magento already exists.'), 

138 ] 

139 

140 

141 """ 

142 _name = 'external.binding' 

143 _description = 'External Binding (abstract)' 

144 

145 sync_date = fields.Datetime(string='Last synchronization date') 

146 # add other fields in concrete models 

147 # XXX we could add a default 'external_id'
