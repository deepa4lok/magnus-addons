436 tables_to_drop = set() 

437 

438 for field in self: 

439 if field.name in models.MAGIC_COLUMNS: 

440 continue 

441 model = self.env[field.model] 

442 self._cr.execute('SELECT relkind FROM pg_class WHERE relname=%s', (model._table,)) 

443 relkind = self._cr.fetchone() 

444 self._cr.execute("""SELECT column_name FROM information_schema.columns 

445 WHERE table_name=%s AND column_name=%s""", 

446 (model._table, field.name)) 

447 column_name = self._cr.fetchone() 

448 if column_name and (relkind and relkind[0] == 'r'): 

449 self._cr.execute('ALTER table "%s" DROP column "%s" cascade' % (model._table, field.name))

451 rel_name = field.relation_table or model._fields[field.name].relation

467 @api.multi 

468 def _prepare_update(self): 

469 """ Check whether the fields in ``self`` may be modified or removed. 

470 This method prevents the modification/deletion of many2one fields 

471 that have an inverse one2many, for instance. 

472 """ 

473 self = self.filtered(lambda record: record.state == 'manual') 

474 if not self: 

475 return 

476 

477 for record in self: 

478 model = self.env[record.model] 

479 field = model._fields[record.name] 

480 if field.type == 'many2one' and model._field_inverses.get(field): 

481 if self._context.get(MODULE_UNINSTALL_FLAG): 

482 # automatically unlink the corresponding one2many field(s) 

483 inverses = self.search([('relation', '=', field.model_name), 

484 ('relation_field', '=', field.name)]) 

485 inverses.unlink() 

486 continue 

487 msg = _("The field '%s' cannot be removed because the field '%s' depends on it.") 

488 raise UserError(msg % (field, model._field_inverses[field][0])) 

489 

490 # remove fields from registry, and check that views are not broken 

491 fields = [self.env[record.model]._pop_field(record.name) for record in self] 

492 domain = expression.OR([('arch_db', 'like', record.name)] for record in self) 

493 views = self.env['ir.ui.view'].search(domain) 

494 try: 

495 for view in views: 

496 view._check_xml() 

497 except Exception: 

498 raise UserError("\n".join([ 

499 _("Cannot rename/delete fields that are still present in views:"), 

500 _("Fields:") + " " + ", ".join(map(str, fields)), 

501 _("View:") + " " + view.name, 

502 ])) 

503 finally: 

504 # the registry has been modified, restore it 

505 self.pool.setup_models(self._cr)
