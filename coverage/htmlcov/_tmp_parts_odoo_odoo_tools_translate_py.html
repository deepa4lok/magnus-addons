155class XMLTranslator(object): 

156 """ A sequence of serialized XML/HTML items, with some of them to translate 

157 (todo) and others already translated (done). The purpose of this object 

158 is to simplify the handling of phrasing elements (like <b>) that must be 

159 translated together with their surrounding text. 

160 

161 For instance, the content of the "div" element below will be translated 

162 as a whole (without surrounding spaces): 

163 

164 <div> 

165 Lorem ipsum dolor sit amet, consectetur adipiscing elit, 

166 <b>sed</b> do eiusmod tempor incididunt ut labore et dolore 

167 magna aliqua. <span class="more">Ut enim ad minim veniam, 

168 <em>quis nostrud exercitation</em> ullamco laboris nisi ut 

169 aliquip ex ea commodo consequat.</span> 

170 </div> 

171 

172 """ 

173 def __init__(self, callback, method, parser=None): 

174 self.callback = callback # callback function to translate terms 

175 self.method = method # serialization method ('xml' or 'html') 

176 self.parser = parser # parser for validating translations 

177 self._done = [] # translated strings 

178 self._todo = [] # todo strings that come after _done 

179 self.needs_trans = False # whether todo needs translation 

180 

181 def todo(self, text, needs_trans=True): 

182 self._todo.append(text) 

183 if needs_trans and text.strip(): 

184 self.needs_trans = True 

185 

186 def all_todo(self): 

187 return not self._done 

188 

189 def get_todo(self): 

190 return "".join(self._todo) 

191 

192 def flush(self): 

193 if self._todo: 

194 todo = "".join(self._todo) 

195 done = self.process_text(todo) if self.needs_trans else todo 

196 self._done.append(done) 

197 del self._todo[:] 

198 self.needs_trans = False 

199 

200 def done(self, text): 

201 self.flush() 

202 self._done.append(text) 

203 

204 def get_done(self): 

205 """ Complete the translations and return the result. """ 

206 self.flush() 

207 return "".join(self._done) 

208 

209 def process_text(self, text): 

210 """ Translate text.strip(), but keep the surrounding spaces from text. """ 

211 term = text.strip() 

212 trans = term and self.callback(term) 

213 if trans: 

214 try: 

215 # parse the translation to validate it 

216 etree.fromstring("<div>%s</div>" % encode(trans), parser=self.parser) 

217 except etree.ParseError: 

218 # fallback: escape the translation 

219 trans = escape(trans) 

220 text = text.replace(term, trans) 

221 return text 

222 

223 def process_attr(self, attr): 

224 """ Translate the given node attribute value. """ 

225 term = attr.strip() 

226 trans = term and self.callback(term) 

227 return attr.replace(term, trans) if trans else attr 

228 

229 def process(self, node): 

230 """ Process the given xml `node`: collect `todo` and `done` items. """ 

231 if ( 

232 isinstance(node, SKIPPED_ELEMENT_TYPES) or 

233 node.tag in SKIPPED_ELEMENTS or 

234 node.get("t-translation", "").strip() == "off" or 

235 node.tag == "attribute" and node.get("name") not in TRANSLATED_ATTRS or 

236 node.getparent() is None and node.text and '<!DOCTYPE' in node.text 

237 ): 

238 # do not translate the contents of the node 

239 tail, node.tail = node.tail, None 

240 self.done(etree.tostring(node, method=self.method)) 

241 self.todo(escape(tail or "")) 

242 return 

243 

244 # process children nodes locally in child_trans 

245 child_trans = XMLTranslator(self.callback, self.method, parser=self.parser) 

246 if node.text: 

247 if avoid_pattern.match(node.text): 

248 child_trans.done(escape(node.text)) # do not translate <!DOCTYPE... 

249 else: 

250 child_trans.todo(escape(node.text)) 

251 for child in node: 

252 child_trans.process(child) 

253 

254 if (child_trans.all_todo() and 

255 node.tag in TRANSLATED_ELEMENTS and 

256 not any(attr.startswith("t-") for attr in node.attrib)): 

257 # serialize the node element as todo 

258 self.todo(self.serialize(node.tag, node.attrib, child_trans.get_todo()), 

259 child_trans.needs_trans) 

260 else: 

261 # complete translations and serialize result as done 

262 for attr in TRANSLATED_ATTRS: 

263 if node.get(attr): 

264 node.set(attr, self.process_attr(node.get(attr))) 

265 self.done(self.serialize(node.tag, node.attrib, child_trans.get_done())) 

266 

267 # add node tail as todo 

268 self.todo(escape(node.tail or "")) 

269 

270 def serialize(self, tag, attrib, content): 

271 """ Return a serialized element with the given `tag`, attributes 

272 `attrib`, and already-serialized `content`. 

273 """ 

274 if content: 

275 elem = etree.tostring(etree.Element(tag, attrib), method='xml') 

276 assert elem.endswith("/>") 

277 return "%s>%s</%s>" % (elem[:-2], content, tag) 

278 else: 

279 return etree.tostring(etree.Element(tag, attrib), method=self.method)

282def xml_translate(callback, value): 

283 """ Translate an XML value (string), using `callback` for translating text 

284 appearing in `value`. 

285 """ 

286 if not value: 

287 return value 

288 

289 trans = XMLTranslator(callback, 'xml') 

290 try: 

291 root = etree.fromstring(encode(value)) 

292 trans.process(root) 

293 return trans.get_done() 

294 except etree.ParseError: 

295 # fallback for translated terms: use an HTML parser and wrap the term 

296 wrapped = "<div>%s</div>" % encode(value) 

297 root = etree.fromstring(wrapped, etree.HTMLParser(encoding='utf-8')) 

298 trans.process(root[0][0]) # html > body > div 

299 return trans.get_done()[5:-6] # remove tags <div> and </div> 

300 

301def html_translate(callback, value): 

302 """ Translate an HTML value (string), using `callback` for translating text 

303 appearing in `value`. 

304 """ 

305 if not value: 

306 return value 

307 

308 try: 

309 parser = etree.HTMLParser(encoding='utf-8') 

310 trans = XMLTranslator(callback, 'html', parser) 

311 wrapped = "<div>%s</div>" % encode(value) 

312 root = etree.fromstring(wrapped, parser) 

313 trans.process(root[0][0]) # html > body > div 

314 value = trans.get_done()[5:-6] # remove tags <div> and </div> 

315 except ValueError: 

316 _logger.exception("Cannot translate malformed HTML, using source value instead") 

317 return value

342 def _get_cr(self, frame, allow_create=True): 

343 # try, in order: cr, cursor, self.env.cr, self.cr, 

344 # request.env.cr 

345 if 'cr' in frame.f_locals: 

346 return frame.f_locals['cr'], False 

347 if 'cursor' in frame.f_locals: 

348 return frame.f_locals['cursor'], False 

349 s = frame.f_locals.get('self') 

350 if hasattr(s, 'env'): 

351 return s.env.cr, False 

352 if hasattr(s, 'cr'): 

353 return s.cr, False 

354 try: 

355 from odoo.http import request 

356 return request.env.cr, False 

357 except RuntimeError: 

358 pass 

359 if allow_create: 

360 # create a new cursor 

361 db = self._get_db() 

362 if db is not None: 

363 return db.cursor(), True 

364 return None, False

366 def _get_uid(self, frame): 

367 # try, in order: uid, user, self.env.uid 

368 if 'uid' in frame.f_locals: 

369 return frame.f_locals['uid'] 

370 if 'user' in frame.f_locals: 

371 return int(frame.f_locals['user']) # user may be a record

375 def _get_lang(self, frame): 

376 # try, in order: context.get('lang'), kwargs['context'].get('lang'), 

377 # self.env.lang, self.localcontext.get('lang'), request.env.lang 

378 lang = None 

379 if frame.f_locals.get('context'): 

380 lang = frame.f_locals['context'].get('lang') 

381 if not lang: 

382 kwargs = frame.f_locals.get('kwargs', {}) 

383 if kwargs.get('context'): 

384 lang = kwargs['context'].get('lang') 

385 if not lang: 

386 s = frame.f_locals.get('self') 

387 if hasattr(s, 'env'): 

388 lang = s.env.lang 

389 if not lang: 

390 if hasattr(s, 'localcontext'): 

391 lang = s.localcontext.get('lang') 

392 if not lang: 

393 try: 

394 from odoo.http import request 

395 lang = request.env.lang 

396 except RuntimeError: 

397 pass 

398 if not lang: 

399 # Last resort: attempt to guess the language of the user 

400 # Pitfall: some operations are performed in sudo mode, and we 

401 # don't know the originial uid, so the language may 

402 # be wrong when the admin language differs. 

403 (cr, dummy) = self._get_cr(frame, allow_create=False) 

404 uid = self._get_uid(frame) 

405 if cr and uid: 

406 env = odoo.api.Environment(cr, uid, {}) 

407 lang = env['res.users'].context_get()['lang'] 

408 return lang

410 def __call__(self, source): 

411 res = source 

412 cr = None 

413 is_new_cr = False 

414 try: 

415 frame = inspect.currentframe() 

416 if frame is None: 

417 return source 

418 frame = frame.f_back 

419 if not frame: 

420 return source 

421 lang = self._get_lang(frame)

428 else:

430 else: 

431 _logger.debug('no translation language detected, skipping translation for "%r" ', source) 

432 except Exception:

434 # if so, double-check the root/base translations filenames 

435 finally: 

436 if cr and is_new_cr: 

437 cr.close() 

438 return res

443def quote(s): 

444 """Returns quoted PO term string, with special PO characters escaped""" 

445 assert r"\n" not in s, "Translation terms may not include escaped newlines ('\\n'), please use only literal newlines! (in '%s')" % s 

446 return '"%s"' % s.replace('\\','\\\\') \ 

447 .replace('"','\\"') \ 

448 .replace('\n', '\\n"\n"') 

449 

450re_escaped_char = re.compile(r"(\\.)") 

451re_escaped_replacements = {'n': '\n', 't': '\t',} 

452 

453def _sub_replacement(match_obj): 

454 return re_escaped_replacements.get(match_obj.group(1)[1], match_obj.group(1)[1]) 

455 

456def unquote(str): 

457 """Returns unquoted PO term string, with special PO characters unescaped""" 

458 return re_escaped_char.sub(_sub_replacement, str[1:-1])

460# class to handle po files 

461class PoFile(object): 

462 def __init__(self, buffer): 

463 self.buffer = buffer 

464 

465 def __iter__(self): 

466 self.buffer.seek(0) 

467 self.lines = self._get_lines() 

468 self.lines_count = len(self.lines) 

469 

470 self.first = True 

471 self.extra_lines= [] 

472 return self 

473 

474 def _get_lines(self): 

475 lines = self.buffer.readlines() 

476 # remove the BOM (Byte Order Mark): 

477 if len(lines): 

478 lines[0] = unicode(lines[0], 'utf8').lstrip(unicode( codecs.BOM_UTF8, "utf8")) 

479 

480 lines.append('') # ensure that the file ends with at least an empty line 

481 return lines 

482 

483 def cur_line(self): 

484 return self.lines_count - len(self.lines) 

485 

486 def next(self): 

487 trans_type = name = res_id = source = trad = None 

488 if self.extra_lines: 

489 trans_type, name, res_id, source, trad, comments = self.extra_lines.pop(0) 

490 if not res_id: 

491 res_id = '0' 

492 else: 

493 comments = [] 

494 targets = [] 

495 line = None 

496 fuzzy = False 

497 while not line: 

498 if 0 == len(self.lines): 

499 raise StopIteration() 

500 line = self.lines.pop(0).strip() 

501 while line.startswith('#'): 

502 if line.startswith('#~ '): 

503 break 

504 if line.startswith('#.'): 

505 line = line[2:].strip() 

506 if not line.startswith('module:'): 

507 comments.append(line) 

508 elif line.startswith('#:'): 

509 # Process the `reference` comments. Each line can specify 

510 # multiple targets (e.g. model, view, code, selection, 

511 # ...). For each target, we will return an additional 

512 # entry. 

513 for lpart in line[2:].strip().split(' '): 

514 trans_info = lpart.strip().split(':',2) 

515 if trans_info and len(trans_info) == 2: 

516 # looks like the translation trans_type is missing, which is not 

517 # unexpected because it is not a GetText standard. Default: 'code' 

518 trans_info[:0] = ['code'] 

519 if trans_info and len(trans_info) == 3: 

520 # this is a ref line holding the destination info (model, field, record) 

521 targets.append(trans_info) 

522 elif line.startswith('#,') and (line[2:].strip() == 'fuzzy'): 

523 fuzzy = True 

524 line = self.lines.pop(0).strip() 

525 if not self.lines: 

526 raise StopIteration() 

527 while not line: 

528 # allow empty lines between comments and msgid 

529 line = self.lines.pop(0).strip() 

530 if line.startswith('#~ '): 

531 while line.startswith('#~ ') or not line.strip(): 

532 if 0 == len(self.lines): 

533 raise StopIteration() 

534 line = self.lines.pop(0) 

535 # This has been a deprecated entry, don't return anything 

536 return self.next() 

537 

538 if not line.startswith('msgid'): 

539 raise Exception("malformed file: bad line: %s" % line) 

540 source = unquote(line[6:]) 

541 line = self.lines.pop(0).strip() 

542 if not source and self.first: 

543 self.first = False 

544 # if the source is "" and it's the first msgid, it's the special 

545 # msgstr with the informations about the traduction and the 

546 # traductor; we skip it 

547 self.extra_lines = [] 

548 while line: 

549 line = self.lines.pop(0).strip() 

550 return self.next() 

551 

552 while not line.startswith('msgstr'): 

553 if not line: 

554 raise Exception('malformed file at %d'% self.cur_line()) 

555 source += unquote(line) 

556 line = self.lines.pop(0).strip() 

557 

558 trad = unquote(line[7:]) 

559 line = self.lines.pop(0).strip() 

560 while line: 

561 trad += unquote(line) 

562 line = self.lines.pop(0).strip() 

563 

564 if targets and not fuzzy: 

565 # Use the first target for the current entry (returned at the 

566 # end of this next() call), and keep the others to generate 

567 # additional entries (returned the next next() calls). 

568 trans_type, name, res_id = targets.pop(0) 

569 for t, n, r in targets: 

570 if t == trans_type == 'code': continue 

571 self.extra_lines.append((t, n, r, source, trad, comments)) 

572 

573 if name is None: 

574 if not fuzzy: 

575 _logger.warning('Missing "#:" formated comment at line %d for the following source:\n\t%s', 

576 self.cur_line(), source[:30]) 

577 return self.next() 

578 return trans_type, name, res_id, source, trad, '\n'.join(comments)

603 'now': datetime.utcnow().strftime('%Y-%m-%d %H:%M')+"+0000", 

604 } 

605 ) 

606 

607 def write(self, modules, tnrs, source, trad, comments=None): 

608 

609 plurial = len(modules) > 1 and 's' or '' 

610 self.buffer.write("#. module%s: %s\n" % (plurial, ', '.join(modules))) 

611 

612 if comments: 

613 self.buffer.write(''.join(('#. %s\n' % c for c in comments))) 

614 

615 code = False 

616 for typy, name, res_id in tnrs: 

617 self.buffer.write("#: %s:%s:%s\n" % (typy, name, res_id)) 

618 if typy == 'code': 

619 code = True 

620 

621 if code: 

622 # only strings in python code are python formated 

623 self.buffer.write("#, python-format\n") 

624 

625 if not isinstance(trad, unicode): 

626 trad = unicode(trad, 'utf8') 

627 if not isinstance(source, unicode): 

628 source = unicode(source, 'utf8') 

629 

630 msg = "msgid %s\n" \ 

631 "msgstr %s\n\n" \ 

632 % (quote(source), quote(trad)) 

633 self.buffer.write(msg.encode('utf8'))

638def trans_export(lang, modules, buffer, format, cr): 

639 

640 def _process(format, modules, rows, buffer, lang): 

641 if format == 'csv': 

642 writer = csv.writer(buffer, 'UNIX') 

643 # write header first 

644 writer.writerow(("module","type","name","res_id","src","value","comments")) 

645 for module, type, name, res_id, src, trad, comments in rows: 

646 comments = '\n'.join(comments) 

647 writer.writerow((module, type, name, res_id, src, trad, comments)) 

648 

649 elif format == 'po': 

650 writer = PoFile(buffer) 

651 writer.write_infos(modules) 

652 

653 # we now group the translations by source. That means one translation per source. 

654 grouped_rows = {} 

655 for module, type, name, res_id, src, trad, comments in rows: 

656 row = grouped_rows.setdefault(src, {}) 

657 row.setdefault('modules', set()).add(module) 

658 if not row.get('translation') and trad != src: 

659 row['translation'] = trad 

660 row.setdefault('tnrs', []).append((type, name, res_id)) 

661 row.setdefault('comments', set()).update(comments) 

662 

663 for src, row in sorted(grouped_rows.items()): 

664 if not lang: 

665 # translation template, so no translation value 

666 row['translation'] = '' 

667 elif not row.get('translation'): 

668 row['translation'] = src 

669 writer.write(row['modules'], row['tnrs'], src, row['translation'], row['comments']) 

670 

671 elif format == 'tgz': 

672 rows_by_module = {} 

673 for row in rows: 

674 module = row[0] 

675 rows_by_module.setdefault(module, []).append(row) 

676 tmpdir = tempfile.mkdtemp() 

677 for mod, modrows in rows_by_module.items(): 

678 tmpmoddir = join(tmpdir, mod, 'i18n') 

679 os.makedirs(tmpmoddir) 

680 pofilename = (lang if lang else mod) + ".po" + ('t' if not lang else '') 

681 buf = file(join(tmpmoddir, pofilename), 'w') 

682 _process('po', [mod], modrows, buf, lang) 

683 buf.close() 

684 

685 tar = tarfile.open(fileobj=buffer, mode='w|gz') 

686 tar.add(tmpdir, '') 

687 tar.close() 

688 

689 else: 

690 raise Exception(_('Unrecognized extension: must be one of ' 

691 '.csv, .po, or .tgz (received .%s).') % format) 

692 

693 translations = trans_generate(lang, modules, cr) 

694 modules = set(t[0] for t in translations) 

695 _process(format, modules, translations, buffer, lang) 

696 del translations

699def trans_parse_rml(de): 

700 res = [] 

701 for n in de: 

702 for m in n: 

703 if isinstance(m, SKIPPED_ELEMENT_TYPES) or not m.text: 

704 continue 

705 string_list = [s.replace('\n', ' ').strip() for s in re.split('\[\[.+?\]\]', m.text)] 

706 for s in string_list: 

707 if s: 

708 res.append(s.encode("utf8")) 

709 res.extend(trans_parse_rml(n)) 

710 return res

713def _push(callback, term, source_line): 

714 """ Sanity check before pushing translation terms """ 

715 term = (term or "").strip().encode('utf8') 

716 # Avoid non-char tokens like ':' '...' '.00' etc. 

717 if len(term) > 8 or any(x.isalpha() for x in term): 

718 callback(term, source_line)

736def _extract_translatable_qweb_terms(element, callback): 

737 """ Helper method to walk an etree document representing 

738 a QWeb template, and call ``callback(term)`` for each 

739 translatable term that is found in the document. 

740 

741 :param etree._Element element: root of etree document to extract terms from 

742 :param Callable callback: a callable in the form ``f(term, source_line)``, 

743 that will be called for each extracted term. 

744 """ 

745 # not using elementTree.iterparse because we need to skip sub-trees in case 

746 # the ancestor element had a reason to be skipped 

747 for el in element: 

748 if isinstance(el, SKIPPED_ELEMENT_TYPES): continue 

749 if (el.tag.lower() not in SKIPPED_ELEMENTS 

750 and "t-js" not in el.attrib 

751 and not ("t-jquery" in el.attrib and "t-operation" not in el.attrib) 

752 and el.get("t-translation", '').strip() != "off"): 

753 _push(callback, el.text, el.sourceline) 

754 for att in ('title', 'alt', 'label', 'placeholder'): 

755 if att in el.attrib: 

756 _push(callback, el.attrib[att], el.sourceline) 

757 _extract_translatable_qweb_terms(el, callback) 

758 _push(callback, el.tail, el.sourceline)

761def babel_extract_qweb(fileobj, keywords, comment_tags, options): 

762 """Babel message extractor for qweb template files. 

763 

764 :param fileobj: the file-like object the messages should be extracted from 

765 :param keywords: a list of keywords (i.e. function names) that should 

766 be recognized as translation functions 

767 :param comment_tags: a list of translator tags to search for and 

768 include in the results 

769 :param options: a dictionary of additional options (optional) 

770 :return: an iterator over ``(lineno, funcname, message, comments)`` 

771 tuples 

772 :rtype: Iterable 

773 """ 

774 result = []

776 result.append((lineno, None, text, [])) 

777 tree = etree.parse(fileobj) 

778 _extract_translatable_qweb_terms(tree.getroot(), handle_text)

786 def push_translation(module, type, name, id, source, comments=None): 

787 # empty and one-letter terms are ignored, they probably are not meant to be 

788 # translated, and would be very hard to translate anyway. 

789 sanitized_term = (source or '').strip() 

790 try: 

791 # verify the minimal size without eventual xml tags 

792 # wrap to make sure html content like '<a>b</a><c>d</c>' is accepted by lxml 

793 wrapped = "<div>%s</div>" % sanitized_term 

794 node = etree.fromstring(wrapped) 

795 sanitized_term = etree.tostring(node, encoding='UTF-8', method='text') 

796 except etree.ParseError: 

797 pass 

798 # remove non-alphanumeric chars 

799 sanitized_term = re.sub(r'\W+', '', sanitized_term) 

800 if not sanitized_term or len(sanitized_term) <= 1: 

801 return 

802 

803 tnx = (module, source, name, id, type, tuple(comments or ())) 

804 to_translate.add(tnx) 

805 

806 query = 'SELECT name, model, res_id, module FROM ir_model_data' 

807 query_models = """SELECT m.id, m.model, imd.module 

808 FROM ir_model AS m, ir_model_data AS imd 

809 WHERE m.id = imd.res_id AND imd.model = 'ir.model'""" 

810 

811 if 'all_installed' in modules: 

812 query += ' WHERE module IN ( SELECT name FROM ir_module_module WHERE state = \'installed\') ' 

813 query_models += " AND imd.module in ( SELECT name FROM ir_module_module WHERE state = 'installed') " 

814 

815 if 'all' not in modules: 

816 query += ' WHERE module IN %s' 

817 query_models += ' AND imd.module IN %s' 

818 query_param = (tuple(modules),) 

819 else: 

820 query += ' WHERE module != %s' 

821 query_models += ' AND imd.module != %s' 

822 query_param = ('__export__',) 

823 

824 query += ' ORDER BY module, model, name' 

825 query_models += ' ORDER BY module, model' 

826 

827 cr.execute(query, query_param) 

828 

829 for (xml_name, model, res_id, module) in cr.fetchall(): 

830 module = encode(module) 

831 model = encode(model) 

832 xml_name = "%s.%s" % (module, encode(xml_name))

835 _logger.error("Unable to find object %r", model)

849 field_name = encode(record.name)

955 def babel_extract_terms(fname, path, root, extract_method="python", trans_type='code', 

956 extra_comments=None, extract_keywords={'_': None}): 

957 module, fabsolutepath, _, display_path = verified_module_filepaths(fname, path, root) 

958 extra_comments = extra_comments or [] 

959 if not module: return 

960 src_file = open(fabsolutepath, 'r') 

961 try: 

962 for extracted in extract.extract(extract_method, src_file, keywords=extract_keywords): 

963 # Babel 0.9.6 yields lineno, message, comments 

964 # Babel 1.3 yields lineno, message, comments, context 

965 lineno, message, comments = extracted[:3] 

966 push_translation(module, trans_type, display_path, lineno, 

967 encode(message), comments + extra_comments) 

968 except Exception: 

969 _logger.exception("Failed to extract terms from %s", fabsolutepath) 

970 finally: 

971 src_file.close() 

972 

973 for (path, recursive) in path_list: 

974 _logger.debug("Scanning files of modules at %s", path) 

975 for root, dummy, files in walksymlinks(path): 

976 for fname in fnmatch.filter(files, '*.py'): 

977 babel_extract_terms(fname, path, root) 

978 # mako provides a babel extractor: http://docs.makotemplates.org/en/latest/usage.html#babel 

979 for fname in fnmatch.filter(files, '*.mako'): 

980 babel_extract_terms(fname, path, root, 'mako', trans_type='report') 

981 # Javascript source files in the static/src/js directory, rest is ignored (libs) 

982 if fnmatch.fnmatch(root, '*/static/src/js*'): 

983 for fname in fnmatch.filter(files, '*.js'): 

984 babel_extract_terms(fname, path, root, 'javascript', 

985 extra_comments=[WEB_TRANSLATION_COMMENT], 

986 extract_keywords={'_t': None, '_lt': None}) 

987 # QWeb template files 

988 if fnmatch.fnmatch(root, '*/static/src/xml*'): 

989 for fname in fnmatch.filter(files, '*.xml'): 

990 babel_extract_terms(fname, path, root, 'odoo.tools.translate:babel_extract_qweb', 

991 extra_comments=[WEB_TRANSLATION_COMMENT]) 

992 if not recursive: 

993 # due to topdown, first iteration is in first level 

994 break 

995 

996 out = [] 

997 # translate strings marked as to be translated 

998 Translation = env['ir.translation'] 

999 for module, source, name, id, type, comments in sorted(to_translate): 

1000 trans = Translation._get_source(name, type, lang, source) if lang else "" 

1001 out.append((module, type, name, id, source, encode(trans) or '', comments)) 

1002 return out

1018def trans_load_data(cr, fileobj, fileformat, lang, lang_name=None, verbose=True, module_name=None, context=None): 

1019 """Populates the ir_translation table.""" 

1020 if verbose: 

1021 _logger.info('loading translation file for language %s', lang) 

1022 

1023 env = odoo.api.Environment(cr, SUPERUSER_ID, context or {}) 

1024 Lang = env['res.lang'] 

1025 Translation = env['ir.translation'] 

1026 

1027 try: 

1028 if not Lang.search_count([('code', '=', lang)]): 

1029 # lets create the language with locale information 

1030 Lang.load_lang(lang=lang, lang_name=lang_name) 

1031 

1032 # Parse also the POT: it will possibly provide additional targets. 

1033 # (Because the POT comments are correct on Launchpad but not the 

1034 # PO comments due to a Launchpad limitation. See LP bug 933496.) 

1035 pot_reader = [] 

1036 

1037 # now, the serious things: we read the language file 

1038 fileobj.seek(0) 

1039 if fileformat == 'csv': 

1040 reader = csv.reader(fileobj, quotechar='"', delimiter=',') 

1041 # read the first line of the file (it contains columns titles) 

1042 for row in reader: 

1043 fields = row 

1044 break 

1045 

1046 elif fileformat == 'po': 

1047 reader = PoFile(fileobj) 

1048 fields = ['type', 'name', 'res_id', 'src', 'value', 'comments'] 

1049 

1050 # Make a reader for the POT file and be somewhat defensive for the 

1051 # stable branch. 

1052 if fileobj.name.endswith('.po'): 

1053 try: 

1054 # Normally the path looks like /path/to/xxx/i18n/lang.po 

1055 # and we try to find the corresponding 

1056 # /path/to/xxx/i18n/xxx.pot file. 

1057 # (Sometimes we have 'i18n_extra' instead of just 'i18n') 

1058 addons_module_i18n, _ignored = os.path.split(fileobj.name) 

1059 addons_module, i18n_dir = os.path.split(addons_module_i18n) 

1060 addons, module = os.path.split(addons_module) 

1061 pot_handle = file_open(os.path.join( 

1062 addons, module, i18n_dir, module + '.pot')) 

1063 pot_reader = PoFile(pot_handle) 

1064 except: 

1065 pass 

1066 

1067 else: 

1068 _logger.info('Bad file format: %s', fileformat) 

1069 raise Exception(_('Bad file format: %s') % fileformat) 

1070 

1071 # Read the POT references, and keep them indexed by source string. 

1072 class Target(object): 

1073 def __init__(self): 

1074 self.value = None 

1075 self.targets = set() # set of (type, name, res_id) 

1076 self.comments = None 

1077 

1078 pot_targets = defaultdict(Target) 

1079 for type, name, res_id, src, _ignored, comments in pot_reader: 

1080 if type is not None: 

1081 target = pot_targets[src] 

1082 target.targets.add((type, name, res_id)) 

1083 target.comments = comments 

1084 

1085 # read the rest of the file 

1086 irt_cursor = Translation._get_import_cursor() 

1087 

1088 def process_row(row): 

1089 """Process a single PO (or POT) entry.""" 

1090 # dictionary which holds values for this line of the csv file 

1091 # {'lang': ..., 'type': ..., 'name': ..., 'res_id': ..., 

1092 # 'src': ..., 'value': ..., 'module':...} 

1093 dic = dict.fromkeys(('type', 'name', 'res_id', 'src', 'value', 

1094 'comments', 'imd_model', 'imd_name', 'module')) 

1095 dic['lang'] = lang 

1096 dic.update(zip(fields, row)) 

1097 

1098 # discard the target from the POT targets. 

1099 src = dic['src'] 

1100 if src in pot_targets: 

1101 target = pot_targets[src] 

1102 target.value = dic['value'] 

1103 target.targets.discard((dic['type'], dic['name'], dic['res_id'])) 

1104 

1105 # This would skip terms that fail to specify a res_id 

1106 res_id = dic['res_id'] 

1107 if not res_id: 

1108 return 

1109 

1110 if isinstance(res_id, (int, long)) or \ 

1111 (isinstance(res_id, basestring) and res_id.isdigit()): 

1112 dic['res_id'] = int(res_id) 

1113 if module_name: 

1114 dic['module'] = module_name 

1115 else: 

1116 # res_id is an xml id 

1117 dic['res_id'] = None 

1118 dic['imd_model'] = dic['name'].split(',')[0] 

1119 if '.' in res_id: 

1120 dic['module'], dic['imd_name'] = res_id.split('.', 1) 

1121 else: 

1122 dic['module'], dic['imd_name'] = module_name, res_id 

1123 

1124 irt_cursor.push(dic) 

1125 

1126 # First process the entries from the PO file (doing so also fills/removes 

1127 # the entries from the POT file). 

1128 for row in reader: 

1129 process_row(row) 

1130 

1131 # Then process the entries implied by the POT file (which is more 

1132 # correct w.r.t. the targets) if some of them remain. 

1133 pot_rows = [] 

1134 for src, target in pot_targets.iteritems(): 

1135 if target.value: 

1136 for type, name, res_id in target.targets: 

1137 pot_rows.append((type, name, res_id, src, target.value, target.comments)) 

1138 pot_targets.clear() 

1139 for row in pot_rows: 

1140 process_row(row) 

1141 

1142 irt_cursor.finish() 

1143 Translation.clear_caches() 

1144 if verbose: 

1145 _logger.info("translation file loaded succesfully") 

1146 

1147 except IOError: 

1148 iso_lang = get_iso_codes(lang) 

1149 filename = '[lang: %s][format: %s]' % (iso_lang or 'new', fileformat) 

1150 _logger.exception("couldn't read translation file %s", filename)
