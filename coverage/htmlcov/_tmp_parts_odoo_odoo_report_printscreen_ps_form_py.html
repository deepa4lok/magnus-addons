49 def _create_table(self, uid, ids, fields, fields_order, results, context, title=''): 

50 pageSize=[297.0,210.0] 

51 

52 new_doc = etree.Element("report") 

53 config = etree.SubElement(new_doc, 'config') 

54 

55 # build header 

56 def _append_node(name, text):

60 _append_node('date', time.strftime('%d/%m/%Y')) 

61 _append_node('PageSize', '%.2fmm,%.2fmm' % tuple(pageSize))

64 _append_node('report-header', title)

66 l = [] 

67 t = 0 

68 strmax = (pageSize[0]-40) * 2.8346 

69 for f in fields_order: 

70 s = 0 

71 if fields[f]['type'] in ('date','time','float','integer'): 

72 s = 60 

73 strmax -= s 

74 else: 

75 t += fields[f].get('size', 56) / 28 + 1 

76 l.append(s)

79 s = fields[fields_order[pos]].get('size', 56) / 28 + 1 

80 l[pos] = strmax * s / t 

81 _append_node('tableSize', ','.join(map(str,l)) )

83 header = etree.SubElement(new_doc, 'header') 

84 for f in fields_order:

87 

88 lines = etree.SubElement(new_doc, 'lines')

91 for f in fields_order:

98 line[f]=round(line[f],precision) 

99 col = etree.SubElement(node_line, 'col', tree='no')
