51class Diag(T): 

52 "This class fills the region with diagonal lines." 

53 

54 def draw(self, can, x1, y1, x2, y2): 

55 line_width = self.line_style.width 

56 interval = self.line_interval * 1.414 

57 x1 -= line_width 

58 y1 -= line_width 

59 x2 += line_width 

60 y2 += line_width 

61 length = max(y2 - y1, x2 - x1) 

62 curx = x1 - length 

63 while curx < x2: 

64 can.line(self.line_style, curx, y1, curx+length, y1+length) 

65 curx += interval

114class Wave(T): 

115 "Fills the region with horizontal wavy lines." 

116 def draw(self, can, x1, y1, x2, y2): 

117 x1 = xscale(x1) 

118 x2 = xscale(x2) 

119 y1 = yscale(y1) 

120 y2 = yscale(y2) 

121 line_width = nscale(self.line_style.width) 

122 interval = nscale(self.line_interval) 

123 

124 can.set_line_style(self.line_style) 

125 x1 -= line_width 

126 x2 += line_width 

127 cury = y1 

128 half = interval/2.0 

129 while cury < y2: 

130 curx = x1 

131 can.newpath() 

132 can.moveto(curx, cury) 

133 while curx < x2: 

134 can.lineto(curx + half, cury + half) 

135 can.lineto(curx + interval, cury) 

136 curx += interval 

137 can.stroke() 

138 cury += interval

166class Lines(T): 

167 """Fills the region with a series of short line segments.""" 

168 def draw(self, can, x1, y1, x2, y2): 

169 interval = nscale(self.line_interval) 

170 cury = y1 

171 j = 0 

172 while cury < y2: 

173 curx = x1 

174 if j % 2 == 1: 

175 curx += interval/2.0 

176 while curx < x2: 

177 can.line(self.line_style, curx, cury, curx+interval/2.0, cury) 

178 curx += interval * 1.5 

179 j += 1 

180 cury += interval
