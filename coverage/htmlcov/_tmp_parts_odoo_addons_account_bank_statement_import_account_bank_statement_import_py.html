31 @api.multi 

32 def import_file(self): 

33 """ Process the file chosen in the wizard, create bank statement(s) and go to reconciliation. """ 

34 self.ensure_one() 

35 # Let the appropriate implementation module parse the file and return the required data 

36 # The active_id is passed in context in case an implementation module requires information about the wizard state (see QIF) 

37 currency_code, account_number, stmts_vals = self.with_context(active_id=self.ids[0])._parse_file(base64.b64decode(self.data_file)) 

38 # Check raw data 

39 self._check_parsed_data(stmts_vals) 

40 # Try to find the currency and journal in odoo 

41 currency, journal = self._find_additional_data(currency_code, account_number) 

42 # If no journal found, ask the user about creating one 

43 if not journal: 

44 # The active_id is passed in context so the wizard can call import_file again once the journal is created 

45 return self.with_context(active_id=self.ids[0])._journal_creation_wizard(currency, account_number) 

46 if not journal.default_debit_account_id or not journal.default_credit_account_id: 

47 raise UserError(_('You have to set a Default Debit Account and a Default Credit Account for the journal: %s') % (journal.name,)) 

48 # Prepare statement data to be used for bank statements creation 

49 stmts_vals = self._complete_stmts_vals(stmts_vals, journal, account_number) 

50 # Create the bank statements 

51 statement_ids, notifications = self._create_bank_statements(stmts_vals) 

52 # Now that the import worked out, set it as the bank_statements_source of the journal 

53 if journal.bank_statements_source != 'file_import': 

54 # Use sudo() because only 'account.group_account_manager' 

55 # has write access on 'account.journal', but 'account.group_account_user' 

56 # must be able to import bank statement files 

57 journal.sudo().bank_statements_source = 'file_import' 

58 # Finally dispatch to reconciliation interface 

59 action = self.env.ref('account.action_bank_reconcile_bank_statements') 

60 return { 

61 'name': action.name, 

62 'tag': action.tag, 

63 'context': { 

64 'statement_ids': statement_ids, 

65 'notifications': notifications 

66 }, 

67 'type': 'ir.actions.client', 

68 }

88 def _parse_file(self, data_file): 

89 """ Each module adding a file support must extends this method. It processes the file if it can, returns super otherwise, resulting in a chain of responsability. 

90 This method parses the given file and returns the data required by the bank statement import process, as specified below. 

91 rtype: triplet (if a value can't be retrieved, use None) 

92 - currency code: string (e.g: 'EUR') 

93 The ISO 4217 currency code, case insensitive 

94 - account number: string (e.g: 'BE1234567890') 

95 The number of the bank account which the statement belongs to 

96 - bank statements data: list of dict containing (optional items marked by o) : 

97 - 'name': string (e.g: '000000123') 

98 - 'date': date (e.g: 2013-06-26) 

99 -o 'balance_start': float (e.g: 8368.56) 

100 -o 'balance_end_real': float (e.g: 8888.88) 

101 - 'transactions': list of dict containing : 

102 - 'name': string (e.g: 'KBC-INVESTERINGSKREDIET 787-5562831-01') 

103 - 'date': date 

104 - 'amount': float 

105 - 'unique_import_id': string 

106 -o 'account_number': string 

107 Will be used to find/create the res.partner.bank in odoo 

108 -o 'note': string 

109 -o 'partner_name': string 

110 -o 'ref': string 

111 """ 

112 raise UserError(_('Could not make sense of the given file.\nDid you install the module to support this type of file ?'))

114 def _check_parsed_data(self, stmts_vals): 

115 """ Basic and structural verifications """ 

116 if len(stmts_vals) == 0: 

117 raise UserError(_('This file doesn\'t contain any statement.')) 

118 

119 no_st_line = True 

120 for vals in stmts_vals: 

121 if vals['transactions'] and len(vals['transactions']) > 0: 

122 no_st_line = False 

123 break 

124 if no_st_line: 

125 raise UserError(_('This file doesn\'t contain any transaction.'))

130 def _find_additional_data(self, currency_code, account_number): 

131 """ Look for a res.currency and account.journal using values extracted from the 

132 statement and make sure it's consistent. 

133 """ 

134 company_currency = self.env.user.company_id.currency_id 

135 journal_obj = self.env['account.journal'] 

136 currency = None 

137 sanitized_account_number = sanitize_account_number(account_number) 

138 

139 if currency_code: 

140 currency = self.env['res.currency'].search([('name', '=ilike', currency_code)], limit=1) 

141 if not currency: 

142 raise UserError(_("No currency found matching '%s'.") % currency_code) 

143 if currency == company_currency: 

144 currency = False 

145 

146 journal = journal_obj.browse(self.env.context.get('journal_id', [])) 

147 if account_number: 

148 # No bank account on the journal : create one from the account number of the statement 

149 if journal and not journal.bank_account_id: 

150 journal.set_bank_account(account_number) 

151 # No journal passed to the wizard : try to find one using the account number of the statement 

152 elif not journal: 

153 journal = journal_obj.search([('bank_account_id.sanitized_acc_number', '=', sanitized_account_number)]) 

154 # Already a bank account on the journal : check it's the same as on the statement 

155 else: 

156 if not self._check_journal_bank_account(journal, sanitized_account_number): 

157 raise UserError(_('The account of this statement (%s) is not the same as the journal (%s).') % (account_number, journal.bank_account_id.acc_number)) 

158 

159 # If importing into an existing journal, its currency must be the same as the bank statement 

160 if journal: 

161 journal_currency = journal.currency_id 

162 if currency is None: 

163 currency = journal_currency 

164 if currency and currency != journal_currency: 

165 statement_cur_code = not currency and company_currency.name or currency.name 

166 journal_cur_code = not journal_currency and company_currency.name or journal_currency.name 

167 raise UserError(_('The currency of the bank statement (%s) is not the same as the currency of the journal (%s) !') % (statement_cur_code, journal_cur_code)) 

168 

169 # If we couldn't find / can't create a journal, everything is lost 

170 if not journal and not account_number: 

171 raise UserError(_('Cannot find in which journal import this statement. Please manually select a journal.')) 

172 

173 return currency, journal

175 def _complete_stmts_vals(self, stmts_vals, journal, account_number): 

176 for st_vals in stmts_vals: 

177 st_vals['journal_id'] = journal.id 

178 if not st_vals.get('reference'): 

179 st_vals['reference'] = self.filename 

180 if st_vals.get('number'): 

181 #build the full name like BNK/2016/00135 by just giving the number '135' 

182 st_vals['name'] = journal.sequence_id.with_context(ir_sequence_date=st_vals.get('date')).get_next_char(st_vals['number']) 

183 del(st_vals['number']) 

184 for line_vals in st_vals['transactions']: 

185 unique_import_id = line_vals.get('unique_import_id') 

186 if unique_import_id: 

187 sanitized_account_number = sanitize_account_number(account_number) 

188 line_vals['unique_import_id'] = (sanitized_account_number and sanitized_account_number + '-' or '') + str(journal.id) + '-' + unique_import_id 

189 

190 if not line_vals.get('bank_account_id'): 

191 # Find the partner and his bank account or create the bank account. The partner selected during the 

192 # reconciliation process will be linked to the bank when the statement is closed. 

193 partner_id = False 

194 bank_account_id = False 

195 identifying_string = line_vals.get('account_number') 

196 if identifying_string: 

197 partner_bank = self.env['res.partner.bank'].search([('acc_number', '=', identifying_string)], limit=1) 

198 if partner_bank: 

199 bank_account_id = partner_bank.id 

200 partner_id = partner_bank.partner_id.id 

201 else: 

202 bank_account_id = self.env['res.partner.bank'].create({'acc_number': line_vals['account_number']}).id 

203 line_vals['partner_id'] = partner_id 

204 line_vals['bank_account_id'] = bank_account_id 

205 

206 return stmts_vals

208 def _create_bank_statements(self, stmts_vals): 

209 """ Create new bank statements from imported values, filtering out already imported transactions, and returns data used by the reconciliation widget """ 

210 BankStatement = self.env['account.bank.statement'] 

211 BankStatementLine = self.env['account.bank.statement.line'] 

212 

213 # Filter out already imported transactions and create statements 

214 statement_ids = [] 

215 ignored_statement_lines_import_ids = [] 

216 for st_vals in stmts_vals: 

217 filtered_st_lines = [] 

218 for line_vals in st_vals['transactions']: 

219 if 'unique_import_id' not in line_vals \ 

220 or not line_vals['unique_import_id'] \ 

221 or not bool(BankStatementLine.sudo().search([('unique_import_id', '=', line_vals['unique_import_id'])], limit=1)): 

222 if line_vals['amount'] != 0: 

223 # Some banks, like ING, create a line for free charges. 

224 # We just skip those lines as there's a 'non-zero' constraint 

225 # on the amount of account.bank.statement.line 

226 filtered_st_lines.append(line_vals) 

227 else: 

228 ignored_statement_lines_import_ids.append(line_vals['unique_import_id']) 

229 if 'balance_start' in st_vals: 

230 st_vals['balance_start'] += float(line_vals['amount']) 

231 

232 if len(filtered_st_lines) > 0: 

233 # Remove values that won't be used to create records 

234 st_vals.pop('transactions', None) 

235 for line_vals in filtered_st_lines: 

236 line_vals.pop('account_number', None) 

237 # Create the satement 

238 st_vals['line_ids'] = [[0, False, line] for line in filtered_st_lines] 

239 statement_ids.append(BankStatement.create(st_vals).id) 

240 if len(statement_ids) == 0: 

241 raise UserError(_('You have already imported that file.')) 

242 

243 # Prepare import feedback 

244 notifications = [] 

245 num_ignored = len(ignored_statement_lines_import_ids) 

246 if num_ignored > 0: 

247 notifications += [{ 

248 'type': 'warning', 

249 'message': _("%d transactions had already been imported and were ignored.") % num_ignored if num_ignored > 1 else _("1 transaction had already been imported and was ignored."), 

250 'details': { 

251 'name': _('Already imported items'), 

252 'model': 'account.bank.statement.line', 

253 'ids': BankStatementLine.search([('unique_import_id', 'in', ignored_statement_lines_import_ids)]).ids 

254 } 

255 }] 

256 return statement_ids, notifications
