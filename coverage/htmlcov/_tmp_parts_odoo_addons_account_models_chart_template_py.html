22def migrate_tags_on_taxes(cr, registry): 

23 ''' This is a utiliy function to help migrate the tags of taxes when the localization has been modified on stable version. If 

24 called accordingly in a post_init_hooked function, it will reset the tags set on taxes as per their equivalent template. 

25 

26 Note: This unusual decision has been made in order to help the improvement of VAT reports on version 9.0, to have them more flexible 

27 and working out of the box when people are creating/using new taxes. 

28 ''' 

29 env = api.Environment(cr, SUPERUSER_ID, {}) 

30 xml_records = env['ir.model.data'].search([ 

31 ('model', '=', 'account.tax.template'), 

32 ('module', 'like', 'l10n_%') 

33 ]) 

34 tax_template_ids = [x['res_id'] for x in xml_records.sudo().read(['res_id'])] 

35 for tax_template in env['account.tax.template'].browse(tax_template_ids): 

36 tax_id = env['account.tax'].search([ 

37 ('name', '=', tax_template.name), 

38 ('type_tax_use', '=', tax_template.type_tax_use), 

39 ('description', '=', tax_template.description) 

40 ]) 

41 tax_id.sudo().write({'tag_ids': [(6, 0, tax_template.tag_ids.ids)]})

292 @api.multi 

293 def _load_template(self, company, code_digits=None, transfer_account_id=None, account_ref=None, taxes_ref=None): 

294 """ Generate all the objects from the templates 

295 

296 :param company: company the wizard is running for 

297 :param code_digits: number of digits the accounts code should have in the COA 

298 :param transfer_account_id: reference to the account template that will be used as intermediary account for transfers between 2 liquidity accounts 

299 :param acc_ref: Mapping between ids of account templates and real accounts created from them 

300 :param taxes_ref: Mapping between ids of tax templates and real taxes created from them 

301 :returns: tuple with a dictionary containing 

302 * the mapping between the account template ids and the ids of the real accounts that have been generated 

303 from them, as first item, 

304 * a similar dictionary for mapping the tax templates and taxes, as second item, 

305 :rtype: tuple(dict, dict, dict) 

306 """ 

307 self.ensure_one() 

308 if account_ref is None: 

309 account_ref = {} 

310 if taxes_ref is None: 

311 taxes_ref = {} 

312 if not code_digits: 

313 code_digits = self.code_digits 

314 if not transfer_account_id: 

315 transfer_account_id = self.transfer_account_id 

316 AccountTaxObj = self.env['account.tax'] 

317 

318 # Generate taxes from templates. 

319 generated_tax_res = self.tax_template_ids._generate_tax(company) 

320 taxes_ref.update(generated_tax_res['tax_template_to_tax']) 

321 

322 # Generating Accounts from templates. 

323 account_template_ref = self.generate_account(taxes_ref, account_ref, code_digits, company) 

324 account_ref.update(account_template_ref) 

325 

326 # writing account values after creation of accounts 

327 company.transfer_account_id = account_template_ref[transfer_account_id.id] 

328 for key, value in generated_tax_res['account_dict'].items(): 

329 if value['refund_account_id'] or value['account_id']: 

330 AccountTaxObj.browse(key).write({ 

331 'refund_account_id': account_ref.get(value['refund_account_id'], False), 

332 'account_id': account_ref.get(value['account_id'], False), 

333 }) 

334 

335 # Create Journals - Only done for root chart template 

336 if not self.parent_id: 

337 self.generate_journals(account_ref, company) 

338 

339 # generate properties function 

340 self.generate_properties(account_ref, company) 

341 

342 # Generate Fiscal Position , Fiscal Position Accounts and Fiscal Position Taxes from templates 

343 self.generate_fiscal_position(taxes_ref, account_ref, company) 

344 

345 # Generate account operation template templates 

346 self.generate_account_reconcile_model(taxes_ref, account_ref, company) 

347 

348 return account_ref, taxes_ref

381 def generate_account(self, tax_template_ref, acc_template_ref, code_digits, company): 

382 """ This method for generating accounts from templates. 

383 

384 :param tax_template_ref: Taxes templates reference for write taxes_id in account_account. 

385 :param acc_template_ref: dictionary with the mappping between the account templates and the real accounts. 

386 :param code_digits: number of digits got from wizard.multi.charts.accounts, this is use for account code. 

387 :param company_id: company_id selected from wizard.multi.charts.accounts.

396 code_acc = account_template.code or ''

400 new_account = self.create_record_with_xmlid(company, account_template, 'account.account', vals) 

401 acc_template_ref[account_template.id] = new_account
