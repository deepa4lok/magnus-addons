45 def interval_clean(self, intervals): 

46 """ Utility method that sorts and removes overlapping inside datetime 

47 intervals. The intervals are sorted based on increasing starting datetime. 

48 Overlapping intervals are merged into a single one. 

49 

50 :param list intervals: list of intervals; each interval is a tuple 

51 (datetime_from, datetime_to) 

52 :return list cleaned: list of sorted intervals without overlap """ 

53 intervals = sorted(intervals, key=itemgetter(0)) # sort on first datetime 

54 cleaned = [] 

55 working_interval = None 

56 while intervals: 

57 current_interval = intervals.pop(0) 

58 if not working_interval: # init 

59 working_interval = [current_interval[0], current_interval[1]] 

60 elif working_interval[1] < current_interval[0]: # interval is disjoint 

61 cleaned.append(tuple(working_interval)) 

62 working_interval = [current_interval[0], current_interval[1]] 

63 elif working_interval[1] < current_interval[1]: # union of greater intervals 

64 working_interval[1] = current_interval[1] 

65 if working_interval: # handle void lists 

66 cleaned.append(tuple(working_interval)) 

67 return cleaned

69 @api.model 

70 def interval_remove_leaves(self, interval, leave_intervals): 

71 """ Utility method that remove leave intervals from a base interval: 

72 

73 - clean the leave intervals, to have an ordered list of not-overlapping 

74 intervals 

75 - initiate the current interval to be the base interval 

76 - for each leave interval: 

77 

78 - finishing before the current interval: skip, go to next 

79 - beginning after the current interval: skip and get out of the loop 

80 because we are outside range (leaves are ordered) 

81 - beginning within the current interval: close the current interval 

82 and begin a new current interval that begins at the end of the leave 

83 interval 

84 - ending within the current interval: update the current interval begin 

85 to match the leave interval ending 

86 

87 :param tuple interval: a tuple (beginning datetime, ending datetime) that 

88 is the base interval from which the leave intervals 

89 will be removed 

90 :param list leave_intervals: a list of tuples (beginning datetime, ending datetime) 

91 that are intervals to remove from the base interval 

92 :return list intervals: a list of tuples (begin datetime, end datetime) 

93 that are the remaining valid intervals """ 

94 if not interval: 

95 return interval 

96 if leave_intervals is None: 

97 leave_intervals = [] 

98 intervals = [] 

99 leave_intervals = self.interval_clean(leave_intervals) 

100 current_interval = [interval[0], interval[1]] 

101 for leave in leave_intervals: 

102 if leave[1] <= current_interval[0]: 

103 continue 

104 if leave[0] >= current_interval[1]: 

105 break 

106 if current_interval[0] < leave[0] < current_interval[1]: 

107 current_interval[1] = leave[0] 

108 intervals.append((current_interval[0], current_interval[1])) 

109 current_interval = [leave[1], interval[1]] 

110 if current_interval[0] <= leave[1]: 

111 current_interval[0] = leave[1] 

112 if current_interval and current_interval[0] < interval[1]: # remove intervals moved outside base interval due to leaves 

113 intervals.append((current_interval[0], current_interval[1])) 

114 return intervals

116 def interval_schedule_hours(self, intervals, hour, remove_at_end=True): 

117 """ Schedule hours in intervals. The last matching interval is truncated 

118 to match the specified hours. 

119 

120 It is possible to truncate the last interval at its beginning or ending. 

121 However this does nothing on the given interval order that should be 

122 submitted accordingly. 

123 

124 :param list intervals: a list of tuples (beginning datetime, ending datetime) 

125 :param int/float hours: number of hours to schedule. It will be converted 

126 into a timedelta, but should be submitted as an 

127 int or float. 

128 :param boolean remove_at_end: remove extra hours at the end of the last 

129 matching interval. Otherwise, do it at the 

130 beginning. 

131 

132 :return list results: a list of intervals. If the number of hours to schedule 

133 is greater than the possible scheduling in the intervals, no extra-scheduling 

134 is done, and results == intervals. """ 

135 results = [] 

136 res = timedelta() 

137 limit = timedelta(hours=hour) 

138 for interval in intervals: 

139 res += interval[1] - interval[0] 

140 if res > limit and remove_at_end: 

141 interval = (interval[0], interval[1] + relativedelta(seconds=seconds(limit - res))) 

142 elif res > limit: 

143 interval = (interval[0] + relativedelta(seconds=seconds(res - limit)), interval[1]) 

144 results.append(interval) 

145 if res > limit: 

146 break 

147 return results

381 def get_working_hours(self, start_dt, end_dt, compute_leaves=False, 

382 resource_id=None, default_interval=None): 

383 hours = 0.0 

384 for day in rrule.rrule(rrule.DAILY, dtstart=start_dt, 

385 until=end_dt.replace(hour=23, minute=59, second=59, microsecond=999999), 

386 byweekday=self.get_weekdays()): 

387 day_start_dt = day.replace(hour=0, minute=0, second=0, microsecond=0) 

388 if start_dt and day.date() == start_dt.date():

390 day_end_dt = day.replace(hour=23, minute=59, second=59, microsecond=999999) 

391 if end_dt and day.date() == end_dt.date(): 

392 day_end_dt = end_dt 

393 hours += self.get_working_hours_of_date( 

394 start_dt=day_start_dt, end_dt=day_end_dt, 

395 compute_leaves=compute_leaves, resource_id=resource_id, 

396 default_interval=default_interval) 

397 return hours

403 @api.multi 

404 def _schedule_hours(self, hours, day_dt=None, 

405 compute_leaves=False, resource_id=None, 

406 default_interval=None): 

407 """ Schedule hours of work, using a calendar and an optional resource to 

408 compute working and leave days. This method can be used backwards, i.e. 

409 scheduling days before a deadline. 

410 

411 :param int hours: number of hours to schedule. Use a negative number to 

412 compute a backwards scheduling. 

413 :param datetime day_dt: reference date to compute working days. If days is 

414 > 0 date is the starting date. If days is < 0 

415 date is the ending date. 

416 :param boolean compute_leaves: if set, compute the leaves based on calendar 

417 and resource. Otherwise no leaves are taken 

418 into account. 

419 :param int resource_id: the id of the resource to take into account when 

420 computing the leaves. If not set, only general 

421 leaves are computed. If set, generic and 

422 specific leaves are computed. 

423 :param tuple default_interval: if no id, try to return a default working 

424 day using default_interval[0] as beginning 

425 hour, and default_interval[1] as ending hour. 

426 Example: default_interval = (8, 16). 

427 Otherwise, a void list of working intervals 

428 is returned when id is None. 

429 

430 :return tuple (datetime, intervals): datetime is the beginning/ending date 

431 of the schedulign; intervals are the 

432 working intervals of the scheduling. 

433 

434 Note: Why not using rrule.rrule ? Because rrule does not seem to allow 

435 getting back in time. 

436 """ 

437 if day_dt is None: 

438 day_dt = datetime.datetime.now() 

439 elif day_dt is not None and hours > 0: 

440 # FORWARD-PORT UP TO SAAS-14 

441 # FIXME: think about a better fix... However, solved from saas-15 thanks to refactoring 

442 # Make sure the to keep the original start date when searching for intervals, since a 

443 # difference in day may occur depending on the TZ. 

444 self = self.with_context(force_start_dt=day_dt) 

445 backwards = (hours < 0) 

446 hours = abs(hours) 

447 intervals = [] 

448 remaining_hours = hours * 1.0 

449 iterations = 0 

450 current_datetime = day_dt 

451 

452 call_args = dict(compute_leaves=compute_leaves, resource_id=resource_id, default_interval=default_interval) 

453 

454 while float_compare(remaining_hours, 0.0, precision_digits=2) in (1, 0) and iterations < 1000: 

455 if backwards: 

456 call_args['end_dt'] = current_datetime 

457 else: 

458 call_args['start_dt'] = current_datetime 

459 

460 working_intervals = self.get_working_intervals_of_day(**call_args) 

461 

462 if not self and not working_intervals: # no calendar -> consider working 8 hours 

463 remaining_hours -= 8.0 

464 elif working_intervals: 

465 if backwards: 

466 working_intervals.reverse() 

467 new_working_intervals = self.interval_schedule_hours(working_intervals, remaining_hours, not backwards) 

468 if backwards: 

469 new_working_intervals.reverse() 

470 

471 res = timedelta() 

472 for interval in working_intervals: 

473 res += interval[1] - interval[0] 

474 remaining_hours -= (seconds(res) / 3600.0) 

475 if backwards: 

476 intervals = new_working_intervals + intervals 

477 else: 

478 intervals = intervals + new_working_intervals 

479 # get next day 

480 if backwards: 

481 current_datetime = datetime.datetime.combine(self.get_previous_day(current_datetime), datetime.time(23, 59, 59)) 

482 else: 

483 current_datetime = datetime.datetime.combine(self.get_next_day(current_datetime), datetime.time()) 

484 # avoid infinite loops 

485 iterations += 1 

486 

487 return intervals

521 :param int days: number of days to schedule. Use a negative number to

524 date is the starting date. If days is < 0 date is the 

525 ending date. 

526 :param boolean compute_leaves: if set, compute the leaves based on calendar 

527 and resource. Otherwise no leaves are taken 

528 into account. 

529 :param int resource_id: the id of the resource to take into account when 

530 computing the leaves. If not set, only general 

531 leaves are computed. If set, generic and 

532 specific leaves are computed. 

533 :param tuple default_interval: if no id, try to return a default working 

534 day using default_interval[0] as beginning 

535 hour, and default_interval[1] as ending hour. 

536 Example: default_interval = (8, 16). 

537 Otherwise, a void list of working intervals 

538 is returned when id is None. 

539 

540 :return tuple (datetime, intervals): datetime is the beginning/ending date 

541 of the schedulign; intervals are the 

542 working intervals of the scheduling. 

543 

544 Implementation note: rrule.rrule is not used because rrule it des not seem 

545 to allow getting back in time. 

546 """ 

547 if day_date is None: 

548 day_date = datetime.datetime.now() 

549 backwards = (days < 0) 

550 days = abs(days) 

551 intervals = [] 

552 planned_days = 0 

553 iterations = 0 

554 current_datetime = day_date.replace(hour=0, minute=0, second=0, microsecond=0) 

555

558 current_datetime,

564 # get next day

566 current_datetime = self.get_previous_day(current_datetime) 

567 else: 

568 current_datetime = self.get_next_day(current_datetime) 

569 # avoid infinite loops 

570 iterations += 1 

571 

572 return intervals

725 def _is_work_day(self, date): 

726 """ Whether the provided date is a work day for the subject resource. 

727 

728 :type date: datetime.date 

729 :rtype: bool 

730 """ 

731 return bool(next(self._iter_work_days(date, date), False))

733 def _iter_work_days(self, from_date, to_date): 

734 """ Lists the current resource's work days between the two provided 

735 dates (inclusive). 

736 

737 Work days are the company or service's open days (as defined by the 

738 resource.calendar) minus the resource's own leaves. 

739 

740 :param datetime.date from_date: start of the interval to check for 

741 work days (inclusive) 

742 :param datetime.date to_date: end of the interval to check for work 

743 days (inclusive) 

744 :rtype: list(datetime.date) 

745 """ 

746 working_intervals = self.calendar_id.get_working_intervals_of_day 

747 # rrule coerces date inputs to datetimes (with time=0) and yields 

748 # datetimes (with time=0 if freq >= daily) 

749 for dt in rrule.rrule(rrule.DAILY, dtstart=from_date, until=to_date): 

750 intervals = working_intervals(dt, compute_leaves=True, resource_id=self.id) 

751 

752 # FIXME: get_working_intervals is new-API mapped to return a list of lists of intervals 

753 if intervals and intervals[0]: 

754 yield dt.date()
