28def LocalService(name): 

29 """ 

30 The odoo.netsvc.LocalService() function is deprecated. It still works 

31 in two cases: workflows and reports. For workflows, instead of using 

32 LocalService('workflow'), odoo.workflow should be used (better yet, 

33 methods on odoo.osv.orm.Model should be used). For reports, 

34 odoo.report.render_report() should be used (methods on the Model should 

35 be provided too in the future). 

36 """ 

37 assert odoo.conf.deprecation.allow_local_service 

38 _logger.warning("LocalService() is deprecated since march 2013 (it was called with '%s')." % name)

40 if name == 'workflow': 

41 return odoo.workflow

43 if name.startswith('report.'):

46 return report

56class PostgreSQLHandler(logging.Handler): 

57 """ PostgreSQL Loggin Handler will store logs in the database, by default

60 def emit(self, record): 

61 ct = threading.current_thread() 

62 ct_db = getattr(ct, 'dbname', None) 

63 dbname = tools.config['log_db'] if tools.config['log_db'] and tools.config['log_db'] != '%d' else ct_db 

64 if not dbname: 

65 return 

66 with tools.ignore(Exception), tools.mute_logger('odoo.sql_db'), sql_db.db_connect(dbname, allow_uri=True).cursor() as cr: 

67 cr.autocommit(True) 

68 msg = tools.ustr(record.msg) 

69 if record.args: 

70 msg = msg % record.args 

71 traceback = getattr(record, 'exc_text', '') 

72 if traceback: 

73 msg = "%s\n%s" % (msg, traceback) 

74 # we do not use record.levelname because it may have been changed by ColoredFormatter. 

75 levelname = logging.getLevelName(record.levelno) 

76 

77 val = ('server', ct_db, record.name, levelname, msg, record.pathname[len(path_prefix)+1:], record.lineno, record.funcName) 

78 cr.execute(""" 

79 INSERT INTO ir_logging(create_date, type, dbname, name, level, message, path, line, func) 

80 VALUES (NOW() at time zone 'UTC', %s, %s, %s, %s, %s, %s, %s, %s) 

81 """, val)

83BLACK, RED, GREEN, YELLOW, BLUE, MAGENTA, CYAN, WHITE, _NOTHING, DEFAULT = range(10) 

84#The background is set with 40 plus the number of the color, and the foreground with 30 

85#These are the sequences need to get colored ouput 

86RESET_SEQ = "\033[0m" 

87COLOR_SEQ = "\033[1;%dm" 

88BOLD_SEQ = "\033[1m" 

89COLOR_PATTERN = "%s%s%%s%s" % (COLOR_SEQ, COLOR_SEQ, RESET_SEQ) 

90LEVEL_COLOR_MAPPING = { 

91 logging.DEBUG: (BLUE, DEFAULT), 

92 logging.INFO: (GREEN, DEFAULT), 

93 logging.WARNING: (YELLOW, DEFAULT), 

94 logging.ERROR: (RED, DEFAULT), 

95 logging.CRITICAL: (WHITE, RED), 

96}

98class DBFormatter(logging.Formatter): 

99 def format(self, record): 

100 record.pid = os.getpid() 

101 record.dbname = getattr(threading.currentThread(), 'dbname', '?') 

102 return logging.Formatter.format(self, record) 

103 

104class ColoredFormatter(DBFormatter): 

105 def format(self, record): 

106 fg_color, bg_color = LEVEL_COLOR_MAPPING.get(record.levelno, (GREEN, DEFAULT)) 

107 record.levelname = COLOR_PATTERN % (30 + fg_color, 40 + bg_color, record.levelname) 

108 return DBFormatter.format(self, record)
