69 #------------------------------------------------------ 

70 @http.route([ 

71 '/web_editor/font_to_img/<icon>', 

72 '/web_editor/font_to_img/<icon>/<color>', 

73 '/web_editor/font_to_img/<icon>/<color>/<int:size>', 

74 '/web_editor/font_to_img/<icon>/<color>/<int:size>/<int:alpha>', 

75 ], type='http', auth="none") 

76 def export_icon_to_png(self, icon, color='#000', size=100, alpha=255, font='/web/static/lib/fontawesome/fonts/fontawesome-webfont.ttf'): 

77 """ This method converts an unicode character to an image (using Font 

78 Awesome font by default) and is used only for mass mailing because 

79 custom fonts are not supported in mail. 

80 :param icon : decimal encoding of unicode character 

81 :param color : RGB code of the color 

82 :param size : Pixels in integer 

83 :param alpha : transparency of the image from 0 to 255 

84 :param font : font path 

85 

86 :returns PNG image converted from given font 

87 """ 

88 # Make sure we have at least size=1 

89 size = max(1, size) 

90 # Initialize font 

91 addons_path = http.addons_manifest['web']['addons_path'] 

92 font_obj = ImageFont.truetype(addons_path + font, size) 

93 

94 # if received character is not a number, keep old behaviour (icon is character) 

95 icon = unichr(int(icon)) if icon.isdigit() else icon 

96 

97 # Determine the dimensions of the icon 

98 image = Image.new("RGBA", (size, size), color=(0, 0, 0, 0)) 

99 draw = ImageDraw.Draw(image) 

100 

101 boxw, boxh = draw.textsize(icon, font=font_obj) 

102 draw.text((0, 0), icon, font=font_obj) 

103 left, top, right, bottom = image.getbbox() 

104 

105 # Create an alpha mask 

106 imagemask = Image.new("L", (boxw, boxh), 0) 

107 drawmask = ImageDraw.Draw(imagemask) 

108 drawmask.text((-left, -top), icon, font=font_obj, fill=alpha) 

109 

110 # Create a solid color image and apply the mask 

111 if color.startswith('rgba'): 

112 color = color.replace('rgba', 'rgb') 

113 color = ','.join(color.split(',')[:-1])+')' 

114 iconimage = Image.new("RGBA", (boxw, boxh), color) 

115 iconimage.putalpha(imagemask) 

116 

117 # Create output image 

118 outimage = Image.new("RGBA", (boxw, size), (0, 0, 0, 0)) 

119 outimage.paste(iconimage, (left, top)) 

120 

121 # output image 

122 output = io.BytesIO() 

123 outimage.save(output, format="PNG") 

124 response = werkzeug.wrappers.Response() 

125 response.mimetype = 'image/png' 

126 response.data = output.getvalue() 

127 response.headers['Cache-Control'] = 'public, max-age=604800' 

128 response.headers['Access-Control-Allow-Origin'] = '*' 

129 response.headers['Access-Control-Allow-Methods'] = 'GET, POST' 

130 response.headers['Connection'] = 'close' 

131 response.headers['Date'] = time.strftime("%a, %d-%b-%Y %T GMT", time.gmtime()) 

132 response.headers['Expires'] = time.strftime("%a, %d-%b-%Y %T GMT", time.gmtime(time.time()+604800*60)) 

133 

134 return response

139 @http.route('/web_editor/attachment/add', type='http', auth='user', methods=['POST']) 

140 def attach(self, func, upload=None, url=None, disable_optimization=None, **kwargs): 

141 # the upload argument doesn't allow us to access the files if more than 

142 # one file is uploaded, as upload references the first file 

143 # therefore we have to recover the files from the request object 

144 Attachments = request.env['ir.attachment'] # registry for the attachment table 

145 

146 uploads = [] 

147 message = None 

148 if not upload: # no image provided, storing the link and the image name 

149 name = url.split("/").pop() # recover filename 

150 attachment = Attachments.create({ 

151 'name': name, 

152 'type': 'url', 

153 'url': url, 

154 'public': True, 

155 'res_model': 'ir.ui.view', 

156 }) 

157 uploads += attachment.read(['name', 'mimetype', 'checksum', 'url']) 

158 else: # images provided 

159 try: 

160 attachments = request.env['ir.attachment'] 

161 for c_file in request.httprequest.files.getlist('upload'): 

162 data = c_file.read() 

163 try: 

164 image = Image.open(cStringIO.StringIO(data)) 

165 w, h = image.size 

166 if w*h > 42e6: # Nokia Lumia 1020 photo resolution 

167 raise ValueError( 

168 u"Image size excessive, uploaded images must be smaller " 

169 u"than 42 million pixel") 

170 if not disable_optimization and image.format in ('PNG', 'JPEG'): 

171 data = tools.image_save_for_web(image) 

172 except IOError, e: 

173 pass 

174 

175 attachment = Attachments.create({ 

176 'name': c_file.filename, 

177 'datas': data.encode('base64'), 

178 'datas_fname': c_file.filename, 

179 'public': True, 

180 'res_model': 'ir.ui.view', 

181 }) 

182 attachments += attachment 

183 uploads += attachments.read(['name', 'mimetype', 'checksum', 'url']) 

184 except Exception, e: 

185 logger.exception("Failed to upload image to attachment") 

186 message = unicode(e) 

187 

188 return """<script type='text/javascript'> 

189 window.parent['%s'](%s, %s); 

190 </script>""" % (func, json.dumps(uploads), json.dumps(message))

195 @http.route('/web_editor/attachment/remove', type='json', auth='user') 

196 def remove(self, ids, **kwargs): 

197 """ Removes a web-based image attachment if it is used by no view (template) 

198 

199 Returns a dict mapping attachments which would not be removed (if any) 

200 mapped to the views preventing their removal 

201 """ 

202 Attachment = attachments_to_remove = request.env['ir.attachment'] 

203 Views = request.env['ir.ui.view'] 

204 

205 # views blocking removal of the attachment 

206 removal_blocked_by = {} 

207 

208 for attachment in Attachment.browse(ids): 

209 # in-document URLs are html-escaped, a straight search will not 

210 # find them 

211 url = tools.html_escape(attachment.local_url) 

212 views = Views.search([ 

213 "|", 

214 ('arch_db', 'like', '"%s"' % url), 

215 ('arch_db', 'like', "'%s'" % url) 

216 ]) 

217 

218 if views: 

219 removal_blocked_by[attachment.id] = views.read(['name']) 

220 else: 

221 attachments_to_remove += attachment 

222 if attachments_to_remove: 

223 attachments_to_remove.unlink() 

224 return removal_blocked_by
