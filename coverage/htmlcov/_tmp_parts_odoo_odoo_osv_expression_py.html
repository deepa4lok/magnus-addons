4""" Domain expression processing 

5 

6The main duty of this module is to compile a domain expression into a 

7SQL query. A lot of things should be documented here, but as a first 

8step in the right direction, some tests in test_expression.py 

9might give you some additional information. 

10 

11For legacy reasons, a domain uses an inconsistent two-levels abstract 

12syntax (domains are regular Python data structures). At the first 

13level, a domain is an expression made of terms (sometimes called 

14leaves) and (domain) operators used in prefix notation. The available 

15operators at this level are '!', '&', and '|'. '!' is a unary 'not', 

16'&' is a binary 'and', and '|' is a binary 'or'. For instance, here 

17is a possible domain. (<term> stands for an arbitrary term, more on 

18this later.):: 

19 

20 ['&', '!', <term1>, '|', <term2>, <term3>] 

21 

22It is equivalent to this pseudo code using infix notation:: 

23 

24 (not <term1>) and (<term2> or <term3>) 

25 

26The second level of syntax deals with the term representation. A term 

27is a triple of the form (left, operator, right). That is, a term uses 

28an infix notation, and the available operators, and possible left and 

29right operands differ with those of the previous level. Here is a 

30possible term:: 

31 

32 ('company_id.name', '=', 'OpenERP') 

33 

34The left and right operand don't have the same possible values. The 

35left operand is field name (related to the model for which the domain 

36applies). Actually, the field name can use the dot-notation to 

37traverse relationships. The right operand is a Python value whose 

38type should match the used operator and field type. In the above 

39example, a string is used because the name field of a company has type 

40string, and because we use the '=' operator. When appropriate, a 'in' 

41operator can be used, and thus the right operand should be a list. 

42 

43Note: the non-uniform syntax could have been more uniform, but this 

44would hide an important limitation of the domain syntax. Say that the 

45term representation was ['=', 'company_id.name', 'OpenERP']. Used in a 

46complete domain, this would look like:: 

47 

48 ['!', ['=', 'company_id.name', 'OpenERP']] 

49 

50and you would be tempted to believe something like this would be 

51possible:: 

52 

53 ['!', ['=', 'company_id.name', ['&', ..., ...]]] 

54 

55That is, a domain could be a valid operand. But this is not the 

56case. A domain is really limited to a two-level nature, and can not 

57take a recursive form: a domain is not a valid second-level operand. 

58

63Normally the 'unaccent' function is obtained from `the PostgreSQL 

64'unaccent' contrib module 

65<http://developer.postgresql.org/pgdocs/postgres/unaccent.html>`_.

128# Domain operators. 

129NOT_OPERATOR = '!' 

130OR_OPERATOR = '|' 

131AND_OPERATOR = '&' 

132DOMAIN_OPERATORS = (NOT_OPERATOR, OR_OPERATOR, AND_OPERATOR) 

133 

134# List of available term operators. It is also possible to use the '<>' 

135# operator, which is strictly the same as '!='; the later should be prefered 

136# for consistency. This list doesn't contain '<>' as it is simpified to '!=' 

137# by the normalize_operator() function (so later part of the code deals with 

138# only one representation). 

139# Internals (i.e. not available to the user) 'inselect' and 'not inselect' 

140# operators are also used. In this case its right operand has the form (subselect, params). 

141TERM_OPERATORS = ('=', '!=', '<=', '<', '>', '>=', '=?', '=like', '=ilike', 

142 'like', 'not like', 'ilike', 'not ilike', 'in', 'not in', 

143 'child_of', 'parent_of') 

144 

145# A subset of the above operators, with a 'negative' semantic. When the 

146# expressions 'in NEGATIVE_TERM_OPERATORS' or 'not in NEGATIVE_TERM_OPERATORS' are used in the code 

147# below, this doesn't necessarily mean that any of those NEGATIVE_TERM_OPERATORS is 

148# legal in the processed term. 

149NEGATIVE_TERM_OPERATORS = ('!=', 'not like', 'not ilike', 'not in') 

150 

151# Negation of domain expressions 

152DOMAIN_OPERATORS_NEGATION = { 

153 AND_OPERATOR: OR_OPERATOR, 

154 OR_OPERATOR: AND_OPERATOR, 

155} 

156TERM_OPERATORS_NEGATION = { 

157 '<': '>=', 

158 '>': '<=', 

159 '<=': '>', 

160 '>=': '<', 

161 '=': '!=', 

162 '!=': '=', 

163 'in': 'not in', 

164 'like': 'not like', 

165 'ilike': 'not ilike', 

166 'not in': 'in', 

167 'not like': 'like', 

168 'not ilike': 'ilike', 

169} 

170 

171TRUE_LEAF = (1, '=', 1) 

172FALSE_LEAF = (0, '=', 1) 

173 

174TRUE_DOMAIN = [TRUE_LEAF] 

175FALSE_DOMAIN = [FALSE_LEAF]

184def normalize_domain(domain): 

185 """Returns a normalized version of ``domain_expr``, where all implicit '&' operators 

186 have been made explicit. One property of normalized domain expressions is that they 

187 can be easily combined together as if they were single domain components. 

188 """ 

189 assert isinstance(domain, (list, tuple)), "Domains to normalize must have a 'domain' form: a list or tuple of domain components" 

190 if not domain: 

191 return TRUE_DOMAIN 

192 result = [] 

193 expected = 1 # expected number of expressions 

194 op_arity = {NOT_OPERATOR: 1, AND_OPERATOR: 2, OR_OPERATOR: 2} 

195 for token in domain: 

196 if expected == 0: # more than expected, like in [A, B] 

197 result[0:0] = [AND_OPERATOR] # put an extra '&' in front 

198 expected = 1 

199 result.append(token) 

200 if isinstance(token, (list, tuple)): # domain term 

201 expected -= 1 

202 else: 

203 expected += op_arity.get(token, 0) - 1 

204 assert expected == 0, 'This domain is syntactically not correct: %s' % (domain) 

205 return result

208def combine(operator, unit, zero, domains): 

209 """Returns a new domain expression where all domain components from ``domains`` 

210 have been added together using the binary operator ``operator``. The given 

211 domains must be normalized. 

212 

213 :param unit: the identity element of the domains "set" with regard to the operation 

214 performed by ``operator``, i.e the domain component ``i`` which, when 

215 combined with any domain ``x`` via ``operator``, yields ``x``. 

216 E.g. [(1,'=',1)] is the typical unit for AND_OPERATOR: adding it 

217 to any domain component gives the same domain. 

218 :param zero: the absorbing element of the domains "set" with regard to the operation 

219 performed by ``operator``, i.e the domain component ``z`` which, when 

220 combined with any domain ``x`` via ``operator``, yields ``z``. 

221 E.g. [(1,'=',1)] is the typical zero for OR_OPERATOR: as soon as 

222 you see it in a domain component the resulting domain is the zero. 

223 :param domains: a list of normalized domains. 

224 """ 

225 result = [] 

226 count = 0 

227 for domain in domains: 

228 if domain == unit: 

229 continue 

230 if domain == zero: 

231 return zero 

232 if domain: 

233 result += domain 

234 count += 1 

235 result = [operator] * (count - 1) + result 

236 return result

240 """AND([D1,D2,...]) returns a domain representing D1 and D2 and ... """

249def distribute_not(domain): 

250 """ Distribute any '!' domain operators found inside a normalized domain. 

251 

252 Because we don't use SQL semantic for processing a 'left not in right' 

253 query (i.e. our 'not in' is not simply translated to a SQL 'not in'), 

254 it means that a '! left in right' can not be simply processed 

255 by __leaf_to_sql by first emitting code for 'left in right' then wrapping 

256 the result with 'not (...)', as it would result in a 'not in' at the SQL 

257 level. 

258 

259 This function is thus responsible for pushing any '!' domain operators 

260 inside the terms themselves. For example:: 

261 

262 ['!','&',('user_id','=',4),('partner_id','in',[1,2])] 

263 will be turned into: 

264 ['|',('user_id','!=',4),('partner_id','not in',[1,2])] 

265 

266 """ 

267 

268 # This is an iterative version of a recursive function that split domain 

269 # into subdomains, processes them and combine the results. The "stack" below 

270 # represents the recursive calls to be done. 

271 result = [] 

272 stack = [False] 

273 

274 for token in domain: 

275 negate = stack.pop() 

276 # negate tells whether the subdomain starting with token must be negated 

277 if is_leaf(token): 

278 if negate: 

279 left, operator, right = token 

280 if operator in TERM_OPERATORS_NEGATION: 

281 result.append((left, TERM_OPERATORS_NEGATION[operator], right)) 

282 else: 

283 result.append(NOT_OPERATOR) 

284 result.append(token) 

285 else: 

286 result.append(token) 

287 elif token == NOT_OPERATOR: 

288 stack.append(not negate) 

289 elif token in DOMAIN_OPERATORS_NEGATION: 

290 result.append(DOMAIN_OPERATORS_NEGATION[token] if negate else token) 

291 stack.append(negate) 

292 stack.append(negate) 

293 else: 

294 result.append(token) 

295 

296 return result

309def generate_table_alias(src_table_alias, joined_tables=[]): 

310 """ Generate a standard table alias name. An alias is generated as following: 

311 - the base is the source table name (that can already be an alias) 

312 - then, each joined table is added in the alias using a 'link field name' 

313 that is used to render unique aliases for a given path 

314 - returns a tuple composed of the alias, and the full table alias to be 

315 added in a from condition with quoting done 

316 Examples: 

317 - src_table_alias='res_users', join_tables=[]: 

318 alias = ('res_users','"res_users"') 

319 - src_model='res_users', join_tables=[(res.partner, 'parent_id')] 

320 alias = ('res_users__parent_id', '"res_partner" as "res_users__parent_id"') 

321 

322 :param model src_table_alias: model source of the alias 

323 :param list joined_tables: list of tuples 

324 (dst_model, link_field) 

325 

326 :return tuple: (table_alias, alias statement for from clause with quotes added) 

327 """ 

328 alias = src_table_alias 

329 if not joined_tables: 

330 return '%s' % alias, '%s' % _quote(alias) 

331 for link in joined_tables: 

332 alias += '__' + link[1] 

333 alias = truncate_alias(alias) 

334 return '%s' % alias, '%s as %s' % (_quote(joined_tables[-1][0]), _quote(alias)) 

335 

336def truncate_alias(alias): 

337 # Use an alternate alias scheme if length exceeds the PostgreSQL limit 

338 # of 63 characters. 

339 if len(alias) >= 64: 

340 # We have to fit a crc32 hash and one underscore 

341 # into a 63 character alias. The remaining space we can use to add 

342 # a human readable prefix. 

343 alias_hash = hex(crc32(alias))[2:] 

344 ALIAS_PREFIX_LENGTH = 63 - len(alias_hash) - 1 

345 alias = "%s_%s" % ( 

346 alias[:ALIAS_PREFIX_LENGTH], alias_hash) 

347 return alias

361def add_join_inner_first(table_joins, join_tuple): 

362 """ Add `join_tuple` to `joins` unless the inner join version of 

363 `join_tuple` is already in joins. If `join_tuple` is an inner join and 

364 the outer join is already in `joins`, remove it first. """ 

365 inner_join = (join_tuple[0], join_tuple[1], join_tuple[2], 'JOIN') 

366 outer_join = (join_tuple[0], join_tuple[1], join_tuple[2], 'LEFT JOIN') 

367 if join_tuple == outer_join: 

368 if inner_join in table_joins: 

369 return False 

370 if join_tuple == inner_join and outer_join in table_joins: 

371 table_joins.remove(outer_join) 

372 if join_tuple not in table_joins: 

373 table_joins.append(join_tuple) 

374 return True 

375 return False

378def normalize_leaf(element): 

379 """ Change a term's operator to some canonical form, simplifying later 

380 processing. """ 

381 if not is_leaf(element): 

382 return element 

383 left, operator, right = element 

384 original = operator 

385 operator = operator.lower() 

386 if operator == '<>': 

387 operator = '!=' 

388 if isinstance(right, bool) and operator in ('in', 'not in'): 

389 _logger.warning("The domain term '%s' should use the '=' or '!=' operator." % ((left, original, right),)) 

390 operator = '=' if operator == 'in' else '!=' 

391 if isinstance(right, (list, tuple)) and operator in ('=', '!='): 

392 _logger.warning("The domain term '%s' should use the 'in' or 'not in' operator." % ((left, original, right),)) 

393 operator = 'in' if operator == '=' else 'not in' 

394 return left, operator, right

402def is_leaf(element, internal=False): 

403 """ Test whether an object is a valid domain term: 

404 - is a list or tuple 

405 - with 3 elements 

406 - second element if a valid op 

407 

408 :param tuple element: a leaf in form (left, operator, right) 

409 :param boolean internal: allow or not the 'inselect' internal operator 

410 in the term. This should be always left to False. 

411 

412 Note: OLD TODO change the share wizard to use this function. 

413 """ 

414 INTERNAL_OPS = TERM_OPERATORS + ('<>',) 

415 if internal: 

416 INTERNAL_OPS += ('inselect', 'not inselect') 

417 return (isinstance(element, tuple) or isinstance(element, list)) \ 

418 and len(element) == 3 \ 

419 and element[1] in INTERNAL_OPS \ 

420 and ((isinstance(element[0], basestring) and element[0]) 

421 or tuple(element) in (TRUE_LEAF, FALSE_LEAF))

458class ExtendedLeaf(object): 

459 """ Class wrapping a domain leaf, and giving some services and management 

460 features on it. In particular it managed join contexts to be able to 

461 construct queries through multiple models. 

462 """

464 # -------------------------------------------------- 

465 # Join / Context manipulation 

466 # running examples: 

467 # - res_users.name, like, foo: name is on res_partner, not on res_users 

468 # - res_partner.bank_ids.name, like, foo: bank_ids is a one2many with _auto_join 

469 # - res_partner.state_id.name, like, foo: state_id is a many2one with _auto_join 

470 # A join: 

471 # - link between src_table and dst_table, using src_field and dst_field 

472 # i.e.: inherits: res_users.partner_id = res_partner.id 

473 # i.e.: one2many: res_partner.id = res_partner_bank.partner_id 

474 # i.e.: many2one: res_partner.state_id = res_country_state.id 

475 # - done in the context of a field 

476 # i.e.: inherits: 'partner_id' 

477 # i.e.: one2many: 'bank_ids' 

478 # i.e.: many2one: 'state_id' 

479 # - table names use aliases: initial table followed by the context field 

480 # names, joined using a '__' 

481 # i.e.: inherits: res_partner as res_users__partner_id 

482 # i.e.: one2many: res_partner_bank as res_partner__bank_ids 

483 # i.e.: many2one: res_country_state as res_partner__state_id 

484 # - join condition use aliases 

485 # i.e.: inherits: res_users.partner_id = res_users__partner_id.id 

486 # i.e.: one2many: res_partner.id = res_partner__bank_ids.parr_id 

487 # i.e.: many2one: res_partner.state_id = res_partner__state_id.id 

488 # Variables explanation: 

489 # - src_table: working table before the join 

490 # -> res_users, res_partner, res_partner 

491 # - dst_table: working table after the join 

492 # -> res_partner, res_partner_bank, res_country_state 

493 # - src_table_link_name: field name used to link the src table, not 

494 # necessarily a field (because 'id' is not a field instance) 

495 # i.e.: inherits: 'partner_id', found in the inherits of the current table 

496 # i.e.: one2many: 'id', not a field 

497 # i.e.: many2one: 'state_id', the current field name 

498 # - dst_table_link_name: field name used to link the dst table, not 

499 # necessarily a field (because 'id' is not a field instance) 

500 # i.e.: inherits: 'id', not a field 

501 # i.e.: one2many: 'partner_id', _fields_id of the current field 

502 # i.e.: many2one: 'id', not a field 

503 # - context_field_name: field name used as a context to make the alias 

504 # i.e.: inherits: 'partner_id': found in the inherits of the current table 

505 # i.e.: one2many: 'bank_ids': current field name 

506 # i.e.: many2one: 'state_id': current field name

509 def __init__(self, leaf, model, join_context=None, internal=False): 

510 """ Initialize the ExtendedLeaf 

511 

512 :attr [string, tuple] leaf: operator or tuple-formatted domain 

513 expression 

514 :attr obj model: current working model 

515 :attr list _models: list of chained models, updated when 

516 adding joins 

517 :attr list join_context: list of join contexts. This is a list of 

518 tuples like ``(lhs, table, lhs_col, col, link)`` 

519 

520 where 

521 

522 lhs 

523 source (left hand) model 

524 model 

525 destination (right hand) model 

526 lhs_col 

527 source model column for join condition 

528 col 

529 destination model column for join condition 

530 link 

531 link column between source and destination model 

532 that is not necessarily (but generally) a real column used 

533 in the condition (i.e. in many2one); this link is used to 

534 compute aliases 

535 """ 

536 assert isinstance(model, BaseModel), 'Invalid leaf creation without table' 

537 self.join_context = join_context or [] 

538 self.leaf = leaf 

539 # normalize the leaf's operator 

540 self.normalize_leaf() 

541 # set working variables; handle the context stack and previous tables 

542 self.model = model 

543 self._models = [] 

544 for item in self.join_context: 

545 self._models.append(item[0]) 

546 self._models.append(model) 

547 # check validity 

548 self.check_leaf(internal)

568 def fetch_join_contexts(self, contexts): 

569 """ Collect the join contexts into the contexts dictionary in a format 

570 compatible with osv.query.Query """ 

571 alias = self._models[0]._table 

572 for context in self.join_context: 

573 previous_alias = alias 

574 alias = truncate_alias (alias + '__' + context[4]) 

575 table_joins = contexts.setdefault(previous_alias, [])

634class expression(object): 

635 """ Parse a domain expression 

636 Use a real polish notation 

637 Leafs are still in a ('foo', '=', 'bar') format 

638 For more info: http://christophe-simonis-at-tiny.blogspot.com/2008/08/new-new-domain-notation.html 

639 """ 

640 

641 def __init__(self, domain, model): 

642 """ Initialize expression object and automatically parse the expression 

643 right after initialization. 

644 

645 :param domain: expression (using domain ('foo', '=', 'bar' format)) 

646 :param model: root model 

647 

648 :attr list result: list that will hold the result of the parsing 

649 as a list of ExtendedLeaf 

650 :attr list joins: list of join conditions, such as 

651 (res_country_state."id" = res_partner."state_id") 

652 :attr root_model: base model for the query 

653 :attr list expression: the domain expression, that will be normalized 

654 and prepared 

655 """ 

656 self._unaccent = get_unaccent_wrapper(model._cr) 

657 self.joins = {} 

658 self.root_model = model 

659 

660 # normalize and prepare the expression for parsing 

661 self.expression = distribute_not(normalize_domain(domain)) 

662 

663 # parse the domain expression 

664 self.parse()

686 def parse(self): 

687 """ Transform the leaves of the expression 

688 

689 The principle is to pop elements from a leaf stack one at a time. 

690 Each leaf is processed. The processing is a if/elif list of various 

691 cases that appear in the leafs (many2one, function fields, ...). 

692 Two things can happen as a processing result: 

693 - the leaf has been modified and/or new leafs have to be introduced 

694 in the expression; they are pushed into the leaf stack, to be 

695 processed right after 

696 - the leaf is added to the result 

697 

698 Some internal var explanation: 

699 :var list path: left operand seen as a sequence of field names 

700 ("foo.bar" -> ["foo", "bar"]) 

701 :var obj model: model object, model containing the field 

702 (the name provided in the left operand) 

703 :var obj field: the field corresponding to `path[0]` 

704 :var obj column: the column corresponding to `path[0]` 

705 :var obj comodel: relational model of field (field.comodel) 

706 (res_partner.bank_ids -> res.partner.bank) 

707 """ 

708 cr, uid, context = self.root_model.env.args 

709 

710 def to_ids(value, comodel): 

711 """ Normalize a single id or name, or a list of those, into a list of ids 

712 :param {int,long,basestring,list,tuple} value: 

713 if int, long -> return [value] 

714 if basestring, convert it into a list of basestrings, then 

715 if list of basestring -> 

716 perform a name_search on comodel for each name 

717 return the list of related ids 

718 """ 

719 names = [] 

720 if isinstance(value, basestring): 

721 names = [value] 

722 elif value and isinstance(value, (tuple, list)) and all(isinstance(item, basestring) for item in value): 

723 names = value 

724 elif isinstance(value, (int, long)): 

725 return [value] 

726 if names: 

727 return list({ 

728 rid 

729 for name in names 

730 for rid, rname in comodel.name_search(name, [], 'ilike', limit=None) 

731 }) 

732 return list(value)

734 def child_of_domain(left, ids, left_model, parent=None, prefix=''): 

735 """ Return a domain implementing the child_of operator for [(left,child_of,ids)], 

736 either as a range using the parent_left/right tree lookup fields 

737 (when available), or as an expanded [(left,in,child_ids)] """ 

738 if not ids: 

739 return FALSE_DOMAIN 

740 if left_model._parent_store and (not left_model.pool._init) and (not context.get('defer_parent_store_computation')): 

741 # TODO: Improve where joins are implemented for many with '.', replace by: 

742 # doms += ['&',(prefix+'.parent_left','<',rec.parent_right),(prefix+'.parent_left','>=',rec.parent_left)] 

743 doms = [] 

744 for rec in left_model.browse(ids): 

745 if doms: 

746 doms.insert(0, OR_OPERATOR) 

747 doms += [AND_OPERATOR, ('parent_left', '<', rec.parent_right), ('parent_left', '>=', rec.parent_left)] 

748 if prefix: 

749 return [(left, 'in', left_model.search(doms).ids)] 

750 return doms 

751 else: 

752 parent_name = parent or left_model._parent_name 

753 child_ids = set(ids) 

754 while ids: 

755 ids = left_model.search([(parent_name, 'in', ids)]).ids 

756 child_ids.update(ids) 

757 return [(left, 'in', list(child_ids))] 

758 

759 def parent_of_domain(left, ids, left_model, parent=None, prefix=''): 

760 """ Return a domain implementing the parent_of operator for [(left,parent_of,ids)], 

761 either as a range using the parent_left/right tree lookup fields 

762 (when available), or as an expanded [(left,in,parent_ids)] """ 

763 if left_model._parent_store and (not left_model.pool._init) and (not context.get('defer_parent_store_computation')): 

764 doms = [] 

765 for rec in left_model.browse(ids): 

766 if doms: 

767 doms.insert(0, OR_OPERATOR) 

768 doms += [AND_OPERATOR, ('parent_right', '>', rec.parent_left), ('parent_left', '<=', rec.parent_left)] 

769 if prefix: 

770 return [(left, 'in', left_model.search(doms).ids)] 

771 return doms 

772 else: 

773 parent_name = parent or left_model._parent_name 

774 parent_ids = set() 

775 for record in left_model.browse(ids): 

776 while record: 

777 parent_ids.add(record.id) 

778 record = record[parent_name] 

779 return [(left, 'in', list(parent_ids))]

781 HIERARCHY_FUNCS = {'child_of': child_of_domain, 

782 'parent_of': parent_of_domain}

784 def pop(): 

785 """ Pop a leaf to process. """ 

786 return self.stack.pop() 

787 

788 def push(leaf): 

789 """ Push a leaf to be processed right after. """ 

790 self.stack.append(leaf) 

791 

792 def push_result(leaf): 

793 """ Push a leaf to the results. This leaf has been fully processed 

794 and validated. """ 

795 self.result.append(leaf) 

796 

797 def get_not_null_leafs(leaf, column, operator): 

798 """ Fetch not null leafs in the current join context. Such leafs 

799 need to be pushed after pusing an auto_join leaf with a negative 

800 operator. """ 

801 if operator in NEGATIVE_TERM_OPERATORS: 

802 return [create_substitution_leaf(leaf, (column, '!=', False)), 

803 create_substitution_leaf(leaf, AND_OPERATOR)] 

804 return [] 

805 

806 self.result = [] 

807 self.stack = [ExtendedLeaf(leaf, self.root_model) for leaf in self.expression] 

808 # process from right to left; expression is from left to right 

809 self.stack.reverse() 

810 

811 while self.stack: 

812 # Get the next leaf to process 

813 leaf = pop() 

814 

815 # Get working variables 

816 if leaf.is_operator(): 

817 left, operator, right = leaf.leaf, None, None 

818 elif leaf.is_true_leaf() or leaf.is_false_leaf(): 

819 # because we consider left as a string 

820 left, operator, right = ('%s' % leaf.leaf[0], leaf.leaf[1], leaf.leaf[2]) 

821 else: 

822 left, operator, right = leaf.leaf 

823 path = left.split('.', 1) 

824 

825 model = leaf.model 

826 field = model._fields.get(path[0]) 

827 comodel = model.env.get(getattr(field, 'comodel_name', None)) 

828 

829 # ---------------------------------------- 

830 # SIMPLE CASE 

831 # 1. leaf is an operator 

832 # 2. leaf is a true/false leaf 

833 # -> add directly to result 

834 # ---------------------------------------- 

835 

836 if leaf.is_operator() or leaf.is_true_leaf() or leaf.is_false_leaf(): 

837 push_result(leaf) 

838 

839 # ---------------------------------------- 

840 # FIELD NOT FOUND 

841 # -> from inherits'd fields -> work on the related model, and add 

842 # a join condition 

843 # -> ('id', 'child_of', '..') -> use a 'to_ids' 

844 # -> but is one on the _log_access special fields, add directly to 

845 # result 

846 # TODO: make these fields explicitly available in self.columns instead! 

847 # -> else: crash 

848 # ---------------------------------------- 

849 

850 elif not field: 

851 raise ValueError("Invalid field %r in leaf %r" % (left, str(leaf))) 

852 

853 elif field.inherited: 

854 # comments about inherits'd fields 

855 # { 'field_name': ('parent_model', 'm2o_field_to_reach_parent', 

856 # field_column_obj, origina_parent_model), ... } 

857 parent_model = model.env[field.related_field.model_name] 

858 parent_fname = model._inherits[parent_model._name] 

859 leaf.add_join_context(parent_model, parent_fname, 'id', parent_fname) 

860 if model._fields[parent_fname].auto_join: 

861 push(create_substitution_leaf(leaf, (left, operator, right), parent_model)) 

862 else: 

863 push(leaf) 

864 

865 elif left == 'id' and operator in HIERARCHY_FUNCS: 

866 ids2 = to_ids(right, model) 

867 dom = HIERARCHY_FUNCS[operator](left, ids2, model) 

868 for dom_leaf in reversed(dom): 

869 new_leaf = create_substitution_leaf(leaf, dom_leaf, model) 

870 push(new_leaf) 

871 

872 # ---------------------------------------- 

873 # PATH SPOTTED 

874 # -> many2one or one2many with _auto_join: 

875 # - add a join, then jump into linked column: column.remaining on 

876 # src_table is replaced by remaining on dst_table, and set for re-evaluation 

877 # - if a domain is defined on the column, add it into evaluation 

878 # on the relational table 

879 # - we want the left join to allow for sibling clauses to select records 

880 # without the field set, but under the scope of this clause we can add the 

881 # implicit column != False so that results in Odoo 8.0 don't change. In the 

882 # case of one2many columns, this causes a separate query for the subselection 

883 # of ids of the kind that we want to avoid by applying auto_join, but it is 

884 # only needed in the case of a negative search operator. 

885 # These semantics are to be reconsidered in master anyway because of analogy 

886 # with filtering on recordsets where (res.partner,).filtered( 

887 # lambda l: l.user_id.login != 'admin') yields partners without user_id set. 

888 # -> many2one, many2many, one2many: replace by an equivalent computed 

889 # domain, given by recursively searching on the remaining of the path 

890 # -> note: hack about columns.property should not be necessary anymore 

891 # as after transforming the column, it will go through this loop once again 

892 # ---------------------------------------- 

893 

894 elif len(path) > 1 and field.store and field.type == 'many2one' and field.auto_join: 

895 # res_partner.state_id = res_partner__state_id.id 

896 not_null_leafs = get_not_null_leafs(leaf, path[0], operator) 

897 leaf.add_join_context(comodel, path[0], 'id', path[0], 'LEFT JOIN') 

898 push(create_substitution_leaf(leaf, (path[1], operator, right), comodel)) 

899 for not_null_leaf in not_null_leafs: 

900 push(not_null_leaf) 

901 

902 elif len(path) > 1 and field.store and field.type == 'one2many' and field.auto_join: 

903 # res_partner.id = res_partner__bank_ids.partner_id 

904 not_null_leafs = get_not_null_leafs(leaf, path[0], operator) 

905 inverse_name = field.inverse_name 

906 inverse = comodel._fields[field.inverse_name] 

907 while inverse.inherited: 

908 # user_id.employee_ids.user_id: 

909 # res_users LEFT JOIN resource_resource as res_users__resource_id 

910 # ON res_users.id = res_users__resource_id.user_id 

911 # LEFT JOIN hr_employee as res_users__resource_id__employee_ids 

912 # ON res_users__resource_id.id = res_users__resource_id__employee_ids.resource_id 

913 parent_model = model.env[inverse.related_field.model_name] 

914 parent_fname = comodel._inherits[parent_model._name] 

915 leaf.add_join_context(parent_model, 'id', inverse_name, parent_fname, 'LEFT JOIN') 

916 parent_model._fields[inverse_name] 

917 inverse = parent_model._fields[inverse_name] 

918 inverse_name = parent_fname 

919 leaf.add_join_context(comodel, 'id', inverse_name, path[0], 'LEFT JOIN') 

920 domain = field.domain(model) if callable(field.domain) else field.domain 

921 push(create_substitution_leaf(leaf, (path[1], operator, right), comodel)) 

922 if domain: 

923 domain = normalize_domain(domain) 

924 for elem in reversed(domain): 

925 push(create_substitution_leaf(leaf, elem, comodel)) 

926 push(create_substitution_leaf(leaf, AND_OPERATOR, comodel)) 

927 for not_null_leaf in not_null_leafs: 

928 push(not_null_leaf) 

929 

930 elif len(path) > 1 and field.store and field.auto_join: 

931 raise NotImplementedError('auto_join attribute not supported on field %s' % field) 

932 

933 elif len(path) > 1 and field.store and field.type == 'many2one': 

934 right_ids = comodel.with_context(active_test=False).search([('.'.join(path[1:]), operator, right)]).ids 

935 leaf.leaf = (path[0], 'in', right_ids) 

936 push(leaf) 

937 

938 # Making search easier when there is a left operand as one2many or many2many 

939 elif len(path) > 1 and field.store and field.type in ('many2many', 'one2many'): 

940 right_ids = comodel.search([('.'.join(path[1:]), operator, right)]).ids 

941 leaf.leaf = (path[0], 'in', right_ids) 

942 push(leaf) 

943 

944 elif not field.store: 

945 # Non-stored field should provide an implementation of search. 

946 if not field.search: 

947 # field does not support search! 

948 _logger.error("Non-stored field %s cannot be searched.", field) 

949 if _logger.isEnabledFor(logging.DEBUG): 

950 _logger.debug(''.join(traceback.format_stack())) 

951 # Ignore it: generate a dummy leaf. 

952 domain = [] 

953 else: 

954 # Let the field generate a domain. 

955 if len(path) > 1: 

956 right = comodel.search([('.'.join(path[1:]), operator, right)]).ids 

957 operator = 'in' 

958 domain = field.determine_domain(model, operator, right) 

959 

960 if not domain: 

961 leaf.leaf = TRUE_LEAF 

962 push(leaf) 

963 else: 

964 for elem in reversed(domain): 

965 push(create_substitution_leaf(leaf, elem, model, internal=True)) 

966 

967 # ------------------------------------------------- 

968 # RELATIONAL FIELDS 

969 # ------------------------------------------------- 

970 

971 # Applying recursivity on field(one2many) 

972 elif field.type == 'one2many' and operator in HIERARCHY_FUNCS: 

973 ids2 = to_ids(right, comodel) 

974 if field.comodel_name != model._name: 

975 dom = HIERARCHY_FUNCS[operator](left, ids2, comodel, prefix=field.comodel_name) 

976 else: 

977 dom = HIERARCHY_FUNCS[operator]('id', ids2, model, parent=left)

980

1088 

1089 elif field.type == 'many2one': 

1090 if operator in HIERARCHY_FUNCS: 

1091 ids2 = to_ids(right, comodel) 

1092 if field.comodel_name != model._name: 

1093 dom = HIERARCHY_FUNCS[operator](left, ids2, comodel, prefix=field.comodel_name) 

1094 else: 

1095 dom = HIERARCHY_FUNCS[operator]('id', ids2, model, parent=left) 

1096 for dom_leaf in reversed(dom): 

1097 push(create_substitution_leaf(leaf, dom_leaf, model))

1099 def _get_expression(comodel, left, right, operator): 

1100 #Special treatment to ill-formed domains 

1101 operator = (operator in ['<', '>', '<=', '>=']) and 'in' or operator 

1102 

1103 dict_op = {'not in': '!=', 'in': '=', '=': 'in', '!=': 'not in'} 

1104 if isinstance(right, tuple): 

1105 right = list(right) 

1106 if (not isinstance(right, list)) and operator in ['not in', 'in']: 

1107 operator = dict_op[operator] 

1108 elif isinstance(right, list) and operator in ['!=', '=']: # for domain (FIELD,'=',['value1','value2']) 

1109 operator = dict_op[operator] 

1110 res_ids = [x[0] for x in comodel.with_context(active_test=False).name_search(right, [], operator, limit=None)] 

1111 if operator in NEGATIVE_TERM_OPERATORS: 

1112 res_ids.append(False) # TODO this should not be appended if False was in 'right' 

1113 return left, 'in', res_ids 

1114 # resolve string-based m2o criterion into IDs 

1115 if isinstance(right, basestring) or \ 

1116 right and isinstance(right, (tuple, list)) and all(isinstance(item, basestring) for item in right): 

1117 push(create_substitution_leaf(leaf, _get_expression(comodel, left, right, operator), model)) 

1118 else: 

1119 # right == [] or right == False and all other cases are handled by __leaf_to_sql() 

1120 push_result(leaf)

1124 # -> check for null only 

1125 # -------------------------------------------------

1128 if operator in ('=', '!=') and not right: 

1129 inselect_operator = 'inselect' if operator in NEGATIVE_TERM_OPERATORS else 'not inselect' 

1130 subselect = "SELECT res_id FROM ir_attachment WHERE res_model=%s AND res_field=%s" 

1131 params = (model._name, left) 

1132 push(create_substitution_leaf(leaf, ('id', inselect_operator, (subselect, params)), model, internal=True)) 

1133 else: 

1134 _logger.error("Binary field '%s' stored in attachment: ignore %s %s %s", 

1135 field.string, left, operator, right)

1137 push(leaf)

1144 # ------------------------------------------------- 

1145 

1146 else: 

1147 if field.type == 'datetime' and right and len(right) == 10: 

1148 if operator in ('>', '<='): 

1149 right += ' 23:59:59' 

1150 else: 

1151 right += ' 00:00:00' 

1152 push(create_substitution_leaf(leaf, (left, operator, right), model)) 

1153 

1154 elif field.translate is True and right: 

1155 need_wildcard = operator in ('like', 'ilike', 'not like', 'not ilike') 

1156 sql_operator = {'=like': 'like', '=ilike': 'ilike'}.get(operator, operator) 

1157 if need_wildcard: 

1158 right = '%%%s%%' % right 

1159 

1160 inselect_operator = 'inselect' 

1161 if sql_operator in NEGATIVE_TERM_OPERATORS: 

1162 # negate operator (fix lp:1071710) 

1163 sql_operator = sql_operator[4:] if sql_operator[:3] == 'not' else '=' 

1164 inselect_operator = 'not inselect'

1172 instr = '(%s)' % ', '.join(['%s'] * len(right))

1175 SELECT ct.id, coalesce(it.value,ct.{quote_left})

1207 def __leaf_to_sql(self, eleaf): 

1208 model = eleaf.model 

1209 leaf = eleaf.leaf 

1210 left, operator, right = leaf 

1211 

1212 # final sanity checks - should never fail 

1213 assert operator in (TERM_OPERATORS + ('inselect', 'not inselect')), \ 

1214 "Invalid operator %r in domain term %r" % (operator, leaf) 

1215 assert leaf in (TRUE_LEAF, FALSE_LEAF) or left in model._fields, \ 

1216 "Invalid field %r in domain term %r" % (left, leaf) 

1217 assert not isinstance(right, BaseModel), \ 

1218 "Invalid value %r in domain term %r" % (right, leaf) 

1219 

1220 table_alias = '"%s"' % (eleaf.generate_alias()) 

1221 

1222 if leaf == TRUE_LEAF: 

1223 query = 'TRUE' 

1224 params = [] 

1225 

1226 elif leaf == FALSE_LEAF: 

1227 query = 'FALSE' 

1228 params = [] 

1229 

1230 elif operator == 'inselect': 

1231 query = '(%s."%s" in (%s))' % (table_alias, left, right[0]) 

1232 params = right[1] 

1233 

1234 elif operator == 'not inselect': 

1235 query = '(%s."%s" not in (%s))' % (table_alias, left, right[0]) 

1236 params = right[1] 

1237 

1238 elif operator in ['in', 'not in']: 

1239 # Two cases: right is a boolean or a list. The boolean case is an 

1240 # abuse and handled for backward compatibility. 

1241 if isinstance(right, bool): 

1242 _logger.warning("The domain term '%s' should use the '=' or '!=' operator." % (leaf,)) 

1243 if operator == 'in': 

1244 r = 'NOT NULL' if right else 'NULL' 

1245 else: 

1246 r = 'NULL' if right else 'NOT NULL' 

1247 query = '(%s."%s" IS %s)' % (table_alias, left, r) 

1248 params = [] 

1249 elif isinstance(right, (list, tuple)): 

1250 params = list(right) 

1251 check_nulls = False 

1252 for i in range(len(params))[::-1]: 

1253 if params[i] == False: 

1254 check_nulls = True 

1255 del params[i] 

1256 

1257 if params: 

1258 if left == 'id': 

1259 instr = ','.join(['%s'] * len(params)) 

1260 else: 

1261 field = model._fields[left] 

1262 instr = ','.join([field.column_format] * len(params)) 

1263 params = map(partial(field.convert_to_column, record=model), params) 

1264 query = '(%s."%s" %s (%s))' % (table_alias, left, operator, instr) 

1265 else: 

1266 # The case for (left, 'in', []) or (left, 'not in', []). 

1267 query = 'FALSE' if operator == 'in' else 'TRUE' 

1268 

1269 if check_nulls and operator == 'in': 

1270 query = '(%s OR %s."%s" IS NULL)' % (query, table_alias, left) 

1271 elif not check_nulls and operator == 'not in': 

1272 query = '(%s OR %s."%s" IS NULL)' % (query, table_alias, left) 

1273 elif check_nulls and operator == 'not in': 

1274 query = '(%s AND %s."%s" IS NOT NULL)' % (query, table_alias, left) # needed only for TRUE. 

1275 else: # Must not happen 

1276 raise ValueError("Invalid domain term %r" % (leaf,)) 

1277 

1278 elif left in model and model._fields[left].type == "boolean" and ((operator == '=' and right is False) or (operator == '!=' and right is True)): 

1279 query = '(%s."%s" IS NULL or %s."%s" = false )' % (table_alias, left, table_alias, left) 

1280 params = [] 

1281 

1282 elif (right is False or right is None) and (operator == '='): 

1283 query = '%s."%s" IS NULL ' % (table_alias, left) 

1284 params = [] 

1285 

1286 elif left in model and model._fields[left].type == "boolean" and ((operator == '!=' and right is False) or (operator == '==' and right is True)): 

1287 query = '(%s."%s" IS NOT NULL and %s."%s" != false)' % (table_alias, left, table_alias, left) 

1288 params = [] 

1289 

1290 elif (right is False or right is None) and (operator == '!='): 

1291 query = '%s."%s" IS NOT NULL' % (table_alias, left) 

1292 params = [] 

1293 

1294 elif operator == '=?': 

1295 if right is False or right is None: 

1296 # '=?' is a short-circuit that makes the term TRUE if right is None or False 

1297 query = 'TRUE' 

1298 params = [] 

1299 else: 

1300 # '=?' behaves like '=' in other cases 

1301 query, params = self.__leaf_to_sql( 

1302 create_substitution_leaf(eleaf, (left, '=', right), model)) 

1303 

1304 elif left == 'id': 

1305 query = '%s.id %s %%s' % (table_alias, operator) 

1306 params = right 

1307 

1308 else: 

1309 need_wildcard = operator in ('like', 'ilike', 'not like', 'not ilike') 

1310 sql_operator = {'=like': 'like', '=ilike': 'ilike'}.get(operator, operator) 

1311 cast = '::text' if sql_operator.endswith('like') else '' 

1312 

1313 if left in model: 

1314 format = need_wildcard and '%s' or model._fields[left].column_format 

1315 unaccent = self._unaccent if sql_operator.endswith('like') else lambda x: x 

1316 column = '%s.%s' % (table_alias, _quote(left)) 

1317 query = '(%s %s %s)' % (unaccent(column + cast), sql_operator, unaccent(format)) 

1318 elif left in MAGIC_COLUMNS: 

1319 query = "(%s.\"%s\"%s %s %%s)" % (table_alias, left, cast, sql_operator) 

1320 params = right 

1321 else: # Must not happen 

1322 raise ValueError("Invalid field %r in domain term %r" % (left, leaf)) 

1323 

1324 add_null = False 

1325 if need_wildcard: 

1326 if isinstance(right, str): 

1327 str_utf8 = right 

1328 elif isinstance(right, unicode): 

1329 str_utf8 = right.encode('utf-8') 

1330 else: 

1331 str_utf8 = str(right) 

1332 params = '%%%s%%' % str_utf8 

1333 add_null = not str_utf8 

1334 elif left in model: 

1335 params = model._fields[left].convert_to_column(right, model) 

1336 

1337 if add_null: 

1338 query = '(%s OR %s."%s" IS NULL)' % (query, table_alias, left) 

1339 

1340 if isinstance(params, basestring): 

1341 params = [params] 

1342 return query, params

1344 def to_sql(self): 

1345 stack = [] 

1346 params = [] 

1347 # Process the domain from right to left, using a stack, to generate a SQL expression. 

1348 self.result.reverse() 

1349 for leaf in self.result: 

1350 if leaf.is_leaf(internal=True): 

1351 q, p = self.__leaf_to_sql(leaf) 

1352 params.insert(0, p) 

1353 stack.append(q) 

1354 elif leaf.leaf == NOT_OPERATOR: 

1355 stack.append('(NOT (%s))' % (stack.pop(),)) 

1356 else: 

1357 ops = {AND_OPERATOR: ' AND ', OR_OPERATOR: ' OR '} 

1358 q1 = stack.pop() 

1359 q2 = stack.pop() 

1360 stack.append('(%s %s %s)' % (q1, ops[leaf.leaf], q2,)) 

1361 

1362 assert len(stack) == 1 

1363 query = stack[0] 

1364 

1365 return query, tools.flatten(params)
