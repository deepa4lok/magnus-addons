106 def _sub_select(self): 

107 select_str = """ 

108 SELECT ail.id AS id, 

109 ai.date_invoice AS date, 

110 ail.product_id, ai.partner_id, ai.payment_term_id, ail.account_analytic_id, 

111 u2.name AS uom_name, 

112 ai.currency_id, ai.journal_id, ai.fiscal_position_id, ai.user_id, ai.company_id, 

113 1 AS nbr, 

114 ai.type, ai.state, pt.categ_id, ai.date_due, ai.account_id, ail.account_id AS account_line_id, 

115 ai.partner_bank_id, 

116 SUM ((invoice_type.sign_qty * ail.quantity) / u.factor * u2.factor) AS product_qty, 

117 SUM(ail.price_subtotal_signed * invoice_type.sign) AS price_total, 

118 SUM(ABS(ail.price_subtotal_signed)) / CASE 

119 WHEN SUM(ail.quantity / u.factor * u2.factor) <> 0::numeric 

120 THEN SUM(ail.quantity / u.factor * u2.factor) 

121 ELSE 1::numeric 

122 END AS price_average, 

123 ai.residual_company_signed / (SELECT count(*) FROM account_invoice_line l where invoice_id = ai.id) * 

124 count(*) * invoice_type.sign AS residual, 

125 ai.commercial_partner_id as commercial_partner_id, 

126 partner.country_id, 

127 SUM(pr.weight * (invoice_type.sign*ail.quantity) / u.factor * u2.factor) AS weight, 

128 SUM(pr.volume * (invoice_type.sign*ail.quantity) / u.factor * u2.factor) AS volume 

129 """ 

130 return select_str

132 def _from(self): 

133 from_str = """ 

134 FROM account_invoice_line ail 

135 JOIN account_invoice ai ON ai.id = ail.invoice_id 

136 JOIN res_partner partner ON ai.commercial_partner_id = partner.id 

137 LEFT JOIN product_product pr ON pr.id = ail.product_id 

138 left JOIN product_template pt ON pt.id = pr.product_tmpl_id 

139 LEFT JOIN product_uom u ON u.id = ail.uom_id 

140 LEFT JOIN product_uom u2 ON u2.id = pt.uom_id 

141 JOIN ( 

142 -- Temporary table to decide if the qty should be added or retrieved (Invoice vs Refund)  

143 SELECT id,(CASE 

144 WHEN ai.type::text = ANY (ARRAY['in_refund'::character varying::text, 'in_invoice'::character varying::text]) 

145 THEN -1 

146 ELSE 1 

147 END) AS sign,(CASE 

148 WHEN ai.type::text = ANY (ARRAY['out_refund'::character varying::text, 'in_invoice'::character varying::text]) 

149 THEN -1 

150 ELSE 1 

151 END) AS sign_qty 

152 FROM account_invoice ai 

153 ) AS invoice_type ON invoice_type.id = ai.id 

154 """
