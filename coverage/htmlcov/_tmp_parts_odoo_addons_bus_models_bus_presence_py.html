14class BusPresence(models.Model): 

15 """ User Presence 

16 Its status is 'online', 'away' or 'offline'. This model should be a one2one, but is not 

17 attached to res_users to avoid database concurrence errors. Since the 'update' method is executed 

18 at each poll, if the user have multiple opened tabs, concurrence errors can happend, but are 'muted-logged'. 

19 """ 

20 

21 _name = 'bus.presence' 

22 _description = 'User Presence' 

23 _log_access = False 

24 

25 _sql_constraints = [('bus_user_presence_unique', 'unique(user_id)', 'A user can only have one IM status.')] 

26 

27 user_id = fields.Many2one('res.users', 'Users', required=True, index=True, ondelete='cascade') 

28 last_poll = fields.Datetime('Last Poll', default=lambda self: fields.Datetime.now()) 

29 last_presence = fields.Datetime('Last Presence', default=lambda self: fields.Datetime.now()) 

30 status = fields.Selection([('online', 'Online'), ('away', 'Away'), ('offline', 'Offline')], 'IM Status', default='offline') 

31 

32 @api.model 

33 def update(self, inactivity_period): 

34 """ Updates the last_poll and last_presence of the current user 

35 :param inactivity_period: duration in milliseconds 

36 """ 

37 presence = self.search([('user_id', '=', self._uid)], limit=1) 

38 # compute last_presence timestamp 

39 last_presence = datetime.datetime.now() - datetime.timedelta(milliseconds=inactivity_period) 

40 values = { 

41 'last_poll': time.strftime(DEFAULT_SERVER_DATETIME_FORMAT), 

42 } 

43 # update the presence or a create a new one 

44 if not presence: # create a new presence for the user 

45 values['user_id'] = self._uid 

46 values['last_presence'] = last_presence.strftime(DEFAULT_SERVER_DATETIME_FORMAT) 

47 self.create(values) 

48 else: # update the last_presence if necessary, and write values 

49 if datetime.datetime.strptime(presence.last_presence, DEFAULT_SERVER_DATETIME_FORMAT) < last_presence: 

50 values['last_presence'] = last_presence.strftime(DEFAULT_SERVER_DATETIME_FORMAT) 

51 # Hide transaction serialization errors, which can be ignored, the presence update is not essential 

52 with tools.mute_logger('odoo.sql_db'): 

53 presence.write(values) 

54 # avoid TransactionRollbackError 

55 self.env.cr.commit() # TODO : check if still necessary
