25 def _parse_node(self, root_node): 

26 result = [] 

27 for node in root_node: 

28 field_name = node.get('name') 

29 if not safe_eval(str(node.attrib.get('invisible',False)),{'context':self.context}): 

30 if node.tag == 'field': 

31 if field_name in self.groupby: 

32 continue 

33 result.append(field_name) 

34 else: 

35 result.extend(self._parse_node(node)) 

36 return result
