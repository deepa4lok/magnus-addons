222 @api.depends('log_contracts') 

223 def _compute_contract_reminder(self): 

224 for record in self: 

225 overdue = False 

226 due_soon = False 

227 total = 0 

228 name = '' 

229 for element in record.log_contracts: 

230 if element.state in ('open', 'toclose') and element.expiration_date: 

231 current_date_str = fields.Date.context_today(record) 

232 due_time_str = element.expiration_date 

233 current_date = fields.Date.from_string(current_date_str) 

234 due_time = fields.Date.from_string(due_time_str) 

235 diff_time = (due_time - current_date).days 

236 if diff_time < 0: 

237 overdue = True 

238 total += 1 

239 if diff_time < 15 and diff_time >= 0: 

240 due_soon = True 

241 total += 1 

242 if overdue or due_soon: 

243 log_contract = self.env['fleet.vehicle.log.contract'].search([('vehicle_id', '=', record.id), ('state', 'in', ('open', 'toclose'))], 

244 limit=1, order='expiration_date asc') 

245 if log_contract: 

246 #we display only the name of the oldest overdue/due soon contract 

247 name = log_contract.cost_subtype_id.name 

248 

249 record.contract_renewal_overdue = overdue 

250 record.contract_renewal_due_soon = due_soon 

251 record.contract_renewal_total = total - 1 # we remove 1 from the real total for display purposes 

252 record.contract_renewal_name = name

254 def _search_contract_renewal_due_soon(self, operator, value): 

255 res = [] 

256 assert operator in ('=', '!=', '<>') and value in (True, False), 'Operation not supported' 

257 if (operator == '=' and value is True) or (operator in ('<>', '!=') and value is False): 

258 search_operator = 'in' 

259 else: 

260 search_operator = 'not in' 

261 today = fields.Date.context_today(self) 

262 datetime_today = fields.Datetime.from_string(today) 

263 limit_date = fields.Datetime.to_string(datetime_today + relativedelta(days=+15)) 

264 self.env.cr.execute("""SELECT cost.vehicle_id, 

265 count(contract.id) AS contract_number 

266 FROM fleet_vehicle_cost cost 

267 LEFT JOIN fleet_vehicle_log_contract contract ON contract.cost_id = cost.id 

268 WHERE contract.expiration_date IS NOT NULL 

269 AND contract.expiration_date > %s 

270 AND contract.expiration_date < %s 

271 AND contract.state IN ('open', 'toclose') 

272 GROUP BY cost.vehicle_id""", (today, limit_date)) 

273 res_ids = [x[0] for x in self.env.cr.fetchall()] 

274 res.append(('id', search_operator, res_ids)) 

275 return res

310 @api.multi 

311 def write(self, vals): 

312 """ 

313 This function write an entry in the openchatter whenever we change important information 

314 on the vehicle like the model, the drive, the state of the vehicle or its license plate 

315 """ 

316 for vehicle in self: 

317 changes = [] 

318 if 'model_id' in vals and vehicle.model_id.id != vals['model_id']: 

319 value = self.env['fleet.vehicle.model'].browse(vals['model_id']).name 

320 oldmodel = vehicle.model_id.name or _('None') 

321 changes.append(_("Model: from '%s' to '%s'") % (oldmodel, value)) 

322 if 'driver_id' in vals and vehicle.driver_id.id != vals['driver_id']: 

323 value = self.env['res.partner'].browse(vals['driver_id']).name 

324 olddriver = (vehicle.driver_id.name) or _('None') 

325 changes.append(_("Driver: from '%s' to '%s'") % (olddriver, value)) 

326 if 'state_id' in vals and vehicle.state_id.id != vals['state_id']: 

327 value = self.env['fleet.vehicle.state'].browse(vals['state_id']).name 

328 oldstate = vehicle.state_id.name or _('None') 

329 changes.append(_("State: from '%s' to '%s'") % (oldstate, value)) 

330 if 'license_plate' in vals and vehicle.license_plate != vals['license_plate']: 

331 old_license_plate = vehicle.license_plate or _('None') 

332 changes.append(_("License Plate: from '%s' to '%s'") % (old_license_plate, vals['license_plate'])) 

333 

334 if len(changes) > 0: 

335 self.message_post(body=", ".join(changes))

423 @api.onchange('liter', 'price_per_liter', 'amount') 

424 def _onchange_liter_price_amount(self): 

425 #need to cast in float because the value receveid from web client maybe an integer (Javascript and JSON do not 

426 #make any difference between 3.0 and 3). This cause a problem if you encode, for example, 2 liters at 1.5 per 

427 #liter => total is computed as 3.0, then trigger an onchange that recomputes price_per_liter as 3/2=1 (instead 

428 #of 3.0/2=1.5) 

429 #If there is no change in the result, we return an empty dict to prevent an infinite loop due to the 3 intertwine 

430 #onchange. And in order to verify that there is no change in the result, we have to limit the precision of the 

431 #computation to 2 decimal 

432 liter = float(self.liter) 

433 price_per_liter = float(self.price_per_liter) 

434 amount = float(self.amount) 

435 if liter > 0 and price_per_liter > 0 and round(liter * price_per_liter, 2) != amount: 

436 self.amount = round(liter * price_per_liter, 2) 

437 elif amount > 0 and liter > 0 and round(amount / liter, 2) != price_per_liter: 

438 self.price_per_liter = round(amount / liter, 2) 

439 elif amount > 0 and price_per_liter > 0 and round(amount / price_per_liter, 2) != liter: 

440 self.liter = round(amount / price_per_liter, 2)

598 @api.model 

599 def scheduler_manage_auto_costs(self): 

600 #This method is called by a cron task 

601 #It creates costs for contracts having the "recurring cost" field setted, depending on their frequency 

602 #For example, if a contract has a reccuring cost of 200 with a weekly frequency, this method creates a cost of 200 on the first day of each week, from the date of the last recurring costs in the database to today 

603 #If the contract has not yet any recurring costs in the database, the method generates the recurring costs from the start_date to today 

604 #The created costs are associated to a contract thanks to the many2one field contract_id 

605 #If the contract has no start_date, no cost will be created, even if the contract has recurring costs 

606 VehicleCost = self.env['fleet.vehicle.cost'] 

607 deltas = {'yearly': relativedelta(years=+1), 'monthly': relativedelta(months=+1), 'weekly': relativedelta(weeks=+1), 'daily': relativedelta(days=+1)} 

608 contracts = self.env['fleet.vehicle.log.contract'].search([('state', '!=', 'closed')], offset=0, limit=None, order=None) 

609 for contract in contracts: 

610 if not contract.start_date or contract.cost_frequency == 'no': 

611 continue 

612 found = False 

613 last_cost_date = contract.start_date 

614 if contract.generated_cost_ids: 

615 last_autogenerated_cost = VehicleCost.search([('contract_id', '=', contract.id), ('auto_generated', '=', True)], offset=0, limit=1, order='date desc') 

616 if last_autogenerated_cost: 

617 found = True 

618 last_cost_date = last_autogenerated_cost.date 

619 startdate = fields.Date.from_string(last_cost_date) 

620 if found: 

621 startdate += deltas.get(contract.cost_frequency) 

622 today = fields.Date.from_string(fields.Date.context_today(self)) 

623 while (startdate <= today) & (startdate <= fields.Date.from_string(contract.expiration_date)): 

624 data = { 

625 'amount': contract.cost_generated, 

626 'date': fields.Date.context_today(self), 

627 'vehicle_id': contract.vehicle_id.id, 

628 'cost_subtype_id': contract.cost_subtype_id.id, 

629 'contract_id': contract.id, 

630 'auto_generated': True 

631 } 

632 self.env['fleet.vehicle.cost'].create(data) 

633 startdate += deltas.get(contract.cost_frequency) 

634 return True

636 @api.model 

637 def scheduler_manage_contract_expiration(self): 

638 #This method is called by a cron task 

639 #It manages the state of a contract, possibly by posting a message on the vehicle concerned and updating its status 

640 date_today = fields.Date.from_string(fields.Date.context_today(self)) 

641 limit_date = fields.Date.to_string(date_today + relativedelta(days=+15)) 

642 contracts = self.search([('state', '=', 'open'), ('expiration_date', '<', limit_date)]) 

643 res = {} 

644 for contract in contracts: 

645 if contract.vehicle_id.id in res: 

646 res[contract.vehicle_id.id] += 1 

647 else: 

648 res[contract.vehicle_id.id] = 1 

649 

650 Vehicle = self.env['fleet.vehicle'] 

651 for vehicle, value in res.items(): 

652 Vehicle.browse(vehicle).message_post(body=_('%s contract(s) need(s) to be renewed and/or closed!') % value) 

653 return contracts.write({'state': 'toclose'})
