41 def compute_target_invoice_amount(self): 

42 try: 

43 if self.amount_untaxed != self.target_invoice_amount: 

44 self.reset_target_invoice_amount() 

45 factor = self.target_invoice_amount / self.amount_untaxed 

46 discount = (1.0 - factor) * 100 

47 for line in self.invoice_line_ids: 

48 line.discount = discount 

49 taxes_grouped = self.get_taxes_values() 

50 tax_lines = self.tax_line_ids.filtered('manual') 

51 for tax in taxes_grouped.values(): 

52 tax_lines += tax_lines.new(tax) 

53 self.tax_line_ids = tax_lines 

54 except ZeroDivisionError: 

55 raise UserError(_('You cannot set a target amount if the invoice line amount is 0'))

89 @api.multi 

90 def _get_timesheet_by_group(self): 

91 self.ensure_one() 

92 aal_ids = [] 

93 analytic_invoice_ids = self.invoice_line_ids.mapped('analytic_invoice_id') 

94 for analytic_invoice in analytic_invoice_ids: 

95 for grp_line in analytic_invoice.user_total_ids: 

96 aal_ids += grp_line.detail_ids 

97 userProject = {} 

98 for aal in aal_ids: 

99 project_id, user_id = aal.project_id if aal.project_id else aal.task_id.project_id , aal.user_id 

100 if project_id.correction_charge and project_id.specs_invoice_report: 

101 if (project_id, user_id) in userProject: 

102 userProject[(project_id, user_id)] = userProject[(project_id, user_id)] + [aal] 

103 else: 

104 userProject[(project_id, user_id)] = [aal] 

105 return userProject

130 @api.multi 

131 def action_wip_move_create(self): 

132 """ Creates invoice related analytics and financial move lines """ 

133 for inv in self: 

134 wip_journal = self.env.ref('magnus_timesheet.wip_journal') 

135 if not wip_journal.sequence_id: 

136 raise UserError(_('Please define sequence on the type WIP journal.')) 

137 sequence = wip_journal.sequence_id 

138 if inv.type in ['out_refund', 'in_invoice','in_refund'] or inv.wip_move_id: 

139 continue 

140 date_end = inv.month_id.date_end 

141 new_name = sequence.with_context(ir_sequence_date=date_end).next_by_id() 

142 if inv.move_id: 

143 wip_move = inv.move_id.wip_move_create( wip_journal, new_name, inv.account_id.id, inv.number) 

144 wip_move.post() 

145 # make the invoice point to that wip move 

146 inv.wip_move_id = wip_move.id 

147 #wip reverse posting 

148 reverse_date = datetime.strptime(wip_move.date, "%Y-%m-%d") + timedelta(days=1) 

149 line_amt = sum(ml.credit + ml.debit for ml in wip_move.line_ids) 

150 reconcile = False 

151 if line_amt > 0: 

152 reconcile = True 

153 reverse_wip_move = wip_move.create_reversals( 

154 date=reverse_date, 

155 journal=wip_journal, 

156 move_prefix='WIP Invoicing Reverse', 

157 line_prefix='WIP Invoicing Reverse', 

158 reconcile=reconcile 

159 ) 

160 if len(reverse_wip_move) == 1: 

161 wip_nxt_seq = sequence.with_context(ir_sequence_date=reverse_wip_move.date).next_by_id() 

162 reverse_wip_move.write({'name':wip_nxt_seq}) 

163 

164 return True

166 @api.multi 

167 def action_cancel(self): 

168 res = super(AccountInvoice, self).action_cancel() 

169 wip_moves = self.env['account.move'] 

170 for inv in self: 

171 if inv.wip_move_id: 

172 wip_moves += inv.wip_move_id 

173 

174 # First, set the invoices as cancelled and detach the move ids 

175 self.write({'wip_move_id': False}) 

176 if wip_moves: 

177 # second, invalidate the move(s) 

178 wip_moves.button_cancel() 

179 # delete the move this invoice was pointing to 

180 # Note that the corresponding move_lines and move_reconciles 

181 # will be automatically deleted too 

182 wip_moves.unlink() 

183 return res
